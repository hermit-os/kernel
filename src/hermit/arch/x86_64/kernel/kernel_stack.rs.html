5type Reg = mem::MaybeUninit<usize>;

7#[unsafe(naked)]
8pub unsafe extern "C" fn call_with_stack(
9	rdi: Reg,
10	rsi: Reg,
11	rdx: Reg,
12	rcx: Reg,
13	r8: Reg,
14	r9: Reg,
15	f: unsafe extern "C" fn(rdi: Reg, rsi: Reg, rdx: Reg, rcx: Reg, r8: Reg, r9: Reg) -> Reg,
16	stack_ptr: *mut (),
17) -> Reg {
18	core::arch::naked_asm!(
19		// Save user stack pointer and switch to supplied stack
20		"cli",
21		"push r12",
22		"mov r12, rsp",
23		"mov rsp, [rsp + 24]",
24		"sti",
25		// Call f
26		"call [r12 + 16]",
27		// Switch back to previous stack
28		"cli",
29		"mov rsp, r12",
30		"pop r12",
31		"sti",
32		"ret",
33	)
34}

36macro_rules! kernel_function_impl {
37	($kernel_function:ident($($arg:ident: $A:ident),*; $($z:ident: Reg),*)) => {
38		/// Executes `f` on the kernel stack.
39		#[allow(dead_code)]
40		#[inline]
41		pub unsafe extern "C" fn $kernel_function<R, $($A),*>($($arg: $A,)* f: unsafe extern "C" fn($($A),*) -> R) -> R {
42			unsafe {
43				$(
44					assert!(mem::size_of::<$A>() <= mem::size_of::<Reg>());
45				)*
46				assert!(mem::size_of::<R>() <= mem::size_of::<Reg>());
47
48				let call_with_stack = mem::transmute::<*const (), unsafe extern "C" fn(
49					$($arg: $A,)*
50					$($z: Reg,)*
51					f: unsafe extern "C" fn(
52						$($arg: $A,)*
53					) -> R,
54					stack_ptr: *mut (),
55				) -> R>(call_with_stack as *const ());
56
57				$(
58					let $z = Reg::uninit();
59				)*
60
61				let kernel_stack = CoreLocal::get().kernel_stack.get().cast();
62
63				call_with_stack(
64					$($arg,)*
65					$($z,)*
66					f,
67					kernel_stack,
68				)
69			}
70		}
71	};
72}
