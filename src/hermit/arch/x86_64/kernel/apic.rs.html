260extern "x86-interrupt" fn tlb_flush_handler(stack_frame: interrupts::ExceptionStackFrame) {
261	swapgs(&stack_frame);
262	debug!("Received TLB Flush Interrupt");
263	increment_irq_counter(TLB_FLUSH_INTERRUPT_NUMBER);
264	let (frame, val) = Cr3::read_raw();
265	unsafe {
266		Cr3::write_raw(frame, val);
267	}
268	eoi();
269	swapgs(&stack_frame);
270}

272extern "x86-interrupt" fn error_interrupt_handler(stack_frame: interrupts::ExceptionStackFrame) {
273	swapgs(&stack_frame);
274	error!("APIC LVT Error Interrupt");
275	error!("ESR: {:#X}", local_apic_read(IA32_X2APIC_ESR));
276	error!("{stack_frame:#?}");
277	eoi();
278	scheduler::abort();
279}

287#[cfg(feature = "smp")]
288extern "x86-interrupt" fn wakeup_handler(stack_frame: interrupts::ExceptionStackFrame) {
289	swapgs(&stack_frame);
290	use crate::scheduler::PerCoreSchedulerExt;
291
292	debug!("Received Wakeup Interrupt");
293	increment_irq_counter(WAKEUP_INTERRUPT_NUMBER);
294	let core_scheduler = core_scheduler();
295	core_scheduler.check_input();
296	eoi();
297	if core_scheduler.is_scheduling() {
298		core_scheduler.reschedule();
299	}
300	swapgs(&stack_frame);
301}

313fn init_ioapic_address(phys_addr: PhysAddr) {
314	if env::is_uefi() {
315		// UEFI systems have already id mapped everything, so we can just set the physical address as the virtual one
316		IOAPIC_ADDRESS
317			.set(VirtAddr::new(phys_addr.as_u64()))
318			.unwrap();
319	} else {
320		let layout = PageLayout::from_size(BasePageSize::SIZE as usize).unwrap();
321		let page_range = PageAlloc::allocate(layout).unwrap();
322		let ioapic_address = VirtAddr::from(page_range.start());
323		IOAPIC_ADDRESS.set(ioapic_address).unwrap();
324		debug!("Mapping IOAPIC at {phys_addr:p} to virtual address {ioapic_address:p}");
325
326		let mut flags = PageTableEntryFlags::empty();
327		flags.device().writable().execute_disable();
328		paging::map::<BasePageSize>(ioapic_address, phys_addr, 1, flags);
329	}
330}

338#[cfg(feature = "acpi")]
339fn detect_from_acpi() -> Result<PhysAddr, ()> {
340	// Get the Multiple APIC Description Table (MADT) from the ACPI information and its specific table header.
341	let madt = acpi::get_madt().ok_or(())?;
342	let madt_header =
343		unsafe { &*(ptr::with_exposed_provenance::<AcpiMadtHeader>(madt.table_start_address())) };
344
345	// Jump to the actual table entries (after the table header).
346	let mut current_address = madt.table_start_address() + mem::size_of::<AcpiMadtHeader>();
347
348	// Loop through all table entries.
349	while current_address < madt.table_end_address() {
350		let record =
351			unsafe { &*(ptr::with_exposed_provenance::<AcpiMadtRecordHeader>(current_address)) };
352		current_address += mem::size_of::<AcpiMadtRecordHeader>();
353
354		match record.entry_type {
355			0 => {
356				// Processor Local APIC
357				let processor_local_apic_record = unsafe {
358					&*(ptr::with_exposed_provenance::<ProcessorLocalApicRecord>(current_address))
359				};
360				debug!("Found Processor Local APIC record: {processor_local_apic_record}");
361
362				if processor_local_apic_record.flags & CPU_FLAG_ENABLED > 0 {
363					add_local_apic_id(processor_local_apic_record.apic_id);
364				}
365			}
366			1 => {
367				// I/O APIC
368				let ioapic_record =
369					unsafe { &*(ptr::with_exposed_provenance::<IoApicRecord>(current_address)) };
370				debug!("Found I/O APIC record: {ioapic_record}");
371
372				init_ioapic_address(PhysAddr::new(ioapic_record.address.into()));
373			}
374			_ => {
375				// Just ignore other entries for now.
376			}
377		}
378
379		current_address += record.length as usize - mem::size_of::<AcpiMadtRecordHeader>();
380	}
381
382	// Successfully derived all information from the MADT.
383	// Return the physical address of the Local APIC.
384	Ok(PhysAddr::new(madt_header.local_apic_address.into()))
385}

387/// Search Floating Pointer Structure of the Multiprocessing Specification
388fn search_mp_floating(memory_range: AddrRange<PhysAddr>) -> Result<&'static ApicMP, ()> {
389	let layout = PageLayout::from_size(BasePageSize::SIZE as usize).unwrap();
390	let page_range = PageBox::new(layout).unwrap();
391	let virtual_address = VirtAddr::from(page_range.start());
392
393	for current_address in memory_range.iter().step_by(BasePageSize::SIZE as usize) {
394		let mut flags = PageTableEntryFlags::empty();
395		flags.normal().writable();
396		paging::map::<BasePageSize>(
397			virtual_address,
398			current_address.align_down(BasePageSize::SIZE),
399			1,
400			flags,
401		);
402
403		for i in 0..BasePageSize::SIZE / 4 {
404			let mut tmp: *const u32 = virtual_address.as_ptr();
405			tmp = unsafe { tmp.offset(i.try_into().unwrap()) };
406			let apic_mp = unsafe { &*tmp.cast::<ApicMP>() };
407			if apic_mp.signature == MP_FLT_SIGNATURE
408				&& !(apic_mp.version > 4 || apic_mp.features[0] != 0)
409			{
410				return Ok(apic_mp);
411			}
412		}
413	}
414
415	Err(())
416}

418/// Detect APIC by the Multiprocessor Specification
419fn detect_from_mp() -> Result<PhysAddr, ()> {
420	let mp_float = if let Ok(mpf) = search_mp_floating(
421		AddrRange::new(PhysAddr::new(0x9f000u64), PhysAddr::new(0xa0000u64)).unwrap(),
422	) {
423		Ok(mpf)
424	} else if let Ok(mpf) = search_mp_floating(
425		AddrRange::new(PhysAddr::new(0xf0000u64), PhysAddr::new(0x10_0000u64)).unwrap(),
426	) {
427		Ok(mpf)
428	} else {
429		Err(())
430	}?;
431
432	info!("Found MP config at {:#x}", { mp_float.mp_config });
433	info!(
434		"System uses Multiprocessing Specification 1.{}",
435		mp_float.version
436	);
437	info!("MP features 1: {}", mp_float.features[0]);
438
439	if mp_float.features[1] & 0x80 > 0 {
440		info!("PIC mode implemented");
441	} else {
442		info!("Virtual-Wire mode implemented");
443	}
444
445	let layout = PageLayout::from_size(BasePageSize::SIZE as usize).unwrap();
446	let page_range = PageBox::new(layout).unwrap();
447	let virtual_address = VirtAddr::from(page_range.start());
448
449	let mut flags = PageTableEntryFlags::empty();
450	flags.normal().writable();
451	paging::map::<BasePageSize>(
452		virtual_address,
453		PhysAddr::from((mp_float.mp_config as usize).align_down(BasePageSize::SIZE as usize)),
454		1,
455		flags,
456	);
457
458	let mut addr: usize =
459		(virtual_address | (u64::from(mp_float.mp_config) & (BasePageSize::SIZE - 1))) as usize;
460	let mp_config: &ApicConfigTable = unsafe { &*(ptr::with_exposed_provenance(addr)) };
461	if mp_config.signature != MP_CONFIG_SIGNATURE {
462		warn!("MP config table invalid!");
463		return Err(());
464	}
465
466	if mp_config.entry_count == 0 {
467		warn!("No MP table entries, guessing IOAPIC...");
468		let default_address = PhysAddr::new(0xfec0_0000);
469
470		init_ioapic_address(default_address);
471	} else {
472		// entries starts directly after the config table
473		addr += mem::size_of::<ApicConfigTable>();
474		for _i in 0..mp_config.entry_count {
475			match unsafe { *(ptr::with_exposed_provenance::<u8>(addr)) } {
476				// CPU entry
477				0 => {
478					let cpu_entry: &ApicProcessorEntry =
479						unsafe { &*(ptr::with_exposed_provenance(addr)) };
480					if cpu_entry.cpu_flags & 0x01 == 0x01 {
481						add_local_apic_id(cpu_entry.id);
482					}
483					addr += mem::size_of::<ApicProcessorEntry>();
484				}
485				// IO-APIC entry
486				2 => {
487					let io_entry: &ApicIoEntry = unsafe { &*(ptr::with_exposed_provenance(addr)) };
488					let ioapic = PhysAddr::new(io_entry.addr.into());
489					info!("IOAPIC found at {ioapic:p}");
490
491					init_ioapic_address(ioapic);
492
493					addr += mem::size_of::<ApicIoEntry>();
494				}
495				_ => {
496					addr += 8;
497				}
498			}
499		}
500	}
501
502	Ok(PhysAddr::new(mp_config.lapic.into()))
503}

518pub fn init() {
519	// Detect CPUs and APICs.
520	let local_apic_physical_address = if env::is_uhyve() {
521		default_apic()
522	} else {
523		detect_from_acpi()
524			.or_else(|()| detect_from_mp())
525			.unwrap_or_else(|()| default_apic())
526	};
527
528	// Initialize x2APIC or xAPIC, depending on what's available.
529	if processor::supports_x2apic() {
530		init_x2apic();
531	} else if env::is_uefi() {
532		// already id mapped in UEFI systems, just use the physical address as virtual one
533		LOCAL_APIC_ADDRESS
534			.set(VirtAddr::new(local_apic_physical_address.as_u64()))
535			.unwrap();
536	} else {
537		// We use the traditional xAPIC mode available on all x86-64 CPUs.
538		// It uses a mapped page for communication.
539		let layout = PageLayout::from_size(BasePageSize::SIZE as usize).unwrap();
540		let page_range = PageAlloc::allocate(layout).unwrap();
541		let local_apic_address = VirtAddr::from(page_range.start());
542		LOCAL_APIC_ADDRESS.set(local_apic_address).unwrap();
543		debug!(
544			"Mapping Local APIC at {local_apic_physical_address:p} to virtual address {local_apic_address:p}"
545		);
546
547		let mut flags = PageTableEntryFlags::empty();
548		flags.device().writable().execute_disable();
549		paging::map::<BasePageSize>(local_apic_address, local_apic_physical_address, 1, flags);
550	}
551
552	// Set gates to ISRs for the APIC interrupts we are going to enable.
553	unsafe {
554		let mut idt = IDT.lock();
555		idt[ERROR_INTERRUPT_NUMBER]
556			.set_handler_fn(error_interrupt_handler)
557			.set_stack_index(0);
558		idt[SPURIOUS_INTERRUPT_NUMBER]
559			.set_handler_fn(spurious_interrupt_handler)
560			.set_stack_index(0);
561		#[cfg(feature = "smp")]
562		{
563			idt[TLB_FLUSH_INTERRUPT_NUMBER]
564				.set_handler_fn(tlb_flush_handler)
565				.set_stack_index(0);
566			interrupts::add_irq_name(TLB_FLUSH_INTERRUPT_NUMBER - 32, "TLB flush");
567			idt[WAKEUP_INTERRUPT_NUMBER]
568				.set_handler_fn(wakeup_handler)
569				.set_stack_index(0);
570			interrupts::add_irq_name(WAKEUP_INTERRUPT_NUMBER - 32, "Wakeup");
571		}
572	}
573
574	// Initialize interrupt handling over APIC.
575	// All interrupts of the PIC have already been masked, so it doesn't need to be disabled again.
576	init_local_apic();
577
578	if !processor::supports_tsc_deadline() {
579		// We have an older APIC Timer without TSC Deadline support, which has a maximum timeout
580		// and needs to be calibrated.
581		calibrate_timer();
582	}
583
584	// currently, IO-APIC isn't supported by uhyve
585	if !env::is_uhyve() {
586		// initialize IO-APIC
587		init_ioapic();
588	}
589}

591fn init_ioapic() {
592	let max_entry = ioapic_max_redirection_entry() + 1;
593	info!("IOAPIC v{} has {} entries", ioapic_version(), max_entry);
594
595	// now lets turn everything else on
596	for i in 0..max_entry {
597		// Turn off the Programmable Interrupt Timer Interrupt (IRQ 0) and
598		// the Real Time Clock (IRQ 2).
599		let enabled = !matches!(i, 0 | 2);
600		ioapic_set_interrupt(i, 0, enabled);
601	}
602}
603
604fn ioapic_set_interrupt(irq: u8, apicid: u8, enabled: bool) {
605	assert!(irq <= 24);
606
607	let off = u32::from(irq * 2);
608	let ioredirect_upper = u32::from(apicid) << 24;
609	let mut ioredirect_lower = u32::from(0x20 + irq);
610	if !enabled {
611		debug!("Disabling irq {irq}");
612		ioredirect_lower |= 1 << 16;
613	}
614
615	ioapic_write(IOAPIC_REG_TABLE + off, ioredirect_lower);
616	ioapic_write(IOAPIC_REG_TABLE + off + 1, ioredirect_upper);
617}

619pub fn init_local_apic() {
620	// Mask out all interrupts we don't need right now.
621	local_apic_write(IA32_X2APIC_LVT_TIMER, APIC_LVT_MASK);
622	local_apic_write(IA32_X2APIC_LVT_THERMAL, APIC_LVT_MASK);
623	local_apic_write(IA32_X2APIC_LVT_PMI, APIC_LVT_MASK);
624	local_apic_write(IA32_X2APIC_LVT_LINT0, APIC_LVT_MASK);
625	local_apic_write(IA32_X2APIC_LVT_LINT1, APIC_LVT_MASK);
626
627	// Set the interrupt number of the Error interrupt.
628	local_apic_write(IA32_X2APIC_LVT_ERROR, u64::from(ERROR_INTERRUPT_NUMBER));
629
630	// allow all interrupts
631	local_apic_write(IA32_X2APIC_TPR, 0x00);
632
633	// Finally, enable the Local APIC by setting the interrupt number for spurious interrupts
634	// and providing the enable bit.
635	local_apic_write(
636		IA32_X2APIC_SIVR,
637		APIC_SIVR_ENABLED | (u64::from(SPURIOUS_INTERRUPT_NUMBER)),
638	);
639}

641fn calibrate_timer() {
642	// The APIC Timer is used to provide a one-shot interrupt for the tickless timer
643	// implemented through processor::get_timer_ticks.
644	// Therefore determine a counter value for 1 microsecond, which is the resolution
645	// used throughout all of Hermit. Wait 30ms for accuracy.
646	let microseconds = 30_000;
647
648	// Be sure that all interrupts for calibration accuracy and initialize the counter are disabled.
649	// Dividing the counter value by 8 still provides enough accuracy for 1 microsecond resolution,
650	// but allows for longer timeouts than a smaller divisor.
651	// For example, on an Intel Xeon E5-2650 v3 @ 2.30GHz, the counter is usually calibrated to
652	// 125, which allows for timeouts of approximately 34 seconds (u32::MAX / 125).
653
654	local_apic_write(IA32_X2APIC_DIV_CONF, APIC_DIV_CONF_DIVIDE_BY_8);
655	local_apic_write(IA32_X2APIC_INIT_COUNT, u64::from(u32::MAX));
656
657	// Wait until the calibration time has elapsed.
658	processor::udelay(microseconds);
659
660	// Save the difference of the initial value and current value as the result of the calibration
661	// and re-enable interrupts.
662	let calibrated_counter_value =
663		(u64::from(u32::MAX - local_apic_read(IA32_X2APIC_CUR_COUNT))) / microseconds;
664	CALIBRATED_COUNTER_VALUE
665		.set(calibrated_counter_value)
666		.unwrap();
667	debug!(
668		"Calibrated APIC Timer with a counter value of {calibrated_counter_value} for 1 microsecond",
669	);
670}

672fn __set_oneshot_timer(wakeup_time: Option<u64>) {
673	let Some(wt) = wakeup_time else {
674		// Disable the APIC Timer.
675		local_apic_write(IA32_X2APIC_LVT_TIMER, APIC_LVT_MASK);
676		return;
677	};
678
679	if processor::supports_tsc_deadline() {
680		// wt is the absolute wakeup time in microseconds based on processor::get_timer_ticks.
681		// We can simply multiply it by the processor frequency to get the absolute Time-Stamp Counter deadline
682		// (see processor::get_timer_ticks).
683		let tsc_deadline = wt * (u64::from(processor::get_frequency()));
684
685		// Enable the APIC Timer in TSC-Deadline Mode and let it start by writing to the respective MSR.
686		local_apic_write(
687			IA32_X2APIC_LVT_TIMER,
688			APIC_LVT_TIMER_TSC_DEADLINE | u64::from(TIMER_INTERRUPT_NUMBER),
689		);
690		let mut ia32_tsc_deadline = IA32_TSC_DEADLINE;
691		unsafe {
692			ia32_tsc_deadline.write(tsc_deadline);
693		}
694		return;
695	}
696
697	// Calculate the relative timeout from the absolute wakeup time.
698	// Maintain a minimum value of one tick, otherwise the timer interrupt does not fire at all.
699	// The Timer Counter Register is also a 32-bit register, which we must not overflow for longer timeouts.
700	let current_time = processor::get_timer_ticks();
701	let ticks = if wt > current_time {
702		wt - current_time
703	} else {
704		1
705	};
706	let init_count = cmp::min(
707		CALIBRATED_COUNTER_VALUE.get().unwrap() * ticks,
708		u64::from(u32::MAX),
709	);
710
711	// Enable the APIC Timer in One-Shot Mode and let it start by setting the initial counter value.
712	local_apic_write(IA32_X2APIC_LVT_TIMER, u64::from(TIMER_INTERRUPT_NUMBER));
713	local_apic_write(IA32_X2APIC_INIT_COUNT, init_count);
714}
715
716pub fn set_oneshot_timer(wakeup_time: Option<u64>) {
717	without_interrupts(|| {
718		__set_oneshot_timer(wakeup_time);
719	});
720}

734/// Initialize the required _start variables for the next CPU to be booted.
735pub fn init_next_processor_variables() {
736	// Allocate stack for the CPU and pass the addresses.
737	let layout = Layout::from_size_align(KERNEL_STACK_SIZE, BasePageSize::SIZE as usize).unwrap();
738	let stack = unsafe { alloc(layout) };
739	assert!(!stack.is_null());
740	CURRENT_STACK_ADDRESS.store(stack, Ordering::Relaxed);
741}

743/// Boot all Application Processors
744/// This algorithm is derived from Intel MultiProcessor Specification 1.4, B.4, but testing has shown
745/// that a second STARTUP IPI and setting the BIOS Reset Vector are no longer necessary.
746/// This is partly confirmed by <https://wiki.osdev.org/Symmetric_Multiprocessing>
747#[cfg(all(target_os = "none", feature = "smp"))]
748pub fn boot_application_processors() {
749	use core::hint;
750
751	use hermit_entry::boot_info::RawBootInfo;
752	use x86_64::structures::paging::Translate;
753
754	use super::start;
755
756	let smp_boot_code = include_bytes!(concat!(core::env!("OUT_DIR"), "/boot.bin"));
757
758	// We shouldn't have any problems fitting the boot code into a single page, but let's better be sure.
759	assert!(
760		smp_boot_code.len() <= BasePageSize::SIZE as usize,
761		"SMP Boot Code is larger than a page"
762	);
763	debug!("SMP boot code is {} bytes long", smp_boot_code.len());
764
765	if env::is_uefi() {
766		// Since UEFI already provides identity-mapped pagetables, we only have to sanity-check the identity mapping
767		let pt = unsafe { crate::arch::mm::paging::identity_mapped_page_table() };
768		let virt_addr = SMP_BOOT_CODE_ADDRESS;
769		let phys_addr = pt.translate_addr(virt_addr.into()).unwrap();
770		assert_eq!(phys_addr.as_u64(), virt_addr.as_u64());
771	} else {
772		// Identity-map the boot code page and copy over the code.
773		debug!("Mapping SMP boot code to physical and virtual address {SMP_BOOT_CODE_ADDRESS:p}");
774		let mut flags = PageTableEntryFlags::empty();
775		flags.normal().writable();
776		paging::map::<BasePageSize>(
777			SMP_BOOT_CODE_ADDRESS,
778			PhysAddr::new(SMP_BOOT_CODE_ADDRESS.as_u64()),
779			1,
780			flags,
781		);
782	}
783	unsafe {
784		// FIXME: do bounds checking. Better yet: do the copy via slices
785		SMP_BOOT_CODE_ADDRESS
786			.as_mut_ptr::<u8>()
787			.copy_from_nonoverlapping(smp_boot_code.as_ptr(), smp_boot_code.len());
788	}
789
790	unsafe {
791		let (frame, val) = Cr3::read_raw();
792		let value = frame.start_address().as_u64() | u64::from(val);
793		// Pass the PML4 page table address to the boot code.
794		*((SMP_BOOT_CODE_ADDRESS + SMP_BOOT_CODE_OFFSET_PML4).as_mut_ptr::<u32>()) =
795			value.try_into().unwrap();
796		// Set entry point
797		debug!(
798			"Set entry point for application processor to {:p}",
799			start::_start as *const ()
800		);
801		(SMP_BOOT_CODE_ADDRESS + SMP_BOOT_CODE_OFFSET_ENTRY)
802			.as_mut_ptr::<unsafe extern "C" fn(Option<&'static RawBootInfo>, cpu_id: u32) -> !>()
803			.write_unaligned(start::_start);
804	}
805
806	// Now wake up each application processor.
807	let apic_ids = CPU_LOCAL_APIC_IDS.lock();
808	let core_id = core_id();
809
810	for (core_id_to_boot, &apic_id) in apic_ids.iter().enumerate() {
811		let core_id_to_boot = core_id_to_boot as u32;
812		if core_id_to_boot != core_id {
813			unsafe {
814				*((SMP_BOOT_CODE_ADDRESS + SMP_BOOT_CODE_OFFSET_CPU_ID).as_mut_ptr()) =
815					core_id_to_boot;
816			}
817			let destination = u64::from(apic_id) << 32;
818
819			debug!("Waking up CPU {core_id_to_boot} with Local APIC ID {apic_id}");
820			init_next_processor_variables();
821
822			// Save the current number of initialized CPUs.
823			let current_processor_count = arch::get_processor_count();
824
825			// Send an INIT IPI.
826			local_apic_write(
827				IA32_X2APIC_ICR,
828				destination
829					| APIC_ICR_LEVEL_TRIGGERED
830					| APIC_ICR_LEVEL_ASSERT
831					| APIC_ICR_DELIVERY_MODE_INIT,
832			);
833			processor::udelay(200);
834
835			local_apic_write(
836				IA32_X2APIC_ICR,
837				destination | APIC_ICR_LEVEL_TRIGGERED | APIC_ICR_DELIVERY_MODE_INIT,
838			);
839			processor::udelay(10000);
840
841			// Send a STARTUP IPI.
842			local_apic_write(
843				IA32_X2APIC_ICR,
844				destination
845					| APIC_ICR_DELIVERY_MODE_STARTUP
846					| ((SMP_BOOT_CODE_ADDRESS.as_u64()) >> 12),
847			);
848			debug!("Waiting for it to respond");
849
850			// Wait until the application processor has finished initializing.
851			// It will indicate this by counting up cpu_online.
852			while current_processor_count == arch::get_processor_count() {
853				hint::spin_loop();
854			}
855		}
856	}
857
858	print_information();
859}

861#[cfg(feature = "smp")]
862pub fn ipi_tlb_flush() {
863	if arch::get_processor_count() > 1 {
864		let apic_ids = CPU_LOCAL_APIC_IDS.lock();
865		let core_id = core_id();
866
867		// Ensure that all memory operations have completed before issuing a TLB flush.
868		unsafe {
869			_mm_mfence();
870		}
871
872		// Send an IPI with our TLB Flush interrupt number to all other CPUs.
873		without_interrupts(|| {
874			for (core_id_to_interrupt, &apic_id) in apic_ids.iter().enumerate() {
875				if core_id_to_interrupt != usize::try_from(core_id).unwrap() {
876					let destination = u64::from(apic_id) << 32;
877					local_apic_write(
878						IA32_X2APIC_ICR,
879						destination
880							| APIC_ICR_LEVEL_ASSERT
881							| APIC_ICR_DELIVERY_MODE_FIXED
882							| u64::from(TLB_FLUSH_INTERRUPT_NUMBER),
883					);
884				}
885			}
886		});
887	}
888}

890/// Send an inter-processor interrupt to wake up a CPU Core that is in a HALT state.
891#[allow(unused_variables)]
892pub fn wakeup_core(core_id_to_wakeup: CoreId) {
893	#[cfg(all(feature = "smp", not(feature = "idle-poll")))]
894	if core_id_to_wakeup != core_id()
895		&& !crate::processor::supports_mwait()
896		&& crate::scheduler::take_core_hlt_state(core_id_to_wakeup)
897	{
898		without_interrupts(|| {
899			let apic_ids = CPU_LOCAL_APIC_IDS.lock();
900			let local_apic_id = apic_ids[core_id_to_wakeup as usize];
901			let destination = u64::from(local_apic_id) << 32;
902			local_apic_write(
903				IA32_X2APIC_ICR,
904				destination
905					| APIC_ICR_LEVEL_ASSERT
906					| APIC_ICR_DELIVERY_MODE_FIXED
907					| u64::from(WAKEUP_INTERRUPT_NUMBER),
908			);
909		});
910	}

913/// Translate the x2APIC MSR into an xAPIC memory address.
914#[inline]
915fn translate_x2apic_msr_to_xapic_address(x2apic_msr: u32) -> VirtAddr {
916	*LOCAL_APIC_ADDRESS.get().unwrap() + ((u64::from(x2apic_msr) & 0xff) << 4)
917}
918
919fn local_apic_read(x2apic_msr: u32) -> u32 {
920	if processor::supports_x2apic() {
921		// x2APIC is simple, we can just read from the given MSR.
922		unsafe { Msr::new(x2apic_msr).read() as u32 }
923	} else {
924		unsafe { *(translate_x2apic_msr_to_xapic_address(x2apic_msr).as_ptr::<u32>()) }
925	}
926}

928fn ioapic_write(reg: u32, value: u32) {
929	unsafe {
930		IOAPIC_ADDRESS
931			.get()
932			.unwrap()
933			.as_mut_ptr::<u32>()
934			.write_volatile(reg);
935
936		(*IOAPIC_ADDRESS.get().unwrap() + 4 * mem::size_of::<u32>())
937			.as_mut_ptr::<u32>()
938			.write_volatile(value);
939	}
940}

942fn ioapic_read(reg: u32) -> u32 {
943	let value;
944
945	unsafe {
946		IOAPIC_ADDRESS
947			.get()
948			.unwrap()
949			.as_mut_ptr::<u32>()
950			.write_volatile(reg);
951
952		value = (*IOAPIC_ADDRESS.get().unwrap() + 4 * mem::size_of::<u32>())
953			.as_ptr::<u32>()
954			.read_volatile();
955	}
956
957	value
958}

960fn ioapic_version() -> u32 {
961	ioapic_read(IOAPIC_REG_VER) & 0xff
962}
963
964fn ioapic_max_redirection_entry() -> u8 {
965	((ioapic_read(IOAPIC_REG_VER) >> 16) & 0xff) as u8
966}
967
968fn local_apic_write(x2apic_msr: u32, value: u64) {
969	if processor::supports_x2apic() {
970		// x2APIC is simple, we can just write the given value to the given MSR.
971		unsafe {
972			Msr::new(x2apic_msr).write(value);
973		}
974	} else {
975		// Write the value.
976		let ptr = translate_x2apic_msr_to_xapic_address(x2apic_msr).as_mut_ptr::<u32>();
977
978		if x2apic_msr == IA32_X2APIC_ICR {
979			// The ICR1 register in xAPIC mode also has a Delivery Status bit.
980			// Wait until previous interrupt was delivered.
981			// This bit does not exist in x2APIC mode (cf. Intel Vol. 3A, 10.12.9).
982			while (unsafe { ptr.read_volatile() } & APIC_ICR_DELIVERY_STATUS_PENDING) > 0 {
983				spin_loop();
984			}
985
986			// Instead of a single 64-bit ICR register, xAPIC has two 32-bit registers (ICR1 and ICR2).
987			// There is a gap between them and the destination field in ICR2 is also 8 bits instead of 32 bits.
988			let destination = ((value >> 8) & 0xff00_0000) as u32;
989			let icr2 = (*LOCAL_APIC_ADDRESS.get().unwrap() + APIC_ICR2).as_mut_ptr::<u32>();
990			unsafe { icr2.write_volatile(destination) };
991
992			// The remaining data without the destination will now be written into ICR1.
993		}
994
995		unsafe {
996			ptr.write_volatile(value as u32);
997		}
998	}
999}
