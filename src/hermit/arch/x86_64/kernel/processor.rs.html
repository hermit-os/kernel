154#[repr(C)]
155pub struct FPUState {
156	xsave_area: Box<[AlignToSixtyFour]>,
157}

159impl FPUState {
160	pub fn new() -> Self {
161		let xsave_size = if supports_xsave() {
162			CpuId::new()
163				.get_extended_state_info()
164				.expect("XSAVE requires extended state info")
165				.xsave_area_size_enabled_features() as usize
166		} else {
167			size_of::<XSaveLegacyRegion>()
168		};
169
170		debug!("XSAVE area size: {xsave_size}");
171
172		// Allocate a 64-byte aligned Vec
173		let n_units = xsave_size.div_ceil(size_of::<AlignToSixtyFour>());
174		let mut xsave_area = vec![AlignToSixtyFour([0; 64]); n_units].into_boxed_slice();
175
176		// SAFETY: We allocated at least the size of XSaveLegacyRegion bytes and have initialized them
177		let legacy_region = unsafe { &mut *xsave_area.as_mut_ptr().cast::<XSaveLegacyRegion>() };
178
179		// Set FPU-related values to their default values after initialization.
180		// Refer to Intel Vol. 3A, Table 9-1. IA-32 and Intel 64 Processor States Following Power-up, Reset, or INIT
181		legacy_region.fpu_control_word = 0x37f;
182		legacy_region.fpu_tag_word = 0xffff;
183		legacy_region.mxcsr = MxCsr::default();
184
185		Self { xsave_area }
186	}
187
188	pub fn restore(&self) {
189		if supports_xsave() {
190			unsafe {
191				_xrstor(self.xsave_area.as_ptr().cast::<u8>(), u64::MAX);
192			}
193		} else {
194			self.restore_common();
195		}
196	}
197
198	pub fn save(&mut self) {
199		if supports_xsave() {
200			if has_xsavec() {
201				unsafe { _xsavec(self.xsave_area.as_mut_ptr().cast::<u8>(), u64::MAX) }
202			} else if has_xsaveopt() {
203				unsafe { _xsaveopt(self.xsave_area.as_mut_ptr().cast::<u8>(), u64::MAX) }
204			} else {
205				unsafe { _xsave(self.xsave_area.as_mut_ptr().cast::<u8>(), u64::MAX) }
206			}
207		} else {
208			self.save_common();
209		}
210	}
211
212	pub fn restore_common(&self) {
213		unsafe {
214			_fxrstor(self.xsave_area.as_ptr().cast::<u8>());
215		}
216	}
217
218	pub fn save_common(&mut self) {
219		unsafe {
220			_fxsave(self.xsave_area.as_mut_ptr().cast::<u8>());
221			asm!("fnclex", options(nomem, nostack));
222		}
223	}

271	fn set_detected_cpu_frequency(
272		&mut self,
273		mhz: u16,
274		source: CpuFrequencySources,
275	) -> Result<(), ()> {
276		//The clock frequency must never be set to zero, otherwise a division by zero will
277		//occur during runtime
278		if mhz == 0 {
279			return Err(());
280		}
281
282		self.mhz = mhz;
283		self.source = source;
284		Ok(())
285	}

287	unsafe fn detect_from_cmdline(&mut self) -> Result<(), ()> {
288		let mhz = env::freq().ok_or(())?;
289		self.set_detected_cpu_frequency(mhz, CpuFrequencySources::CommandLine)
290	}

292	unsafe fn detect_from_cpuid(&mut self, cpuid: &CpuId<CpuIdReaderNative>) -> Result<(), ()> {
293		let processor_frequency_info = cpuid.get_processor_frequency_info();
294
295		match processor_frequency_info {
296			Some(freq_info) => {
297				let mhz = freq_info.processor_base_frequency();
298				self.set_detected_cpu_frequency(mhz, CpuFrequencySources::CpuId)
299			}
300			None => Err(()),
301		}
302	}

304	unsafe fn detect_from_cpuid_tsc_info(
305		&mut self,
306		cpuid: &CpuId<CpuIdReaderNative>,
307	) -> Result<(), ()> {
308		let tsc_info = cpuid.get_tsc_info().ok_or(())?;
309		let freq = tsc_info.tsc_frequency().ok_or(())?;
310		let mhz = (freq / 1_000_000u64) as u16;
311		self.set_detected_cpu_frequency(mhz, CpuFrequencySources::CpuIdTscInfo)
312	}
313
314	unsafe fn detect_from_cpuid_hypervisor_info(
315		&mut self,
316		cpuid: &CpuId<CpuIdReaderNative>,
317	) -> Result<(), ()> {
318		const KHZ_TO_HZ: u64 = 1000;
319		const MHZ_TO_HZ: u64 = 1_000_000;
320		let hypervisor_info = cpuid.get_hypervisor_info().ok_or(())?;
321		let freq = u64::from(hypervisor_info.tsc_frequency().ok_or(())?) * KHZ_TO_HZ;
322		let mhz: u16 = (freq / MHZ_TO_HZ).try_into().unwrap();
323		self.set_detected_cpu_frequency(mhz, CpuFrequencySources::HypervisorTscInfo)
324	}
325
326	unsafe fn detect_from_cpuid_brand_string(
327		&mut self,
328		cpuid: &CpuId<CpuIdReaderNative>,
329	) -> Result<(), ()> {
330		let processor_brand = cpuid.get_processor_brand_string().ok_or(())?;
331
332		let brand_string = processor_brand.as_str();
333		let ghz_find = brand_string.find("GHz");
334
335		let ghz_find = ghz_find.ok_or(())?;
336
337		let index = ghz_find - 4;
338		let thousand_char = brand_string.chars().nth(index).unwrap();
339		let decimal_char = brand_string.chars().nth(index + 1).unwrap();
340		let hundred_char = brand_string.chars().nth(index + 2).unwrap();
341		let ten_char = brand_string.chars().nth(index + 3).unwrap();
342
343		let thousand = thousand_char.to_digit(10).ok_or(())?;
344		if decimal_char != '.' {
345			return Err(());
346		}
347		let hundred = hundred_char.to_digit(10).ok_or(())?;
348		let ten = ten_char.to_digit(10).ok_or(())?;
349
350		let mhz = (thousand * 1000 + hundred * 100 + ten * 10) as u16;
351		self.set_detected_cpu_frequency(mhz, CpuFrequencySources::CpuIdBrandString)
352	}
353
354	fn detect_from_fdt(&mut self) -> Result<(), ()> {
355		fn mhz_from_fdt() -> Option<NonZero<u16>> {
356			let khz = env::fdt()?
357				.find_node("/hermit,tsc")?
358				.property("khz")?
359				.as_usize()?;
360			let khz = u32::try_from(khz).ok()?;
361			let mhz = u16::try_from(khz / 1000).ok()?;
362			NonZero::new(mhz)
363		}
364
365		let mhz = mhz_from_fdt().ok_or(())?;
366		self.set_detected_cpu_frequency(mhz.get(), CpuFrequencySources::Fdt)?;
367
368		Ok(())
369	}

371	fn detect_from_hypervisor(&mut self) -> Result<(), ()> {
372		fn detect_from_uhyve() -> Result<u16, ()> {
373			match env::boot_info().platform_info {
374				PlatformInfo::Uhyve { cpu_freq, .. } => Ok(u16::try_from(
375					cpu_freq.map(NonZeroU32::get).unwrap_or_default() / 1000,
376				)
377				.unwrap()),
378				_ => Err(()),
379			}
380		}
381		// future implementations could add support for different hypervisors
382		// by adding or_else here
383		self.set_detected_cpu_frequency(detect_from_uhyve()?, CpuFrequencySources::Hypervisor)
384	}

386	extern "x86-interrupt" fn measure_frequency_timer_handler(
387		_stack_frame: interrupts::ExceptionStackFrame,
388	) {
389		MEASUREMENT_TIMER_TICKS.fetch_add(1, Ordering::Relaxed);
390		pic::eoi(pit::PIT_INTERRUPT_NUMBER);
391	}

393	#[cfg(not(target_os = "none"))]
394	fn measure_frequency(&mut self) -> Result<(), ()> {
395		// return just Ok because the real implementation must run in ring 0
396		self.source = CpuFrequencySources::Measurement;
397		Ok(())
398	}

400	#[cfg(target_os = "none")]
401	fn measure_frequency(&mut self) -> Result<(), ()> {
402		use crate::arch::x86_64::kernel::interrupts::IDT;
403
404		// The PIC is not initialized for uhyve, so we cannot measure anything.
405		if env::is_uhyve() {
406			return Err(());
407		}
408
409		// Measure the CPU frequency by counting 3 ticks of a 100Hz timer.
410		let tick_count = 3;
411		let measurement_frequency = 100;
412
413		// Use the Programmable Interval Timer (PIT) for this measurement, which is the only
414		// system timer with a known constant frequency.
415		unsafe {
416			let mut idt = IDT.lock();
417			idt[pit::PIT_INTERRUPT_NUMBER]
418				.set_handler_fn(Self::measure_frequency_timer_handler)
419				.set_stack_index(0);
420		}
421		pit::init(measurement_frequency);
422
423		// we need a timer interrupt to meature the frequency
424		interrupts::enable();
425
426		// Determine the current timer tick.
427		// We are probably loading this value in the middle of a time slice.
428		let first_tick = MEASUREMENT_TIMER_TICKS.load(Ordering::Relaxed);
429		let start = get_timestamp();
430
431		// Wait until the tick count changes.
432		// As soon as it has done, we are at the start of a new time slice.
433		let start_tick = loop {
434			let tick = MEASUREMENT_TIMER_TICKS.load(Ordering::Relaxed);
435			if tick != first_tick {
436				break Some(tick);
437			}
438
439			if get_timestamp() - start > 120_000_000 {
440				break None;
441			}
442
443			spin_loop();
444		}
445		.ok_or_else(|| {
446			interrupts::disable();
447			pit::deinit();
448		})?;
449
450		// Count the number of CPU cycles during 3 timer ticks.
451		let start = get_timestamp();
452
453		loop {
454			let tick = MEASUREMENT_TIMER_TICKS.load(Ordering::Relaxed);
455			if tick - start_tick >= tick_count {
456				break;
457			}
458
459			spin_loop();
460		}
461
462		let end = get_timestamp();
463
464		// we don't longer need a timer interrupt
465		interrupts::disable();
466
467		// Deinitialize the PIT again.
468		// Now we can calculate our CPU frequency and implement a constant frequency tick counter
469		// using RDTSC timestamps.
470		pit::deinit();
471
472		// Calculate the CPU frequency out of this measurement.
473		let cycle_count = end - start;
474		let mhz = (measurement_frequency * cycle_count / (1_000_000 * tick_count)) as u16;
475		self.set_detected_cpu_frequency(mhz, CpuFrequencySources::Measurement)
476	}
477
478	unsafe fn detect(&mut self) {
479		let cpuid = CpuId::new();
480		unsafe {
481			self.detect_from_fdt()
482				.or_else(|_e| self.detect_from_cpuid(&cpuid))
483				.or_else(|_e| self.detect_from_cpuid_tsc_info(&cpuid))
484				.or_else(|_e| self.detect_from_cpuid_hypervisor_info(&cpuid))
485				.or_else(|_e| self.detect_from_hypervisor())
486				.or_else(|_e| self.detect_from_cmdline())
487				.or_else(|_e| self.detect_from_cpuid_brand_string(&cpuid))
488				.or_else(|_e| self.measure_frequency())
489				.or_else(|_e| {
490					warn!(
491						"Could not determine the processor frequency! Guess a frequency of 2Ghz!"
492					);
493					self.set_detected_cpu_frequency(2000, CpuFrequencySources::Visionary)
494				})
495				.unwrap();
496		}
497	}
498
499	fn get(&self) -> u16 {
500		self.mhz
501	}

675#[derive(Debug)]
676struct CpuSpeedStep {
677	eist_available: bool,
678	eist_enabled: bool,
679	eist_locked: bool,
680	energy_bias_preference: bool,
681	max_pstate: u8,
682	is_turbo_pstate: bool,
683}
684
685impl CpuSpeedStep {
686	const fn new() -> Self {
687		CpuSpeedStep {
688			eist_available: false,
689			eist_enabled: false,
690			eist_locked: false,
691			energy_bias_preference: false,
692			max_pstate: 0,
693			is_turbo_pstate: false,
694		}
695	}
696
697	fn detect_features(&mut self, cpuid: &CpuId<CpuIdReaderNative>) {
698		let feature_info = cpuid
699			.get_feature_info()
700			.expect("CPUID Feature Info not available!");
701
702		self.eist_available = feature_info.has_eist();
703		if !self.eist_available {
704			return;
705		}
706
707		let misc = unsafe { Msr::new(IA32_MISC_ENABLE).read() };
708		self.eist_enabled = (misc & IA32_MISC_ENABLE_ENHANCED_SPEEDSTEP) > 0;
709		self.eist_locked = (misc & IA32_MISC_ENABLE_SPEEDSTEP_LOCK) > 0;
710		if !self.eist_enabled || self.eist_locked {
711			return;
712		}
713
714		self.max_pstate = (unsafe { Msr::new(MSR_PLATFORM_INFO).read() } >> 8) as u8;
715		if (misc & IA32_MISC_ENABLE_TURBO_DISABLE) == 0 {
716			let turbo_pstate = unsafe { Msr::new(MSR_TURBO_RATIO_LIMIT).read() } as u8;
717			if turbo_pstate > self.max_pstate {
718				self.max_pstate = turbo_pstate;
719				self.is_turbo_pstate = true;
720			}
721		}
722
723		if let Some(thermal_power_info) = cpuid.get_thermal_power_info() {
724			self.energy_bias_preference = thermal_power_info.has_energy_bias_pref();
725		}
726	}
727
728	fn configure(&self) {
729		if !self.eist_available || !self.eist_enabled || self.eist_locked {
730			return;
731		}
732
733		if self.energy_bias_preference {
734			unsafe {
735				Msr::new(IA32_ENERGY_PERF_BIAS).write(0);
736			}
737		}
738
739		let mut perf_ctl_mask = u64::from(self.max_pstate) << 8;
740		if self.is_turbo_pstate {
741			perf_ctl_mask |= 1 << 32;
742		}
743
744		unsafe {
745			Msr::new(IA32_PERF_CTL).write(perf_ctl_mask);
746		}
747	}
748}
749
750impl fmt::Display for CpuSpeedStep {
751	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
752		if self.eist_available {
753			write!(f, "Available, ")?;
754
755			if !self.eist_enabled {
756				write!(f, "but disabled")?;
757			} else if self.eist_locked {
758				write!(f, "but locked")?;
759			} else {
760				write!(f, "enabled with maximum P-State {}", self.max_pstate)?;
761				if self.is_turbo_pstate {
762					write!(f, " (Turbo Mode)")?;
763				}
764
765				if self.energy_bias_preference {
766					write!(f, ", disabled Performance/Energy Bias")?;
767				}
768			}
769		} else {
770			write!(f, "Not Available")?;
771		}
772
773		Ok(())
774	}

781pub fn configure() {
782	let cpuid = CpuId::new();
783
784	// setup MSR EFER
785	unsafe {
786		Efer::update(|flags| {
787			flags.insert(
788				EferFlags::SYSTEM_CALL_EXTENSIONS
789					| EferFlags::LONG_MODE_ACTIVE
790					| EferFlags::NO_EXECUTE_ENABLE,
791			);
792		});
793	}
794
795	//
796	// CR0 CONFIGURATION
797	//
798	unsafe {
799		Cr0::update(|flags| {
800			// Enable the FPU.
801			flags.insert(Cr0Flags::MONITOR_COPROCESSOR | Cr0Flags::NUMERIC_ERROR);
802			flags.remove(Cr0Flags::EMULATE_COPROCESSOR);
803
804			// if set, the first FPU access will trigger interrupt 7.
805			flags.insert(Cr0Flags::TASK_SWITCHED);
806
807			// Prevent writes to read-only pages in Ring 0.
808			flags.insert(Cr0Flags::WRITE_PROTECT);
809
810			debug!("Setting CR0 = {flags:?}");
811		});
812	}
813
814	//
815	// CR4 CONFIGURATION
816	//
817	unsafe {
818		Cr4::update(|flags| {
819			let has_pge = cpuid
820				.get_feature_info()
821				.is_some_and(|feature_info| feature_info.has_pge());
822
823			if has_pge {
824				flags.insert(Cr4Flags::PAGE_GLOBAL);
825			}
826
827			// Enable Machine Check Exceptions.
828			// No need to check for support here, all x86-64 CPUs support it.
829			flags.insert(Cr4Flags::MACHINE_CHECK_EXCEPTION);
830
831			// Enable full SSE support and indicates that the OS saves SSE context using FXSR.
832			// No need to check for support here, all x86-64 CPUs support at least SSE2.
833			flags.insert(Cr4Flags::OSFXSR | Cr4Flags::OSXMMEXCPT_ENABLE);
834
835			if supports_xsave() {
836				// Indicate that the OS saves extended context (AVX, AVX2, MPX, etc.) using XSAVE.
837				flags.insert(Cr4Flags::OSXSAVE);
838			}
839
840			// Disable Performance-Monitoring Counters
841			flags.remove(Cr4Flags::PERFORMANCE_MONITOR_COUNTER);
842			// clear TSD => every privilege level is able
843			// to use rdtsc
844			flags.remove(Cr4Flags::TIMESTAMP_DISABLE);
845
846			if supports_fsgs() {
847				flags.insert(Cr4Flags::FSGSBASE);
848				debug!("Enable FSGSBASE support");
849			}
850			#[cfg(feature = "fsgsbase")]
851			if !supports_fsgs() {
852				error!("FSGSBASE support is enabled, but the processor doesn't support it!");
853				crate::scheduler::shutdown(1);
854			}
855
856			debug!("Setting CR4 = {flags:?}");
857		});
858	}
859
860	//
861	// XCR0 CONFIGURATION
862	//
863	if supports_xsave() {
864		// Enable saving the context for all known vector extensions.
865		// Must happen after CR4_ENABLE_OS_XSAVE has been set.
866		// FIXME: migrate to `XCr0::update()` once available:
867		// https://github.com/rust-osdev/x86_64/pull/527
868		let mut flags = XCr0::read();
869		flags.insert(XCr0Flags::X87 | XCr0Flags::SSE);
870
871		if supports_avx() {
872			flags.insert(XCr0Flags::AVX);
873		}
874
875		if xcr0_supports_avx512_opmask() {
876			flags.insert(XCr0Flags::OPMASK);
877		}
878
879		if xcr0_supports_avx512_zmm_hi16() {
880			flags.insert(XCr0Flags::HI16_ZMM);
881		}
882
883		if xcr0_supports_avx512_zmm_hi256() {
884			flags.insert(XCr0Flags::ZMM_HI256);
885		}
886
887		debug!("Setting XCR0 = {flags:?}");
888		unsafe {
889			XCr0::write(flags);
890		}
891	}
892
893	// enable support of syscall and sysret
894	#[cfg(feature = "common-os")]
895	{
896		use x86_64::PrivilegeLevel;
897		use x86_64::registers::model_specific::{LStar, SFMask, Star};
898		use x86_64::registers::rflags::RFlags;
899		use x86_64::structures::gdt::SegmentSelector;
900
901		use crate::arch::x86_64::kernel::syscall;
902
903		let has_syscall = match cpuid.get_extended_processor_and_feature_identifiers() {
904			Some(finfo) => finfo.has_syscall_sysret(),
905			None => false,
906		};
907
908		if has_syscall {
909			info!("Enable SYSCALL support");
910		} else {
911			panic!("Syscall support is missing");
912		}
913		let cs_sysret = SegmentSelector::new(5, PrivilegeLevel::Ring3);
914		let ss_sysret = SegmentSelector::new(4, PrivilegeLevel::Ring3);
915		let cs_syscall = SegmentSelector::new(1, PrivilegeLevel::Ring0);
916		let ss_syscall = SegmentSelector::new(2, PrivilegeLevel::Ring0);
917		Star::write(cs_sysret, ss_sysret, cs_syscall, ss_syscall).unwrap();
918		let syscall_handler_addr = syscall::syscall_handler as *const ();
919		let syscall_handler_addr = VirtAddr::from_ptr(syscall_handler_addr);
920		LStar::write(syscall_handler_addr);
921		SFMask::write(RFlags::INTERRUPT_FLAG); // clear IF flag during system call
922	}
923
924	// Initialize the FS register, which is later used for Thread-Local Storage.
925	writefs(0);
926
927	//
928	// ENHANCED INTEL SPEEDSTEP CONFIGURATION
929	//
930	FEATURES.cpu_speedstep.configure();
931}

965pub fn seed_entropy() -> Option<[u8; 32]> {
966	let mut buf = [0; 32];
967
968	if !FEATURES.supports_rdseed {
969		return None;
970	}
971
972	for word in buf.chunks_mut(8) {
973		let mut value = 0;
974
975		// Some RDRAND implementations on AMD CPUs have had bugs where the carry
976		// flag was incorrectly set without there actually being a random value
977		// available. Even though no bugs are known for RDSEED, we should not
978		// consider the default values random for extra security.
979		while unsafe { _rdseed64_step(&mut value) != 1 } || value == 0 || value == u64::MAX {
980			// Spin as per the recommendation in the
981			// IntelÂ® Digital Random Number Generator (DRNG) implementation guide
982			spin_loop();
983		}
984
985		word.copy_from_slice(&value.to_ne_bytes());
986	}
987
988	Some(buf)
989}

1076/// Causes a triple fault.
1077///
1078/// Triple faults cause CPU resets.
1079/// On KVM, this results in `KVM_EXIT_SHUTDOWN`.
1080/// This is the preferred way of shutting down the CPU on firecracker and in QEMU's `microvm` virtual platform.
1081///
1082/// See [Triple Faulting the CPU](http://www.rcollins.org/Productivity/TripleFault.html).
1083fn triple_fault() -> ! {
1084	let idt = DescriptorTablePointer {
1085		limit: 0,
1086		base: VirtAddr::zero(),
1087	};
1088	unsafe { lidt(&idt) };
1089	int3();
1090	unreachable!()
1091}
