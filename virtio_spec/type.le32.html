pub type le32 = Le<u32>;
Expand description

A 32-bit unsigned integer stored in little-endian byte order.

§Examples

Basic usage:

use endian_num::le32;

let n = 0x1A;

if cfg!(target_endian = "little") {
    assert_eq!(le32::from_ne(n).0, n);
} else {
    assert_eq!(le32::from_ne(n).0, n.swap_bytes());
}
