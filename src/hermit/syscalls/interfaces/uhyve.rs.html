11/// Perform a SerialWriteBuffer hypercall with `buf` as payload.
12#[inline]
13#[cfg_attr(target_arch = "riscv64", expect(dead_code))]
14pub(crate) fn serial_buf_hypercall(buf: &[u8]) {
15	let len = buf.len();
16	let buf = virtual_to_physical(VirtAddr::from_ptr(ptr::from_ref::<[u8]>(buf)))
17		.unwrap()
18		.as_u64()
19		.into();
20	let p = SerialWriteBufferParams { buf, len };
21	uhyve_hypercall(Hypercall::SerialWriteBuffer(&p));
22}

24/// Calculates the physical address of the struct passed as reference.
25#[inline]
26fn data_addr<T>(data: &T) -> u64 {
27	paging::virtual_to_physical(VirtAddr::from_ptr(ptr::from_ref(data)))
28		.unwrap()
29		.as_u64()
30}

32/// Calculates the hypercall data argument
33#[inline]
34fn hypercall_data(hypercall: &Hypercall<'_>) -> u64 {
35	match hypercall {
36		Hypercall::Cmdsize(data) => data_addr(*data),
37		Hypercall::Cmdval(data) => data_addr(*data),
38		Hypercall::Exit(data) => data_addr(*data),
39		Hypercall::FileClose(data) => data_addr(*data),
40		Hypercall::FileLseek(data) => data_addr(*data),
41		Hypercall::FileOpen(data) => data_addr(*data),
42		Hypercall::FileRead(data) => data_addr(*data),
43		Hypercall::FileUnlink(data) => data_addr(*data),
44		Hypercall::FileWrite(data) => data_addr(*data),
45		Hypercall::SerialWriteBuffer(data) => data_addr(*data),
46		Hypercall::SerialWriteByte(byte) => u64::from(*byte),
47		h => todo!("unimplemented hypercall {h:?}"),
48	}
49}

51/// Perform a hypercall to the uhyve hypervisor
52#[inline]
53#[allow(unused_variables)] // until riscv64 is implemented
54pub(crate) fn uhyve_hypercall(hypercall: Hypercall<'_>) {
55	let ptr = HypercallAddress::from(&hypercall) as u16;
56	let data = hypercall_data(&hypercall);
57
58	#[cfg(target_arch = "x86_64")]
59	unsafe {
60		use x86_64::instructions::port::Port;
61
62		let data =
63			u32::try_from(data).expect("Hypercall data must lie in the first 4GiB of memory");
64		Port::new(ptr).write(data);
65	}
66
67	#[cfg(target_arch = "aarch64")]
68	unsafe {
69		use core::arch::asm;
70		asm!(
71			"str x8, [{ptr}]",
72			ptr = in(reg) u64::from(ptr),
73			in("x8") data,
74			options(nostack),
75		);
76	}
77
78	#[cfg(target_arch = "riscv64")]
79	todo!()
80}
