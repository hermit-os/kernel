81	/// Returns the bar at bar-register `slot`.
82	pub fn get_bar(&self, slot: u8) -> Option<Bar> {
83		let header = self.header();
84		if let Some(endpoint) = EndpointHeader::from_header(header, &self.access) {
85			return endpoint.bar(slot, &self.access);
86		}
87
88		None
89	}

91	/// Configure the bar at register `slot`
92	pub fn set_bar(&self, slot: u8, bar: Bar) {
93		let value = match bar {
94			Bar::Io { port } => (port | 1) as usize,
95			Bar::Memory32 {
96				address,
97				size: _,
98				prefetchable,
99			} => {
100				if prefetchable {
101					(address | (1 << 3)) as usize
102				} else {
103					address as usize
104				}
105			}
106			Bar::Memory64 {
107				address,
108				size: _,
109				prefetchable,
110			} => {
111				if prefetchable {
112					(address | (2 << 1) | (1 << 3)) as usize
113				} else {
114					(address | (2 << 1)) as usize
115				}
116			}
117		};
118		let mut header = EndpointHeader::from_header(self.header(), &self.access).unwrap();
119		unsafe {
120			header.write_bar(slot, &self.access, value).unwrap();
121		}
122	}
123
124	/// Memory maps pci bar with specified index to identical location in virtual memory.
125	/// no_cache determines if we set the `Cache Disable` flag in the page-table-entry.
126	/// Returns (virtual-pointer, size) if successful, else None (if bar non-existent or IOSpace)
127	pub fn memory_map_bar(&self, index: u8, no_cache: bool) -> Option<(VirtAddr, usize)> {
128		let (address, size, prefetchable, _width) = match self.get_bar(index) {
129			Some(Bar::Io { .. }) => {
130				warn!("Cannot map IOBar!");
131				return None;
132			}
133			Some(Bar::Memory32 {
134				address,
135				size,
136				prefetchable,
137			}) => (
138				u64::from(address),
139				usize::try_from(size).unwrap(),
140				prefetchable,
141				32,
142			),
143			Some(Bar::Memory64 {
144				address,
145				size,
146				prefetchable,
147			}) => (address, usize::try_from(size).unwrap(), prefetchable, 64),
148			_ => {
149				return None;
150			}
151		};
152
153		if address == 0 {
154			return None;
155		}
156
157		debug!("Mapping bar {index} at {address:#x} with length {size:#x}");
158
159		if !prefetchable {
160			warn!("Currently only mapping of prefetchable bars is supported!");
161		}
162
163		// Since the bios/bootloader manages the physical address space, the address got from the bar is unique and not overlapping.
164		// We therefore do not need to reserve any additional memory in our kernel.
165		// Map bar into RW^X virtual memory
166		let physical_address = address;
167		let virtual_address =
168			crate::mm::map(PhysAddr::new(physical_address), size, true, true, no_cache);
169
170		Some((virtual_address, size))
171	}

173	pub fn get_irq(&self) -> Option<InterruptLine> {
174		let header = self.header();
175		if let Some(endpoint) = EndpointHeader::from_header(header, &self.access) {
176			let (_pin, line) = endpoint.interrupt(&self.access);
177			Some(line)
178		} else {
179			None
180		}
181	}
182
183	pub fn set_irq(&self, pin: InterruptPin, line: InterruptLine) {
184		let mut header = EndpointHeader::from_header(self.header(), &self.access).unwrap();
185		header.update_interrupt(&self.access, |(_pin, _line)| (pin, line));
186	}
187
188	pub fn device_id(&self) -> DeviceId {
189		let (_vendor_id, device_id) = self.header().id(&self.access);
190		device_id
191	}
192
193	pub fn id(&self) -> (VendorId, DeviceId) {
194		self.header().id(&self.access)
195	}
196
197	pub fn status(&self) -> StatusRegister {
198		self.header().status(&self.access)
199	}
200
201	pub fn capabilities(&self) -> Option<CapabilityIterator<&T>> {
202		EndpointHeader::from_header(self.header(), &self.access)
203			.map(|header| header.capabilities(&self.access))
204	}

207impl<T: ConfigRegionAccess> fmt::Display for PciDevice<T> {
208	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
209		let header = self.header();
210		let header_type = header.header_type(&self.access);
211		let (vendor_id, device_id) = header.id(&self.access);
212		let (_dev_rev, class_id, subclass_id, _interface) = header.revision_and_class(&self.access);
213
214		if let Some(endpoint) = EndpointHeader::from_header(header, &self.access) {
215			#[cfg(feature = "pci-ids")]
216			let (class_name, vendor_name, device_name) = {
217				use pci_ids::{Class, Device, FromId, Subclass};
218
219				let class_name = Class::from_id(class_id).map_or("Unknown Class", |class| {
220					class
221						.subclasses()
222						.find(|s| s.id() == subclass_id)
223						.map(Subclass::name)
224						.unwrap_or_else(|| class.name())
225				});
226
227				let (vendor_name, device_name) = Device::from_vid_pid(vendor_id, device_id)
228					.map(|device| (device.vendor().name(), device.name()))
229					.unwrap_or(("Unknown Vendor", "Unknown Device"));
230
231				(class_name, vendor_name, device_name)
232			};
233
234			#[cfg(not(feature = "pci-ids"))]
235			let (class_name, vendor_name, device_name) =
236				("Unknown Class", "Unknown Vendor", "Unknown Device");
237
238			// Output detailed readable information about this device.
239			write!(
240				f,
241				"{:02X}:{:02X} {} [{:02X}{:02X}]: {} {} [{:04X}:{:04X}]",
242				self.address.bus(),
243				self.address.device(),
244				class_name,
245				class_id,
246				subclass_id,
247				vendor_name,
248				device_name,
249				vendor_id,
250				device_id
251			)?;
252
253			// If the devices uses an IRQ, output this one as well.
254			let (_, irq) = endpoint.interrupt(&self.access);
255			if irq != 0 && irq != u8::MAX {
256				write!(f, ", IRQ {irq}")?;
257			}
258
259			let mut slot: u8 = 0;
260			while usize::from(slot) < MAX_BARS {
261				if let Some(pci_bar) = endpoint.bar(slot, &self.access) {
262					match pci_bar {
263						Bar::Memory64 {
264							address,
265							size,
266							prefetchable,
267						} => {
268							write!(
269								f,
270								", BAR{slot} Memory64 {{ address: {address:#X}, size: {size:#X}, prefetchable: {prefetchable} }}"
271							)?;
272							slot += 1;
273						}
274						Bar::Memory32 {
275							address,
276							size,
277							prefetchable,
278						} => {
279							write!(
280								f,
281								", BAR{slot} Memory32 {{ address: {address:#X}, size: {size:#X}, prefetchable: {prefetchable} }}"
282							)?;
283						}
284						Bar::Io { port } => {
285							write!(f, ", BAR{slot} IO {{ port: {port:#X} }}")?;
286						}
287					}
288				}
289				slot += 1;
290			}
291		} else {
292			// Output detailed readable information about this device.
293			write!(
294				f,
295				"{:02X}:{:02X} {:?} [{:04X}:{:04X}]",
296				self.address.bus(),
297				self.address.device(),
298				header_type,
299				vendor_id,
300				device_id
301			)?;
302		}
303
304		Ok(())
305	}
306}

330impl PciDriver {
331	#[cfg(feature = "console")]
332	fn get_console_driver(&self) -> Option<&InterruptTicketMutex<VirtioConsoleDriver>> {
333		#[allow(unreachable_patterns)]
334		match self {
335			Self::VirtioConsole(drv) => Some(drv),
336			_ => None,
337		}
338	}
339
340	#[cfg(feature = "vsock")]
341	fn get_vsock_driver(&self) -> Option<&InterruptTicketMutex<VirtioVsockDriver>> {
342		#[allow(unreachable_patterns)]
343		match self {
344			Self::VirtioVsock(drv) => Some(drv),
345			_ => None,
346		}
347	}
348
349	#[cfg(feature = "fuse")]
350	fn get_filesystem_driver(&self) -> Option<&InterruptTicketMutex<VirtioFsDriver>> {
351		match self {
352			Self::VirtioFs(drv) => Some(drv),
353			#[allow(unreachable_patterns)]
354			_ => None,
355		}
356	}
357
358	fn get_interrupt_handler(&self) -> (InterruptLine, fn()) {
359		#[allow(unreachable_patterns)]
360		match self {
361			#[cfg(feature = "vsock")]
362			Self::VirtioVsock(drv) => {
363				fn vsock_handler() {
364					if let Some(driver) = get_vsock_driver() {
365						driver.lock().handle_interrupt();
366					}
367				}
368
369				let irq_number = drv.lock().get_interrupt_number();
370
371				(irq_number, vsock_handler)
372			}
373			#[cfg(feature = "fuse")]
374			Self::VirtioFs(drv) => {
375				fn fuse_handler() {}
376
377				let irq_number = drv.lock().get_interrupt_number();
378
379				(irq_number, fuse_handler)
380			}
381			#[cfg(feature = "console")]
382			Self::VirtioConsole(drv) => {
383				fn console_handler() {
384					if let Some(driver) = get_console_driver() {
385						driver.lock().handle_interrupt();
386					}
387				}
388
389				let irq_number = drv.lock().get_interrupt_number();
390				(irq_number, console_handler)
391			}
392			_ => todo!(),
393		}
394	}
395}

401pub(crate) fn get_interrupt_handlers() -> HashMap<InterruptLine, InterruptHandlerQueue, RandomState>
402{
403	let mut handlers: HashMap<InterruptLine, InterruptHandlerQueue, RandomState> =
404		HashMap::with_hasher(RandomState::with_seeds(0, 0, 0, 0));
405
406	for drv in PCI_DRIVERS.finalize().iter() {
407		let (irq_number, handler) = drv.get_interrupt_handler();
408
409		if let Some(map) = handlers.get_mut(&irq_number) {
410			map.push_back(handler);
411		} else {
412			let mut map: InterruptHandlerQueue = VecDeque::new();
413			map.push_back(handler);
414			handlers.insert(irq_number, map);
415		}
416	}
417
418	#[cfg(target_arch = "x86_64")]
419	{
420		use crate::kernel::serial::get_serial_handler;
421		let (irq_number, handler) = get_serial_handler();
422
423		if let Some(map) = handlers.get_mut(&irq_number) {
424			map.push_back(handler);
425		} else {
426			let mut map: InterruptHandlerQueue = VecDeque::new();
427			map.push_back(handler);
428			handlers.insert(irq_number, map);
429		}
430	}
431
432	#[cfg(any(feature = "rtl8139", feature = "virtio-net",))]
433	if let Some(device) = NETWORK_DEVICE.lock().as_ref() {
434		handlers
435			.entry(device.get_interrupt_number())
436			.or_default()
437			.push_back(crate::executor::network::network_handler);
438	}
439
440	handlers
441}

473pub(crate) fn init() {
474	// virtio: 4.1.2 PCI Device Discovery
475	without_interrupts(|| {
476		for adapter in PCI_DEVICES.finalize().iter().filter(|x| {
477			let (vendor_id, device_id) = x.id();
478			vendor_id == 0x1af4 && (0x1000..=0x107f).contains(&device_id)
479		}) {
480			info!(
481				"Found virtio device with device id {:#x}",
482				adapter.device_id()
483			);
484
485			#[cfg(any(
486				all(feature = "virtio-net", not(feature = "rtl8139"),),
487				feature = "fuse",
488				feature = "vsock",
489				feature = "console",
490			))]
491			match pci_virtio::init_device(adapter) {
492				#[cfg(all(not(feature = "rtl8139"), feature = "virtio-net",))]
493				Ok(VirtioDriver::Network(drv)) => *crate::executor::device::NETWORK_DEVICE.lock() = Some(drv),
494
495				#[cfg(feature = "console")]
496				Ok(VirtioDriver::Console(drv)) => {
497					register_driver(PciDriver::VirtioConsole(InterruptTicketMutex::new(*drv)));
498					info!("Switch to virtio console");
499					crate::console::CONSOLE
500						.lock()
501						.replace_device(IoDevice::Virtio(VirtioUART::new()));
502				}
503				#[cfg(feature = "vsock")]
504				Ok(VirtioDriver::Vsock(drv)) => {
505					register_driver(PciDriver::VirtioVsock(InterruptTicketMutex::new(*drv)));
506				}
507				#[cfg(feature = "fuse")]
508				Ok(VirtioDriver::FileSystem(drv)) => {
509					register_driver(PciDriver::VirtioFs(InterruptTicketMutex::new(drv)));
510				}
511				_ => {}
512			}
513		}
514
515		// Searching for Realtek RTL8139, which is supported by Qemu
516		#[cfg(feature = "rtl8139")]
517		for adapter in PCI_DEVICES.finalize().iter().filter(|x| {
518			let (vendor_id, device_id) = x.id();
519			vendor_id == 0x10ec && (0x8138..=0x8139).contains(&device_id)
520		}) {
521			info!(
522				"Found Realtek network device with device id {:#x}",
523				adapter.device_id()
524			);
525
526			if let Ok(drv) = rtl8139::init_device(adapter) {
527				*crate::executor::device::NETWORK_DEVICE.lock() = Some(drv);
528			}
529		}
530	});
531}
