25/// A trait for accessing the network interface
26pub(crate) trait NetworkDriver: Driver {
27	/// Returns smoltcp's checksum capabilities
28	fn get_checksums(&self) -> ChecksumCapabilities {
29		ChecksumCapabilities::default()
30	}
31	/// Returns the mac address of the device.
32	fn get_mac_address(&self) -> [u8; 6];
33	/// Returns the current MTU of the device.
34	fn get_mtu(&self) -> u16;
35	/// Get buffer with the received packet
36	fn receive_packet(&mut self) -> Option<(RxToken, TxToken)>;
37	/// Send packet with the size `len`
38	fn send_packet<R, F>(&mut self, len: usize, f: F) -> R
39	where
40		F: FnOnce(&mut [u8]) -> R;
41	/// Check if a packet is available
42	#[allow(dead_code)]
43	fn has_packet(&self) -> bool;
44	/// Enable / disable the polling mode of the network interface
45	fn set_polling_mode(&mut self, value: bool);
46	/// Handle interrupt and check if a packet is available
47	fn handle_interrupt(&mut self);
48}

50// Default IP level MTU to use.
51const DEFAULT_IP_MTU: u16 = 1500;
52
53/// Default MTU to use.
54///
55/// This is 1500 IP MTU and a 14-byte ethernet header.
56const DEFAULT_MTU: u16 = DEFAULT_IP_MTU + 14;
57
58/// Determines the MTU that should be used as configured by crate features
59/// or environment variables.
60pub(crate) fn mtu() -> u16 {
61	if let Some(my_mtu) = hermit_var!("HERMIT_MTU") {
62		u16::from_str(&my_mtu).unwrap()
63	} else {
64		DEFAULT_MTU
65	}
66}
