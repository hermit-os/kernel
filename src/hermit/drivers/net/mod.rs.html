21/// A trait for accessing the network interface
22pub(crate) trait NetworkDriver: Driver + smoltcp::phy::Device {
23	/// Returns the mac address of the device.
24	fn get_mac_address(&self) -> [u8; 6];
25	/// Check if a packet is available
26	#[allow(dead_code)]
27	fn has_packet(&self) -> bool;
28	/// Enable / disable the polling mode of the network interface
29	fn set_polling_mode(&mut self, value: bool);
30	/// Handle interrupt and check if a packet is available
31	#[allow(dead_code)]
32	fn handle_interrupt(&mut self);
33}

35/// Determines the MTU that should be used as configured by crate features
36/// or environment variables.
37#[cfg(any(
38	all(target_arch = "riscv64", feature = "gem-net", not(feature = "pci")),
39	feature = "rtl8139",
40	feature = "virtio-net",
41))]
42pub(crate) fn mtu() -> u16 {
43	use core::str::FromStr;
44
45	// Default IP level MTU to use.
46	const DEFAULT_IP_MTU: u16 = 1500;
47
48	/// Default MTU to use.
49	///
50	/// This is 1500 IP MTU and a 14-byte ethernet header.
51	const DEFAULT_MTU: u16 = DEFAULT_IP_MTU + 14;
52
53	let Some(my_mtu) = hermit_var!("HERMIT_MTU") else {
54		return DEFAULT_MTU;
55	};
56
57	u16::from_str(&my_mtu).unwrap()
58}
