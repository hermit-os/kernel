26/// Physical and virtual address range of the 2 MiB pages that map the kernel.
27static KERNEL_ADDR_RANGE: Lazy<Range<VirtAddr>> = Lazy::new(|| {
28	if cfg!(target_os = "none") {
29		// Calculate the start and end addresses of the 2 MiB page(s) that map the kernel.
30		env::get_base_address().align_down(LargePageSize::SIZE)
31			..(env::get_base_address() + env::get_image_size()).align_up(LargePageSize::SIZE)
32	} else {
33		VirtAddr::zero()..VirtAddr::zero()
34	}
35});
36
37pub(crate) fn kernel_start_address() -> VirtAddr {
38	KERNEL_ADDR_RANGE.start
39}
40
41pub(crate) fn kernel_end_address() -> VirtAddr {
42	KERNEL_ADDR_RANGE.end
43}

45#[cfg(target_os = "none")]
46pub(crate) fn init() {
47	use crate::arch::mm::paging;
48
49	Lazy::force(&KERNEL_ADDR_RANGE);
50
51	arch::mm::init();
52	arch::mm::init_page_tables();
53
54	let total_mem = physicalmem::total_memory_size();
55	let kernel_addr_range = KERNEL_ADDR_RANGE.clone();
56	info!("Total memory size: {} MiB", total_mem >> 20);
57	info!(
58		"Kernel region: {:p}..{:p}",
59		kernel_addr_range.start, kernel_addr_range.end
60	);
61
62	// we reserve physical memory for the required page tables
63	// In worst case, we use page size of BasePageSize::SIZE
64	let npages = total_mem / BasePageSize::SIZE as usize;
65	let npage_3tables = npages / (BasePageSize::SIZE as usize / mem::align_of::<usize>()) + 1;
66	let npage_2tables =
67		npage_3tables / (BasePageSize::SIZE as usize / mem::align_of::<usize>()) + 1;
68	let npage_1tables =
69		npage_2tables / (BasePageSize::SIZE as usize / mem::align_of::<usize>()) + 1;
70	let reserved_space = (npage_3tables + npage_2tables + npage_1tables)
71		* BasePageSize::SIZE as usize
72		+ 2 * LargePageSize::SIZE as usize;
73	#[cfg(any(target_arch = "x86_64", target_arch = "riscv64"))]
74	let has_1gib_pages = arch::processor::supports_1gib_pages();
75	let has_2mib_pages = arch::processor::supports_2mib_pages();
76
77	let min_mem = if env::is_uefi() {
78		// On UEFI, the given memory is guaranteed free memory and the kernel is located before the given memory
79		reserved_space
80	} else {
81		(kernel_addr_range.end.as_u64() - env::get_ram_address().as_u64() + reserved_space as u64)
82			as usize
83	};
84	info!("Minimum memory size: {}", min_mem >> 20);
85	let avail_mem = total_mem
86		.checked_sub(min_mem)
87		.unwrap_or_else(|| panic!("Not enough memory available!"))
88		.align_down(LargePageSize::SIZE as usize);
89
90	let mut map_addr;
91	let mut map_size;
92	let heap_start_addr;
93
94	#[cfg(feature = "common-os")]
95	{
96		info!("Using HermitOS as common OS!");
97
98		// we reserve at least 75% of the memory for the user space
99		let reserve: usize = (avail_mem * 75) / 100;
100		// 64 MB is enough as kernel heap
101		let reserve = core::cmp::min(reserve, 0x0400_0000);
102
103		let virt_size: usize = reserve.align_down(LargePageSize::SIZE as usize);
104		let virt_addr =
105			arch::mm::virtualmem::allocate_aligned(virt_size, LargePageSize::SIZE as usize)
106				.unwrap();
107		heap_start_addr = virt_addr;
108
109		info!(
110			"Heap: size {} MB, start address {:p}",
111			virt_size >> 20,
112			virt_addr
113		);
114
115		#[cfg(any(target_arch = "x86_64", target_arch = "riscv64"))]
116		if has_1gib_pages && virt_size > HugePageSize::SIZE as usize {
117			// Mount large pages to the next huge page boundary
118			let npages = (virt_addr.align_up(HugePageSize::SIZE) - virt_addr) as usize
119				/ LargePageSize::SIZE as usize;
120			if let Err(n) = paging::map_heap::<LargePageSize>(virt_addr, npages) {
121				map_addr = virt_addr + n as u64 * LargePageSize::SIZE;
122				map_size = virt_size - (map_addr - virt_addr) as usize;
123			} else {
124				map_addr = virt_addr.align_up(HugePageSize::SIZE);
125				map_size = virt_size - (map_addr - virt_addr) as usize;
126			}
127		} else {
128			map_addr = virt_addr;
129			map_size = virt_size;
130		}
131
132		#[cfg(not(any(target_arch = "x86_64", target_arch = "riscv64")))]
133		{
134			map_addr = virt_addr;
135			map_size = virt_size;
136		}
137	}
138
139	#[cfg(not(feature = "common-os"))]
140	{
141		// we reserve 10% of the memory for stack allocations
142		let stack_reserve: usize = (avail_mem * 10) / 100;
143
144		// At first, we map only a small part into the heap.
145		// Afterwards, we already use the heap and map the rest into
146		// the virtual address space.
147
148		#[cfg(not(feature = "mmap"))]
149		let virt_size: usize = (avail_mem - stack_reserve).align_down(LargePageSize::SIZE as usize);
150		#[cfg(feature = "mmap")]
151		let virt_size: usize = ((avail_mem * 75) / 100).align_down(LargePageSize::SIZE as usize);
152
153		let virt_addr =
154			arch::mm::virtualmem::allocate_aligned(virt_size, LargePageSize::SIZE as usize)
155				.unwrap();
156		heap_start_addr = virt_addr;
157
158		info!(
159			"Heap: size {} MB, start address {:p}",
160			virt_size >> 20,
161			virt_addr
162		);
163
164		#[cfg(any(target_arch = "x86_64", target_arch = "riscv64"))]
165		if has_1gib_pages && virt_size > HugePageSize::SIZE as usize {
166			// Mount large pages to the next huge page boundary
167			let npages = (virt_addr.align_up(HugePageSize::SIZE) - virt_addr) / LargePageSize::SIZE;
168			if let Err(n) = paging::map_heap::<LargePageSize>(virt_addr, npages as usize) {
169				map_addr = virt_addr + n as u64 * LargePageSize::SIZE;
170				map_size = virt_size - (map_addr - virt_addr) as usize;
171			} else {
172				map_addr = virt_addr.align_up(HugePageSize::SIZE);
173				map_size = virt_size - (map_addr - virt_addr) as usize;
174			}
175		} else {
176			map_addr = virt_addr;
177			map_size = virt_size;
178		}
179
180		#[cfg(not(any(target_arch = "x86_64", target_arch = "riscv64")))]
181		{
182			map_addr = virt_addr;
183			map_size = virt_size;
184		}
185	}
186
187	#[cfg(any(target_arch = "x86_64", target_arch = "riscv64"))]
188	if has_1gib_pages
189		&& map_size > HugePageSize::SIZE as usize
190		&& map_addr.is_aligned_to(HugePageSize::SIZE)
191	{
192		let size = map_size.align_down(HugePageSize::SIZE as usize);
193		if let Err(num_pages) =
194			paging::map_heap::<HugePageSize>(map_addr, size / HugePageSize::SIZE as usize)
195		{
196			map_size -= num_pages * HugePageSize::SIZE as usize;
197			map_addr += num_pages as u64 * HugePageSize::SIZE;
198		} else {
199			map_size -= size;
200			map_addr += size;
201		}
202	}
203
204	if has_2mib_pages
205		&& map_size > LargePageSize::SIZE as usize
206		&& map_addr.is_aligned_to(LargePageSize::SIZE)
207	{
208		let size = map_size.align_down(LargePageSize::SIZE as usize);
209		if let Err(num_pages) =
210			paging::map_heap::<LargePageSize>(map_addr, size / LargePageSize::SIZE as usize)
211		{
212			map_size -= num_pages * LargePageSize::SIZE as usize;
213			map_addr += num_pages as u64 * LargePageSize::SIZE;
214		} else {
215			map_size -= size;
216			map_addr += size;
217		}
218	}
219
220	if map_size > BasePageSize::SIZE as usize && map_addr.is_aligned_to(BasePageSize::SIZE) {
221		let size = map_size.align_down(BasePageSize::SIZE as usize);
222		if let Err(num_pages) =
223			paging::map_heap::<BasePageSize>(map_addr, size / BasePageSize::SIZE as usize)
224		{
225			map_size -= num_pages * BasePageSize::SIZE as usize;
226			map_addr += num_pages as u64 * BasePageSize::SIZE;
227		} else {
228			map_size -= size;
229			map_addr += size;
230		}
231	}
232
233	let heap_end_addr = map_addr;
234
235	unsafe {
236		ALLOCATOR.init(
237			heap_start_addr.as_mut_ptr(),
238			(heap_end_addr - heap_start_addr) as usize,
239		);
240	}
241
242	info!("Heap is located at {heap_start_addr:p}..{heap_end_addr:p} ({map_size} Bytes unmapped)");
243}

250/// Soft-deprecated in favor of `DeviceAlloc`
251pub(crate) fn allocate(size: usize, no_execution: bool) -> VirtAddr {
252	let size = size.align_up(BasePageSize::SIZE as usize);
253	let physical_address = arch::mm::physicalmem::allocate(size).unwrap();
254	let virtual_address = arch::mm::virtualmem::allocate(size).unwrap();
255
256	let count = size / BasePageSize::SIZE as usize;
257	let mut flags = PageTableEntryFlags::empty();
258	flags.normal().writable();
259	if no_execution {
260		flags.execute_disable();
261	}
262
263	arch::mm::paging::map::<BasePageSize>(virtual_address, physical_address, count, flags);
264
265	virtual_address
266}

268/// Soft-deprecated in favor of `DeviceAlloc`
269pub(crate) fn deallocate(virtual_address: VirtAddr, size: usize) {
270	let size = size.align_up(BasePageSize::SIZE as usize);
271
272	if let Some(phys_addr) = arch::mm::paging::virtual_to_physical(virtual_address) {
273		arch::mm::paging::unmap::<BasePageSize>(
274			virtual_address,
275			size / BasePageSize::SIZE as usize,
276		);
277		arch::mm::virtualmem::deallocate(virtual_address, size);
278		arch::mm::physicalmem::deallocate(phys_addr, size);
279	} else {
280		panic!(
281			"No page table entry for virtual address {:p}",
282			virtual_address
283		);
284	}
285}

287/// Maps a given physical address and size in virtual space and returns address.
288#[cfg(feature = "pci")]
289pub(crate) fn map(
290	physical_address: PhysAddr,
291	size: usize,
292	writable: bool,
293	no_execution: bool,
294	no_cache: bool,
295) -> VirtAddr {
296	let size = size.align_up(BasePageSize::SIZE as usize);
297	let count = size / BasePageSize::SIZE as usize;
298
299	let mut flags = PageTableEntryFlags::empty();
300	flags.normal();
301	if writable {
302		flags.writable();
303	}
304	if no_execution {
305		flags.execute_disable();
306	}
307	if no_cache {
308		flags.device();
309	}
310
311	let virtual_address = arch::mm::virtualmem::allocate(size).unwrap();
312	arch::mm::paging::map::<BasePageSize>(virtual_address, physical_address, count, flags);
313
314	virtual_address
315}

318/// unmaps virtual address, without 'freeing' physical memory it is mapped to!
319pub(crate) fn unmap(virtual_address: VirtAddr, size: usize) {
320	let size = size.align_up(BasePageSize::SIZE as usize);
321
322	if arch::mm::paging::virtual_to_physical(virtual_address).is_some() {
323		arch::mm::paging::unmap::<BasePageSize>(
324			virtual_address,
325			size / BasePageSize::SIZE as usize,
326		);
327		arch::mm::virtualmem::deallocate(virtual_address, size);
328	} else {
329		panic!(
330			"No page table entry for virtual address {:p}",
331			virtual_address
332		);
333	}
