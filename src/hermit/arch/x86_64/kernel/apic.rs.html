260extern "x86-interrupt" fn tlb_flush_handler(stack_frame: interrupts::ExceptionStackFrame) {
261	swapgs(&stack_frame);
262	debug!("Received TLB Flush Interrupt");
263	increment_irq_counter(TLB_FLUSH_INTERRUPT_NUMBER);
264	let (frame, val) = Cr3::read_raw();
265	unsafe {
266		Cr3::write_raw(frame, val);
267	}
268	eoi();
269	swapgs(&stack_frame);
270}

272extern "x86-interrupt" fn error_interrupt_handler(stack_frame: interrupts::ExceptionStackFrame) {
273	swapgs(&stack_frame);
274	error!("APIC LVT Error Interrupt");
275	error!("ESR: {:#X}", local_apic_read(IA32_X2APIC_ESR));
276	error!("{stack_frame:#?}");
277	eoi();
278	scheduler::abort();
279}

284	scheduler::abort();

287#[cfg(feature = "smp")]
288extern "x86-interrupt" fn wakeup_handler(stack_frame: interrupts::ExceptionStackFrame) {
289	swapgs(&stack_frame);
290	use crate::scheduler::PerCoreSchedulerExt;
291
292	debug!("Received Wakeup Interrupt");
293	increment_irq_counter(WAKEUP_INTERRUPT_NUMBER);
294	let core_scheduler = core_scheduler();
295	core_scheduler.check_input();
296	eoi();
297	if core_scheduler.is_scheduling() {
298		core_scheduler.reschedule();
299	}
300	swapgs(&stack_frame);
301}

313fn init_ioapic_address(phys_addr: PhysAddr) {
314	if env::is_uefi() {
315		// UEFI systems have already id mapped everything, so we can just set the physical address as the virtual one
316		IOAPIC_ADDRESS
317			.set(VirtAddr::new(phys_addr.as_u64()))
318			.unwrap();
319	} else {
320		let layout = PageLayout::from_size(BasePageSize::SIZE as usize).unwrap();
321		let page_range = PageAlloc::allocate(layout).unwrap();
322		let ioapic_address = VirtAddr::from(page_range.start());
323		IOAPIC_ADDRESS.set(ioapic_address).unwrap();
324		debug!("Mapping IOAPIC at {phys_addr:p} to virtual address {ioapic_address:p}");
325
326		let mut flags = PageTableEntryFlags::empty();
327		flags.device().writable().execute_disable();
328		paging::map::<BasePageSize>(ioapic_address, phys_addr, 1, flags);
329	}
330}

338#[cfg(feature = "acpi")]
339fn detect_from_acpi() -> Result<PhysAddr, ()> {
340	// Get the Multiple APIC Description Table (MADT) from the ACPI information and its specific table header.
341	let madt = acpi::get_madt().ok_or(())?;
342	let madt_header =
343		unsafe { &*(ptr::with_exposed_provenance::<AcpiMadtHeader>(madt.table_start_address())) };
344
345	// Jump to the actual table entries (after the table header).
346	let mut current_address = madt.table_start_address() + mem::size_of::<AcpiMadtHeader>();
347
348	// Loop through all table entries.
349	while current_address < madt.table_end_address() {
350		let record =
351			unsafe { &*(ptr::with_exposed_provenance::<AcpiMadtRecordHeader>(current_address)) };
352		current_address += mem::size_of::<AcpiMadtRecordHeader>();
353
354		match record.entry_type {
355			0 => {
356				// Processor Local APIC
357				let processor_local_apic_record = unsafe {
358					&*(ptr::with_exposed_provenance::<ProcessorLocalApicRecord>(current_address))
359				};
360				debug!("Found Processor Local APIC record: {processor_local_apic_record}");
361
362				if processor_local_apic_record.flags & CPU_FLAG_ENABLED > 0 {
363					add_local_apic_id(processor_local_apic_record.apic_id);
364				}
365			}
366			1 => {
367				// I/O APIC
368				let ioapic_record =
369					unsafe { &*(ptr::with_exposed_provenance::<IoApicRecord>(current_address)) };
370				debug!("Found I/O APIC record: {ioapic_record}");
371
372				init_ioapic_address(PhysAddr::new(ioapic_record.address.into()));
373			}
374			_ => {
375				// Just ignore other entries for now.
376			}
377		}
378
379		current_address += record.length as usize - mem::size_of::<AcpiMadtRecordHeader>();
380	}
381
382	// Successfully derived all information from the MADT.
383	// Return the physical address of the Local APIC.
384	Ok(PhysAddr::new(madt_header.local_apic_address.into()))
385}

387/// Helper function to search Floating Pointer Structure of the Multiprocessing Specification
388fn search_mp_floating(memory_range: AddrRange<PhysAddr>) -> Result<&'static ApicMP, ()> {
389	let layout = PageLayout::from_size(BasePageSize::SIZE as usize).unwrap();
390	let page_range = PageBox::new(layout).unwrap();
391	let virtual_address = VirtAddr::from(page_range.start());
392
393	for current_address in memory_range.iter().step_by(BasePageSize::SIZE as usize) {
394		let mut flags = PageTableEntryFlags::empty();
395		flags.normal().writable();
396		paging::map::<BasePageSize>(
397			virtual_address,
398			current_address.align_down(BasePageSize::SIZE),
399			1,
400			flags,
401		);
402
403		for i in 0..BasePageSize::SIZE / 4 {
404			let mut tmp: *const u32 = virtual_address.as_ptr();
405			tmp = unsafe { tmp.offset(i.try_into().unwrap()) };
406			let apic_mp = unsafe { &*tmp.cast::<ApicMP>() };
407			if apic_mp.signature == MP_FLT_SIGNATURE
408				&& !(apic_mp.version > 4 || apic_mp.features[0] != 0)
409			{
410				return Ok(apic_mp);
411			}
412		}
413	}
414
415	Err(())
416}

418/// Helper function to detect APIC by the Multiprocessor Specification
419fn detect_from_mp() -> Result<PhysAddr, ()> {
420	let mp_float = if let Ok(mpf) = search_mp_floating(
421		AddrRange::new(PhysAddr::new(0x9f000u64), PhysAddr::new(0xa0000u64)).unwrap(),
422	) {
423		Ok(mpf)
424	} else if let Ok(mpf) = search_mp_floating(
425		AddrRange::new(PhysAddr::new(0xf0000u64), PhysAddr::new(0x10_0000u64)).unwrap(),
426	) {
427		Ok(mpf)
428	} else {
429		Err(())
430	}?;
431
432	info!("Found MP config at {:#x}", { mp_float.mp_config });
433	info!(
434		"System uses Multiprocessing Specification 1.{}",
435		mp_float.version
436	);
437	info!("MP features 1: {}", mp_float.features[0]);
438
439	if mp_float.features[1] & 0x80 > 0 {
440		info!("PIC mode implemented");
441	} else {
442		info!("Virtual-Wire mode implemented");
443	}
444
445	let layout = PageLayout::from_size(BasePageSize::SIZE as usize).unwrap();
446	let page_range = PageBox::new(layout).unwrap();
447	let virtual_address = VirtAddr::from(page_range.start());
448
449	let mut flags = PageTableEntryFlags::empty();
450	flags.normal().writable();
451	paging::map::<BasePageSize>(
452		virtual_address,
453		PhysAddr::from((mp_float.mp_config as usize).align_down(BasePageSize::SIZE as usize)),
454		1,
455		flags,
456	);
457
458	let mut addr: usize =
459		(virtual_address | (u64::from(mp_float.mp_config) & (BasePageSize::SIZE - 1))) as usize;
460	let mp_config: &ApicConfigTable = unsafe { &*(ptr::with_exposed_provenance(addr)) };
461	if mp_config.signature != MP_CONFIG_SIGNATURE {
462		warn!("MP config table invalid!");
463		return Err(());
464	}
465
466	if mp_config.entry_count == 0 {
467		warn!("No MP table entries, guessing IOAPIC...");
468		let default_address = PhysAddr::new(0xfec0_0000);
469
470		init_ioapic_address(default_address);
471	} else {
472		// entries starts directly after the config table
473		addr += mem::size_of::<ApicConfigTable>();
474		for _i in 0..mp_config.entry_count {
475			match unsafe { *(ptr::with_exposed_provenance::<u8>(addr)) } {
476				// CPU entry
477				0 => {
478					let cpu_entry: &ApicProcessorEntry =
479						unsafe { &*(ptr::with_exposed_provenance(addr)) };
480					if cpu_entry.cpu_flags & 0x01 == 0x01 {
481						add_local_apic_id(cpu_entry.id);
482					}
483					addr += mem::size_of::<ApicProcessorEntry>();
484				}
485				// IO-APIC entry
486				2 => {
487					let io_entry: &ApicIoEntry = unsafe { &*(ptr::with_exposed_provenance(addr)) };
488					let ioapic = PhysAddr::new(io_entry.addr.into());
489					info!("IOAPIC found at {ioapic:p}");
490
491					init_ioapic_address(ioapic);
492
493					addr += mem::size_of::<ApicIoEntry>();
494				}
495				_ => {
496					addr += 8;
497				}
498			}
499		}
500	}
501
502	Ok(PhysAddr::new(mp_config.lapic.into()))
503}

505fn default_apic() -> PhysAddr {
506	let default_address = PhysAddr::new(0xfee0_0000);
507
508	warn!("Using default APIC address: {default_address:p}");
509
510	// currently, uhyve doesn't support an IO-APIC
511	if !env::is_uhyve() {
512		init_ioapic_address(default_address);
513	}
514
515	default_address
516}

522pub fn init() {
523	// Detect CPUs and APICs.
524	let local_apic_physical_address = if env::is_uhyve() {
525		default_apic()
526	} else {
527		detect_from_acpi()
528			.or_else(|()| detect_from_mp())
529			.unwrap_or_else(|()| default_apic())
530	};
531
532	// Initialize x2APIC or xAPIC, depending on what's available.
533	if processor::supports_x2apic() {
534		init_x2apic();
535	} else if env::is_uefi() {
536		// already id mapped in UEFI systems, just use the physical address as virtual one
537		LOCAL_APIC_ADDRESS
538			.set(VirtAddr::new(local_apic_physical_address.as_u64()))
539			.unwrap();
540	} else {
541		// We use the traditional xAPIC mode available on all x86-64 CPUs.
542		// It uses a mapped page for communication.
543		let layout = PageLayout::from_size(BasePageSize::SIZE as usize).unwrap();
544		let page_range = PageAlloc::allocate(layout).unwrap();
545		let local_apic_address = VirtAddr::from(page_range.start());
546		LOCAL_APIC_ADDRESS.set(local_apic_address).unwrap();
547		debug!(
548			"Mapping Local APIC at {local_apic_physical_address:p} to virtual address {local_apic_address:p}"
549		);
550
551		let mut flags = PageTableEntryFlags::empty();
552		flags.device().writable().execute_disable();
553		paging::map::<BasePageSize>(local_apic_address, local_apic_physical_address, 1, flags);
554	}
555
556	// Set gates to ISRs for the APIC interrupts we are going to enable.
557	unsafe {
558		let mut idt = IDT.lock();
559		idt[ERROR_INTERRUPT_NUMBER]
560			.set_handler_fn(error_interrupt_handler)
561			.set_stack_index(0);
562		idt[SPURIOUS_INTERRUPT_NUMBER]
563			.set_handler_fn(spurious_interrupt_handler)
564			.set_stack_index(0);
565		#[cfg(feature = "smp")]
566		{
567			idt[TLB_FLUSH_INTERRUPT_NUMBER]
568				.set_handler_fn(tlb_flush_handler)
569				.set_stack_index(0);
570			interrupts::add_irq_name(TLB_FLUSH_INTERRUPT_NUMBER - 32, "TLB flush");
571			idt[WAKEUP_INTERRUPT_NUMBER]
572				.set_handler_fn(wakeup_handler)
573				.set_stack_index(0);
574			interrupts::add_irq_name(WAKEUP_INTERRUPT_NUMBER - 32, "Wakeup");
575		}
576	}
577
578	// Initialize interrupt handling over APIC.
579	// All interrupts of the PIC have already been masked, so it doesn't need to be disabled again.
580	init_local_apic();
581
582	if !processor::supports_tsc_deadline() {
583		// We have an older APIC Timer without TSC Deadline support, which has a maximum timeout
584		// and needs to be calibrated.
585		calibrate_timer();
586	}
587
588	// currently, IO-APIC isn't supported by uhyve
589	if !env::is_uhyve() {
590		// initialize IO-APIC
591		init_ioapic();
592	}
593}

595fn init_ioapic() {
596	let max_entry = ioapic_max_redirection_entry() + 1;
597	info!("IOAPIC v{} has {} entries", ioapic_version(), max_entry);
598
599	// now lets turn everything else on
600	for i in 0..max_entry {
601		// Turn off the Programmable Interrupt Timer Interrupt (IRQ 0) and
602		// the Real Time Clock (IRQ 2).
603		let enabled = !matches!(i, 0 | 2);
604		ioapic_set_interrupt(i, 0, enabled);
605	}
606}
607
608fn ioapic_set_interrupt(irq: u8, apicid: u8, enabled: bool) {
609	assert!(irq <= 24);
610
611	let off = u32::from(irq * 2);
612	let ioredirect_upper = u32::from(apicid) << 24;
613	let mut ioredirect_lower = u32::from(0x20 + irq);
614	if !enabled {
615		debug!("Disabling irq {irq}");
616		ioredirect_lower |= 1 << 16;
617	}
618
619	ioapic_write(IOAPIC_REG_TABLE + off, ioredirect_lower);
620	ioapic_write(IOAPIC_REG_TABLE + off + 1, ioredirect_upper);
621}

623pub fn init_local_apic() {
624	// Mask out all interrupts we don't need right now.
625	local_apic_write(IA32_X2APIC_LVT_TIMER, APIC_LVT_MASK);
626	local_apic_write(IA32_X2APIC_LVT_THERMAL, APIC_LVT_MASK);
627	local_apic_write(IA32_X2APIC_LVT_PMI, APIC_LVT_MASK);
628	local_apic_write(IA32_X2APIC_LVT_LINT0, APIC_LVT_MASK);
629	local_apic_write(IA32_X2APIC_LVT_LINT1, APIC_LVT_MASK);
630
631	// Set the interrupt number of the Error interrupt.
632	local_apic_write(IA32_X2APIC_LVT_ERROR, u64::from(ERROR_INTERRUPT_NUMBER));
633
634	// allow all interrupts
635	local_apic_write(IA32_X2APIC_TPR, 0x00);
636
637	// Finally, enable the Local APIC by setting the interrupt number for spurious interrupts
638	// and providing the enable bit.
639	local_apic_write(
640		IA32_X2APIC_SIVR,
641		APIC_SIVR_ENABLED | (u64::from(SPURIOUS_INTERRUPT_NUMBER)),
642	);
643}

645fn calibrate_timer() {
646	// The APIC Timer is used to provide a one-shot interrupt for the tickless timer
647	// implemented through processor::get_timer_ticks.
648	// Therefore determine a counter value for 1 microsecond, which is the resolution
649	// used throughout all of Hermit. Wait 30ms for accuracy.
650	let microseconds = 30_000;
651
652	// Be sure that all interrupts for calibration accuracy and initialize the counter are disabled.
653	// Dividing the counter value by 8 still provides enough accuracy for 1 microsecond resolution,
654	// but allows for longer timeouts than a smaller divisor.
655	// For example, on an Intel Xeon E5-2650 v3 @ 2.30GHz, the counter is usually calibrated to
656	// 125, which allows for timeouts of approximately 34 seconds (u32::MAX / 125).
657
658	local_apic_write(IA32_X2APIC_DIV_CONF, APIC_DIV_CONF_DIVIDE_BY_8);
659	local_apic_write(IA32_X2APIC_INIT_COUNT, u64::from(u32::MAX));
660
661	// Wait until the calibration time has elapsed.
662	processor::udelay(microseconds);
663
664	// Save the difference of the initial value and current value as the result of the calibration
665	// and re-enable interrupts.
666	let calibrated_counter_value =
667		(u64::from(u32::MAX - local_apic_read(IA32_X2APIC_CUR_COUNT))) / microseconds;
668	CALIBRATED_COUNTER_VALUE
669		.set(calibrated_counter_value)
670		.unwrap();
671	debug!(
672		"Calibrated APIC Timer with a counter value of {calibrated_counter_value} for 1 microsecond",
673	);
674}

676fn __set_oneshot_timer(wakeup_time: Option<u64>) {
677	if let Some(wt) = wakeup_time {
678		if processor::supports_tsc_deadline() {
679			// wt is the absolute wakeup time in microseconds based on processor::get_timer_ticks.
680			// We can simply multiply it by the processor frequency to get the absolute Time-Stamp Counter deadline
681			// (see processor::get_timer_ticks).
682			let tsc_deadline = wt * (u64::from(processor::get_frequency()));
683
684			// Enable the APIC Timer in TSC-Deadline Mode and let it start by writing to the respective MSR.
685			local_apic_write(
686				IA32_X2APIC_LVT_TIMER,
687				APIC_LVT_TIMER_TSC_DEADLINE | u64::from(TIMER_INTERRUPT_NUMBER),
688			);
689			let mut ia32_tsc_deadline = IA32_TSC_DEADLINE;
690			unsafe {
691				ia32_tsc_deadline.write(tsc_deadline);
692			}
693		} else {
694			// Calculate the relative timeout from the absolute wakeup time.
695			// Maintain a minimum value of one tick, otherwise the timer interrupt does not fire at all.
696			// The Timer Counter Register is also a 32-bit register, which we must not overflow for longer timeouts.
697			let current_time = processor::get_timer_ticks();
698			let ticks = if wt > current_time {
699				wt - current_time
700			} else {
701				1
702			};
703			let init_count = cmp::min(
704				CALIBRATED_COUNTER_VALUE.get().unwrap() * ticks,
705				u64::from(u32::MAX),
706			);
707
708			// Enable the APIC Timer in One-Shot Mode and let it start by setting the initial counter value.
709			local_apic_write(IA32_X2APIC_LVT_TIMER, u64::from(TIMER_INTERRUPT_NUMBER));
710			local_apic_write(IA32_X2APIC_INIT_COUNT, init_count);
711		}
712	} else {
713		// Disable the APIC Timer.
714		local_apic_write(IA32_X2APIC_LVT_TIMER, APIC_LVT_MASK);
715	}
716}

718pub fn set_oneshot_timer(wakeup_time: Option<u64>) {
719	without_interrupts(|| {
720		__set_oneshot_timer(wakeup_time);
721	});

736/// Initialize the required _start variables for the next CPU to be booted.
737pub fn init_next_processor_variables() {
738	// Allocate stack for the CPU and pass the addresses.
739	let layout = Layout::from_size_align(KERNEL_STACK_SIZE, BasePageSize::SIZE as usize).unwrap();
740	let stack = unsafe { alloc(layout) };
741	assert!(!stack.is_null());
742	CURRENT_STACK_ADDRESS.store(stack, Ordering::Relaxed);
743}

745/// Boot all Application Processors
746/// This algorithm is derived from Intel MultiProcessor Specification 1.4, B.4, but testing has shown
747/// that a second STARTUP IPI and setting the BIOS Reset Vector are no longer necessary.
748/// This is partly confirmed by <https://wiki.osdev.org/Symmetric_Multiprocessing>
749#[cfg(all(target_os = "none", feature = "smp"))]
750pub fn boot_application_processors() {
751	use core::hint;
752
753	use x86_64::structures::paging::Translate;
754
755	use super::start;
756
757	let smp_boot_code = include_bytes!(concat!(core::env!("OUT_DIR"), "/boot.bin"));
758
759	// We shouldn't have any problems fitting the boot code into a single page, but let's better be sure.
760	assert!(
761		smp_boot_code.len() < BasePageSize::SIZE as usize,
762		"SMP Boot Code is larger than a page"
763	);
764	debug!("SMP boot code is {} bytes long", smp_boot_code.len());
765
766	if env::is_uefi() {
767		// Since UEFI already provides identity-mapped pagetables, we only have to sanity-check the identity mapping
768		let pt = unsafe { crate::arch::mm::paging::identity_mapped_page_table() };
769		let virt_addr = SMP_BOOT_CODE_ADDRESS;
770		let phys_addr = pt.translate_addr(virt_addr.into()).unwrap();
771		assert_eq!(phys_addr.as_u64(), virt_addr.as_u64());
772	} else {
773		// Identity-map the boot code page and copy over the code.
774		debug!("Mapping SMP boot code to physical and virtual address {SMP_BOOT_CODE_ADDRESS:p}");
775		let mut flags = PageTableEntryFlags::empty();
776		flags.normal().writable();
777		paging::map::<BasePageSize>(
778			SMP_BOOT_CODE_ADDRESS,
779			PhysAddr::new(SMP_BOOT_CODE_ADDRESS.as_u64()),
780			1,
781			flags,
782		);
783	}
784	unsafe {
785		ptr::copy_nonoverlapping(
786			smp_boot_code.as_ptr(),
787			SMP_BOOT_CODE_ADDRESS.as_mut_ptr(),
788			smp_boot_code.len(),
789		);
790	}
791
792	unsafe {
793		let (frame, val) = Cr3::read_raw();
794		let value = frame.start_address().as_u64() | u64::from(val);
795		// Pass the PML4 page table address to the boot code.
796		*((SMP_BOOT_CODE_ADDRESS + SMP_BOOT_CODE_OFFSET_PML4).as_mut_ptr::<u32>()) =
797			value.try_into().unwrap();
798		// Set entry point
799		debug!(
800			"Set entry point for application processor to {:p}",
801			start::_start as *const ()
802		);
803		ptr::write_unaligned(
804			(SMP_BOOT_CODE_ADDRESS + SMP_BOOT_CODE_OFFSET_ENTRY).as_mut_ptr(),
805			start::_start as usize,
806		);
807	}
808
809	// Now wake up each application processor.
810	let apic_ids = CPU_LOCAL_APIC_IDS.lock();
811	let core_id = core_id();
812
813	for (core_id_to_boot, &apic_id) in apic_ids.iter().enumerate() {
814		let core_id_to_boot = core_id_to_boot as u32;
815		if core_id_to_boot != core_id {
816			unsafe {
817				*((SMP_BOOT_CODE_ADDRESS + SMP_BOOT_CODE_OFFSET_CPU_ID).as_mut_ptr()) =
818					core_id_to_boot;
819			}
820			let destination = u64::from(apic_id) << 32;
821
822			debug!("Waking up CPU {core_id_to_boot} with Local APIC ID {apic_id}");
823			init_next_processor_variables();
824
825			// Save the current number of initialized CPUs.
826			let current_processor_count = arch::get_processor_count();
827
828			// Send an INIT IPI.
829			local_apic_write(
830				IA32_X2APIC_ICR,
831				destination
832					| APIC_ICR_LEVEL_TRIGGERED
833					| APIC_ICR_LEVEL_ASSERT
834					| APIC_ICR_DELIVERY_MODE_INIT,
835			);
836			processor::udelay(200);
837
838			local_apic_write(
839				IA32_X2APIC_ICR,
840				destination | APIC_ICR_LEVEL_TRIGGERED | APIC_ICR_DELIVERY_MODE_INIT,
841			);
842			processor::udelay(10000);
843
844			// Send a STARTUP IPI.
845			local_apic_write(
846				IA32_X2APIC_ICR,
847				destination
848					| APIC_ICR_DELIVERY_MODE_STARTUP
849					| ((SMP_BOOT_CODE_ADDRESS.as_u64()) >> 12),
850			);
851			debug!("Waiting for it to respond");
852
853			// Wait until the application processor has finished initializing.
854			// It will indicate this by counting up cpu_online.
855			while current_processor_count == arch::get_processor_count() {
856				hint::spin_loop();
857			}
858		}
859	}
860
861	print_information();
862}

864#[cfg(feature = "smp")]
865pub fn ipi_tlb_flush() {
866	if arch::get_processor_count() > 1 {
867		let apic_ids = CPU_LOCAL_APIC_IDS.lock();
868		let core_id = core_id();
869
870		// Ensure that all memory operations have completed before issuing a TLB flush.
871		unsafe {
872			_mm_mfence();
873		}
874
875		// Send an IPI with our TLB Flush interrupt number to all other CPUs.
876		without_interrupts(|| {
877			for (core_id_to_interrupt, &apic_id) in apic_ids.iter().enumerate() {
878				if core_id_to_interrupt != usize::try_from(core_id).unwrap() {
879					let destination = u64::from(apic_id) << 32;
880					local_apic_write(
881						IA32_X2APIC_ICR,
882						destination
883							| APIC_ICR_LEVEL_ASSERT
884							| APIC_ICR_DELIVERY_MODE_FIXED
885							| u64::from(TLB_FLUSH_INTERRUPT_NUMBER),
886					);
887				}
888			}
889		});
890	}
891}

893/// Send an inter-processor interrupt to wake up a CPU Core that is in a HALT state.
894#[allow(unused_variables)]
895pub fn wakeup_core(core_id_to_wakeup: CoreId) {
896	#[cfg(all(feature = "smp", not(feature = "idle-poll")))]
897	if core_id_to_wakeup != core_id()
898		&& !crate::processor::supports_mwait()
899		&& crate::scheduler::take_core_hlt_state(core_id_to_wakeup)
900	{
901		without_interrupts(|| {
902			let apic_ids = CPU_LOCAL_APIC_IDS.lock();
903			let local_apic_id = apic_ids[core_id_to_wakeup as usize];
904			let destination = u64::from(local_apic_id) << 32;
905			local_apic_write(
906				IA32_X2APIC_ICR,
907				destination
908					| APIC_ICR_LEVEL_ASSERT
909					| APIC_ICR_DELIVERY_MODE_FIXED
910					| u64::from(WAKEUP_INTERRUPT_NUMBER),
911			);
912		});
913	}
914}

916/// Translate the x2APIC MSR into an xAPIC memory address.
917#[inline]
918fn translate_x2apic_msr_to_xapic_address(x2apic_msr: u32) -> VirtAddr {
919	*LOCAL_APIC_ADDRESS.get().unwrap() + ((u64::from(x2apic_msr) & 0xff) << 4)
920}

922fn local_apic_read(x2apic_msr: u32) -> u32 {
923	if processor::supports_x2apic() {
924		// x2APIC is simple, we can just read from the given MSR.
925		unsafe { Msr::new(x2apic_msr).read() as u32 }
926	} else {
927		unsafe { *(translate_x2apic_msr_to_xapic_address(x2apic_msr).as_ptr::<u32>()) }
928	}
929}

932	unsafe {
933		core::ptr::write_volatile(IOAPIC_ADDRESS.get().unwrap().as_mut_ptr::<u32>(), reg);
934		core::ptr::write_volatile(
935			(*IOAPIC_ADDRESS.get().unwrap() + 4 * mem::size_of::<u32>()).as_mut_ptr::<u32>(),
936			value,
937		);
938	}
939}

941fn ioapic_read(reg: u32) -> u32 {
942	let value;
943
944	unsafe {
945		core::ptr::write_volatile(IOAPIC_ADDRESS.get().unwrap().as_mut_ptr::<u32>(), reg);
946		value = core::ptr::read_volatile(
947			(*IOAPIC_ADDRESS.get().unwrap() + 4 * mem::size_of::<u32>()).as_ptr::<u32>(),
948		);
949	}
950
951	value
952}

954fn ioapic_version() -> u32 {
955	ioapic_read(IOAPIC_REG_VER) & 0xff
956}
957
958fn ioapic_max_redirection_entry() -> u8 {
959	((ioapic_read(IOAPIC_REG_VER) >> 16) & 0xff) as u8
960}
961
962fn local_apic_write(x2apic_msr: u32, value: u64) {
963	if processor::supports_x2apic() {
964		// x2APIC is simple, we can just write the given value to the given MSR.
965		unsafe {
966			Msr::new(x2apic_msr).write(value);
967		}
968	} else {
969		// Write the value.
970		let value_ref = unsafe {
971			&mut *(translate_x2apic_msr_to_xapic_address(x2apic_msr).as_mut_ptr::<u32>())
972		};
973
974		if x2apic_msr == IA32_X2APIC_ICR {
975			// The ICR1 register in xAPIC mode also has a Delivery Status bit.
976			// Wait until previous interrupt was delivered.
977			// This bit does not exist in x2APIC mode (cf. Intel Vol. 3A, 10.12.9).
978			while (unsafe { core::ptr::read_volatile(value_ref) }
979				& APIC_ICR_DELIVERY_STATUS_PENDING)
980				> 0
981			{
982				spin_loop();
983			}
984
985			// Instead of a single 64-bit ICR register, xAPIC has two 32-bit registers (ICR1 and ICR2).
986			// There is a gap between them and the destination field in ICR2 is also 8 bits instead of 32 bits.
987			let destination = ((value >> 8) & 0xff00_0000) as u32;
988			let icr2 = unsafe {
989				&mut *((*LOCAL_APIC_ADDRESS.get().unwrap() + APIC_ICR2).as_mut_ptr::<u32>())
990			};
991			*icr2 = destination;
992
993			// The remaining data without the destination will now be written into ICR1.
994		}
995
996		*value_ref = value as u32;
997	}
998}
