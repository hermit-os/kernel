91/// Real Boot Processor initialization as soon as we have put the first Welcome message on the screen.
92#[cfg(target_os = "none")]
93pub fn boot_processor_init() {
94	processor::detect_features();
95	processor::configure();
96
97	if cfg!(feature = "vga") && !env::is_uhyve() {
98		#[cfg(feature = "vga")]
99		vga::init();
100	}
101
102	crate::mm::init();
103	crate::mm::print_information();
104	CoreLocal::get().add_irq_counter();
105	env::init();
106	gdt::add_current_core();
107	interrupts::load_idt();
108	pic::init();
109
110	processor::detect_frequency();
111	crate::logging::KERNEL_LOGGER.set_time(true);
112	processor::print_information();
113	debug!("Cr0 = {:?}", Cr0::read());
114	debug!("Cr4 = {:?}", Cr4::read());
115	interrupts::install();
116	systemtime::init();
117
118	if !env::is_uhyve() {
119		#[cfg(feature = "acpi")]
120		acpi::init();
121	}
122	if is_uhyve_with_pci() || !is_uhyve() {
123		#[cfg(feature = "pci")]
124		pci::init();
125	}
126
127	apic::init();
128	scheduler::install_timer_handler();
129	finish_processor_init();
130}

132/// Application Processor initialization
133#[cfg(all(target_os = "none", feature = "smp"))]
134pub fn application_processor_init() {
135	CoreLocal::install();
136	processor::configure();
137	gdt::add_current_core();
138	interrupts::load_idt();
139	if processor::supports_x2apic() {
140		apic::init_x2apic();
141	}
142	apic::init_local_apic();
143	debug!("Cr0 = {:?}", Cr0::read());
144	debug!("Cr4 = {:?}", Cr4::read());
145	finish_processor_init();
146}

148fn finish_processor_init() {
149	if env::is_uhyve() {
150		// uhyve does not use apic::detect_from_acpi and therefore does not know the number of processors and
151		// their APIC IDs in advance.
152		// Therefore, we have to add each booted processor into the CPU_LOCAL_APIC_IDS vector ourselves.
153		// Fortunately, the Local APIC IDs of uhyve are sequential and therefore match the Core IDs.
154		apic::add_local_apic_id(core_id() as u8);
155
156		// uhyve also boots each processor into _start itself and does not use apic::boot_application_processors.
157		// Therefore, the current processor already needs to prepare the processor variables for a possible next processor.
158		apic::init_next_processor_variables();
159	}
160}

162pub fn boot_next_processor() {
163	// This triggers apic::boot_application_processors (bare-metal/QEMU) or uhyve
164	// to initialize the next processor.
165	let cpu_online = CPU_ONLINE.fetch_add(1, Ordering::Release);
166
167	if !env::is_uhyve() {
168		if cpu_online == 0 {
169			#[cfg(all(target_os = "none", feature = "smp"))]
170			apic::boot_application_processors();
171		}
172
173		if !cfg!(feature = "smp") {
174			apic::print_information();
175		}
176	}
177}

190#[cfg(target_os = "none")]
191#[inline(never)]
192#[unsafe(no_mangle)]
193unsafe extern "C" fn pre_init(boot_info: Option<&'static RawBootInfo>, cpu_id: u32) -> ! {
194	use x86_64::registers::control::Cr0Flags;
195
196	// Enable caching
197	unsafe {
198		Cr0::update(|flags| flags.remove(Cr0Flags::CACHE_DISABLE | Cr0Flags::NOT_WRITE_THROUGH));
199	}
200
201	if cpu_id == 0 {
202		env::set_boot_info(*boot_info.unwrap());
203
204		crate::boot_processor_main()
205	} else {
206		#[cfg(not(feature = "smp"))]
207		{
208			let style = anstyle::Style::new().fg_color(Some(anstyle::AnsiColor::Red.into()));
209			let preamble = format_args!("[            ][{cpu_id}][{style}ERROR{style:#}]");
210			println!(
211				"{preamble} Secondary core booted, but Hermit was not built with SMP support!"
212			);
213			loop {
214				processor::halt();
215			}
216		}
217		#[cfg(feature = "smp")]
218		crate::application_processor_main();
219	}
220}

223const LOADER_START: usize = 0x0100_0000_0000;
224#[cfg(feature = "common-os")]
225const LOADER_STACK_SIZE: usize = 0x8000;

227#[cfg(feature = "common-os")]
228pub fn load_application<F, T>(code_size: u64, tls_size: u64, func: F) -> T
229where
230	F: FnOnce(&'static mut [u8], Option<&'static mut [u8]>) -> T,
231{
232	use core::ptr::slice_from_raw_parts_mut;
233
234	use align_address::Align;
235	use free_list::PageLayout;
236	use x86_64::structures::paging::{PageSize, Size4KiB as BasePageSize};
237
238	use crate::arch::x86_64::mm::paging::{self, PageTableEntryFlags, PageTableEntryFlagsExt};
239	use crate::mm::{FrameAlloc, PageRangeAllocator};
240
241	let code_size = (code_size as usize + LOADER_STACK_SIZE).align_up(BasePageSize::SIZE as usize);
242	let layout = PageLayout::from_size_align(code_size, BasePageSize::SIZE as usize).unwrap();
243	let frame_range = FrameAlloc::allocate(layout).unwrap();
244	let physaddr = PhysAddr::from(frame_range.start());
245
246	let mut flags = PageTableEntryFlags::empty();
247	flags.normal().writable().user().execute_enable();
248	paging::map::<BasePageSize>(
249		VirtAddr::from(LOADER_START),
250		physaddr,
251		code_size / BasePageSize::SIZE as usize,
252		flags,
253	);
254
255	let code_slice = unsafe { &mut *slice_from_raw_parts_mut(LOADER_START as *mut u8, code_size) };
256
257	if tls_size > 0 {
258		// To access TLS blocks on x86-64, TLS offsets are *subtracted* from the thread register value.
259		// So the thread pointer needs to be `block_ptr + tls_offset`.
260		// GNU style TLS requires `fs:0` to represent the same address as the thread pointer.
261		// Since the thread pointer points to the end of the TLS blocks, we need to store it there.
262		let tcb_size = core::mem::size_of::<*mut ()>();
263		let tls_offset = tls_size as usize;
264
265		let tls_memsz = (tls_offset + tcb_size).align_up(BasePageSize::SIZE as usize);
266		let layout = PageLayout::from_size(tls_memsz).unwrap();
267		let frame_range = FrameAlloc::allocate(layout).unwrap();
268		let physaddr = PhysAddr::from(frame_range.start());
269
270		let mut flags = PageTableEntryFlags::empty();
271		flags.normal().writable().user().execute_disable();
272		let tls_virt = VirtAddr::from(LOADER_START + code_size + BasePageSize::SIZE as usize);
273		paging::map::<BasePageSize>(
274			tls_virt,
275			physaddr,
276			tls_memsz / BasePageSize::SIZE as usize,
277			flags,
278		);
279		let block =
280			unsafe { &mut *slice_from_raw_parts_mut(tls_virt.as_mut_ptr(), tls_offset + tcb_size) };
281		for elem in block.iter_mut() {
282			*elem = 0;
283		}
284
285		// thread_ptr = block_ptr + tls_offset
286		let thread_ptr = block[tls_offset..].as_mut_ptr().cast::<()>();
287		unsafe {
288			thread_ptr.cast::<*mut ()>().write(thread_ptr);
289		}
290		crate::arch::x86_64::kernel::processor::writefs(thread_ptr as usize);
291
292		func(code_slice, Some(block))
293	} else {
294		func(code_slice, None)
295	}
296}

299pub unsafe fn jump_to_user_land(entry_point: usize, code_size: usize, arg: &[&str]) -> ! {
300	use alloc::ffi::CString;
301
302	use align_address::Align;
303	use x86_64::structures::paging::{PageSize, Size4KiB as BasePageSize};
304
305	use crate::arch::x86_64::kernel::scheduler::TaskStacks;
306
307	info!("Create new file descriptor table");
308	core_scheduler().recreate_objmap().unwrap();
309
310	let entry_point: usize = LOADER_START | entry_point;
311	let stack_pointer: usize = LOADER_START
312		+ (code_size + LOADER_STACK_SIZE).align_up(BasePageSize::SIZE.try_into().unwrap())
313		- 8;
314
315	let stack_pointer =
316		stack_pointer - 128 /* red zone */ - arg.len() * core::mem::size_of::<*mut u8>();
317	let argv = unsafe { core::slice::from_raw_parts_mut(stack_pointer as *mut *mut u8, arg.len()) };
318	let len = arg.iter().fold(0, |acc, x| acc + x.len() + 1);
319	// align stack pointer to fulfill the requirements of the x86_64 ABI
320	let stack_pointer = (stack_pointer - len).align_down(16) - core::mem::size_of::<usize>();
321
322	let mut pos: usize = 0;
323	for (i, s) in arg.iter().enumerate() {
324		if let Ok(s) = CString::new(*s) {
325			let bytes = s.as_bytes_with_nul();
326			argv[i] = (stack_pointer + pos) as *mut u8;
327			pos += bytes.len();
328
329			unsafe {
330				core::ptr::copy_nonoverlapping(bytes.as_ptr(), argv[i], bytes.len());
331			}
332		} else {
333			panic!("Unable to create C string!");
334		}
335	}
336
337	debug!("Jump to user space at 0x{entry_point:x}, stack pointer 0x{stack_pointer:x}");
338
339	unsafe {
340		asm!(
341			"and rsp, {0}",
342			"swapgs",
343			"push {1}",
344			"push {2}",
345			"push {3}",
346			"push {4}",
347			"push {5}",
348			"mov rdi, {6}",
349			"mov rsi, {7}",
350			"iretq",
351			const u64::MAX - (TaskStacks::MARKER_SIZE as u64 - 1),
352			const 0x23usize,
353			in(reg) stack_pointer,
354			const 0x1202u64,
355			const 0x2busize,
356			in(reg) entry_point,
357			in(reg) argv.len(),
358			in(reg) argv.as_ptr(),
359			options(nostack, noreturn)
360		);
361	}
