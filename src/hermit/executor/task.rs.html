10#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
11struct AsyncTaskId(u32);
12
13impl fmt::Display for AsyncTaskId {
14	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
15		write!(f, "{}", self.0)
16	}
17}
18
19impl AsyncTaskId {
20	fn new() -> Self {
21		static NEXT_ID: AtomicU32 = AtomicU32::new(0);
22		AsyncTaskId(NEXT_ID.fetch_add(1, Ordering::Relaxed))
23	}
24}
25
26pub(crate) struct AsyncTask {
27	id: AsyncTaskId,
28	future: Pin<Box<dyn Future<Output = ()> + Send>>,
29}
30
31impl AsyncTask {
32	pub fn new(future: impl Future<Output = ()> + Send + 'static) -> AsyncTask {
33		AsyncTask {
34			id: AsyncTaskId::new(),
35			future: Box::pin(future),
36		}
37	}
38}
39
40impl Future for AsyncTask {
41	type Output = ();
42
43	fn poll(mut self: Pin<&mut Self>, cx: &mut Context<'_>) -> Poll<Self::Output> {
44		trace!("Run async task {}", self.id);
45		self.as_mut().future.as_mut().poll(cx)
46	}
47}
