90/// The Virtq trait unifies access to the two different Virtqueue types
91/// [packed::PackedVq] and [split::SplitVq].
92///
93/// The trait provides a common interface for both types. Which in some case
94/// might not provide the complete feature set of each queue. Drivers who
95/// do need these features should refrain from providing support for both
96/// Virtqueue types and use the structs directly instead.
97#[enum_dispatch]
98pub trait Virtq: Send {
99	/// The `notif` parameter indicates if the driver wants to have a notification for this specific
100	/// transfer. This is only for performance optimization. As it is NOT ensured, that the device sees the
101	/// updated notification flags before finishing transfers!
102	fn dispatch(
103		&mut self,
104		tkn: AvailBufferToken,
105		notif: bool,
106		buffer_type: BufferType,
107	) -> Result<(), VirtqError>;
108
109	/// Dispatches the provided TransferToken to the respective queue and does
110	/// return when, the queue finished the transfer.
111	///
112	/// The returned [UsedBufferToken] can be copied from
113	/// or return the underlying buffers.
114	///
115	/// **INFO:**
116	/// Currently this function is constantly polling the queue while keeping the notifications disabled.
117	/// Upon finish notifications are enabled again.
118	fn dispatch_blocking(
119		&mut self,
120		tkn: AvailBufferToken,
121		buffer_type: BufferType,
122	) -> Result<UsedBufferToken, VirtqError> {
123		self.dispatch(tkn, false, buffer_type)?;
124
125		self.disable_notifs();
126
127		let result: UsedBufferToken;
128		// Keep Spinning until the receive queue is filled
129		loop {
130			// TODO: normally, we should check if the used buffer in question is the one
131			// we just made available. However, this shouldn't be a problem as the queue this
132			// function is called on makes use of this blocking dispatch function exclusively
133			// and thus dispatches cannot be interleaved.
134			if let Ok(buffer_tkn) = self.try_recv() {
135				result = buffer_tkn;
136				break;
137			}
138		}
139
140		self.enable_notifs();
141
142		Ok(result)
143	}
144
145	/// Enables interrupts for this virtqueue upon receiving a transfer
146	fn enable_notifs(&mut self);
147
148	/// Disables interrupts for this virtqueue upon receiving a transfer
149	fn disable_notifs(&mut self);
150
151	/// Checks if new used descriptors have been written by the device.
152	/// This activates the queue and polls the descriptor ring of the queue.
153	fn try_recv(&mut self) -> Result<UsedBufferToken, VirtqError>;
154
155	/// Dispatches a batch of [AvailBufferToken]s. The buffers are provided to the queue in
156	/// sequence. After the last buffer has been written, the queue marks the first buffer as available and triggers
157	/// a device notification if wanted by the device.
158	///
159	/// The `notif` parameter indicates if the driver wants to have a notification for this specific
160	/// transfer. This is only for performance optimization. As it is NOT ensured, that the device sees the
161	/// updated notification flags before finishing transfers!
162	fn dispatch_batch(
163		&mut self,
164		tkns: Vec<(AvailBufferToken, BufferType)>,
165		notif: bool,
166	) -> Result<(), VirtqError>;
167
168	/// Dispatches a batch of [AvailBufferToken]s. The tokens will be placed in to the `await_queue`
169	/// upon finish.
170	///
171	/// The `notif` parameter indicates if the driver wants to have a notification for this specific
172	/// transfer. This is only for performance optimization. As it is NOT ensured, that the device sees the
173	/// updated notification flags before finishing transfers!
174	///
175	/// The buffers are provided to the queue in
176	/// sequence. After the last buffer has been written, the queue marks the first buffer as available and triggers
177	/// a device notification if wanted by the device.
178	///
179	/// Tokens to get a reference to the provided await_queue, where they will be placed upon finish.
180	fn dispatch_batch_await(
181		&mut self,
182		tkns: Vec<(AvailBufferToken, BufferType)>,
183		notif: bool,
184	) -> Result<(), VirtqError>;
185
186	/// Returns the size of a Virtqueue. This represents the overall size and not the capacity the
187	/// queue currently has for new descriptors.
188	fn size(&self) -> VqSize;
189
190	// Returns the index (ID) of a Virtqueue.
191	fn index(&self) -> VqIndex;
192
193	fn has_used_buffers(&self) -> bool;
194}

196/// These methods are an implementation detail and are meant only for consumption by the default method
197/// implementations in [Virtq].
198trait VirtqPrivate {
199	type Descriptor: VirtqDescriptor;
200
201	fn create_indirect_ctrl(
202		buffer_tkn: &AvailBufferToken,
203	) -> Result<Box<[Self::Descriptor]>, VirtqError>;
204
205	fn indirect_desc(table: &[Self::Descriptor]) -> Self::Descriptor {
206		Self::Descriptor::incomplete_desc(
207			DeviceAlloc
208				.phys_addr_from(table.as_ptr().cast_mut())
209				.as_u64()
210				.into(),
211			(mem::size_of_val(table) as u32).into(),
212			virtq::DescF::INDIRECT,
213		)
214	}
215
216	/// Consumes the [AvailBufferToken] and returns a [TransferToken], that can be used to actually start the transfer.
217	///
218	/// After this call, the buffers are no longer writable.
219	fn transfer_token_from_buffer_token(
220		buff_tkn: AvailBufferToken,
221		buffer_type: BufferType,
222	) -> TransferToken<Self::Descriptor> {
223		let ctrl_desc = match buffer_type {
224			BufferType::Direct => None,
225			BufferType::Indirect => Some(Self::create_indirect_ctrl(&buff_tkn).unwrap()),
226		};
227
228		TransferToken {
229			buff_tkn,
230			ctrl_desc,
231		}
232	}
233
234	// The descriptors returned by the iterator will be incomplete, as they do not
235	// have all the information necessary.
236	fn descriptor_iter(
237		buffer_tkn: &AvailBufferToken,
238	) -> Result<impl DoubleEndedIterator<Item = Self::Descriptor>, VirtqError> {
239		let send_desc_iter = buffer_tkn
240			.send_buff
241			.iter()
242			.map(|elem| (elem, elem.len(), virtq::DescF::empty()));
243		let recv_desc_iter = buffer_tkn
244			.recv_buff
245			.iter()
246			.map(|elem| (elem, elem.capacity(), virtq::DescF::WRITE));
247		let mut all_desc_iter =
248			send_desc_iter
249				.chain(recv_desc_iter)
250				.map(|(mem_descr, len, incomplete_flags)| {
251					Self::Descriptor::incomplete_desc(
252						DeviceAlloc
253							.phys_addr_from(mem_descr.as_ptr().cast_mut())
254							.as_u64()
255							.into(),
256						len.into(),
257						incomplete_flags | virtq::DescF::NEXT,
258					)
259				});
260
261		let mut last_desc = all_desc_iter
262			.next_back()
263			.ok_or(VirtqError::BufferNotSpecified)?;
264		*last_desc.flags_mut() -= virtq::DescF::NEXT;
265
266		Ok(all_desc_iter.chain([last_desc]))
267	}
268}

312/// The struct represents buffers which are ready to be send via the
313/// virtqueue. Buffers can no longer be written or retrieved.
314pub struct TransferToken<Descriptor> {
315	/// Must be some in order to prevent drop
316	/// upon reuse.
317	buff_tkn: AvailBufferToken,
318	// Contains the [MemDescr] for the indirect table if the transfer is indirect.
319	ctrl_desc: Option<Box<[Descriptor]>>,
320}
321
322/// Public Interface for TransferToken
323impl<Descriptor> TransferToken<Descriptor> {
324	/// Returns the number of descritprors that will be placed in the queue.
325	/// This number can differ from the `BufferToken.num_descr()` function value
326	/// as indirect buffers only consume one descriptor in the queue, but can have
327	/// more descriptors that are accessible via the descriptor in the queue.
328	fn num_consuming_descr(&self) -> u16 {
329		if self.ctrl_desc.is_some() {
330			1
331		} else {
332			self.buff_tkn.num_descr()
333		}
334	}
335}

337#[derive(Debug)]
338pub enum BufferElem {
339	Sized(Box<dyn Any + Send, DeviceAlloc>),
340	Vector(Vec<u8, DeviceAlloc>),
341}
342
343impl BufferElem {
344	// Returns the initialized length of the element. Assumes [Self::Sized] to
345	// be initialized, since the type of the object is erased and we cannot
346	// detect if the content is actually a [MaybeUninit]. However, this function
347	// should be only relevant for read buffer elements, which should not be uninit.
348	// If the element belongs to a write buffer, it is likely that [Self::capacity]
349	// is more appropriate.
350	pub fn len(&self) -> u32 {
351		match self {
352			BufferElem::Sized(sized) => mem::size_of_val(sized.as_ref()),
353			BufferElem::Vector(vec) => vec.len(),
354		}
355		.try_into()
356		.unwrap()
357	}
358
359	pub fn capacity(&self) -> u32 {
360		match self {
361			BufferElem::Sized(sized) => mem::size_of_val(sized.as_ref()),
362			BufferElem::Vector(vec) => vec.capacity(),
363		}
364		.try_into()
365		.unwrap()
366	}
367
368	pub fn as_ptr(&self) -> *const u8 {
369		match self {
370			BufferElem::Sized(sized) => ptr::from_ref(sized.as_ref()).cast::<u8>(),
371			BufferElem::Vector(vec) => vec.as_ptr(),
372		}
373	}
374}

376/// The struct represents buffers which are ready to be written or to be send.
377///
378/// BufferTokens can be written in two ways:
379/// * in one step via `BufferToken.write()
380///   * consumes BufferToken and returns a TransferToken
381/// * sequentially via `BufferToken.write_seq()
382///
383/// # Structure of the Token
384/// The token can potentially hold both a *send* and a *recv* buffer, but MUST hold
385/// one.
386/// The *send* buffer is the data the device will read during a transfer, the *recv* buffer
387/// is the data the device will write to during a transfer.
388///
389/// # What are Buffers
390/// A buffer represents multiple chunks of memory. Where each chunk can be of different size.
391/// The chunks are named descriptors in the following.
392///
393/// **For Example:**
394/// A buffer could consist of 3 descriptors:
395/// 1. First descriptor of 30 bytes
396/// 2. Second descriptor of 10 bytes
397/// 3. Third descriptor of 100 bytes
398///
399/// Each of these descriptors consumes one "element" of the
400/// respective virtqueue.
401/// The maximum number of descriptors per buffer is bounded by the size of the virtqueue.
402pub struct AvailBufferToken {
403	pub(crate) send_buff: SmallVec<[BufferElem; 2]>,
404	pub(crate) recv_buff: SmallVec<[BufferElem; 2]>,
405}
406
407pub(crate) struct UsedDeviceWritableBuffer {
408	elems: SmallVec<[BufferElem; 2]>,
409	remaining_written_len: u32,
410}
411
412impl UsedDeviceWritableBuffer {
413	/// # Safety
414	/// The type `T` of the [`MaybeUninit<T>`] that was provided for the [BufferElem] must have
415	/// been what the device will write at that portion of the buffer. This may not be the case if
416	/// the descriptor is used by the device for another type (e.g. in the case of merged buffers
417	/// in the network driver), as the objects are assumed to be initialized by this function and
418	/// it's undefined behavior to call [MaybeUninit::assume_init] when an object of the correct type
419	/// is not initialized.
420	pub unsafe fn pop_front_downcast<T>(&mut self) -> Option<Box<T, DeviceAlloc>>
421	where
422		T: Any,
423	{
424		if self.remaining_written_len < u32::try_from(size_of::<T>()).unwrap() {
425			return None;
426		}
427
428		// May panic, but we have written data remaining so there should always be an item
429		let elem = if self.elems.len() <= 2 {
430			self.elems.swap_remove(0)
431		} else {
432			self.elems.remove(0)
433		};
434
435		if let BufferElem::Sized(sized) = elem {
436			match sized.downcast::<MaybeUninit<T>>() {
437				Ok(cast) => {
438					self.remaining_written_len -= u32::try_from(size_of::<T>()).unwrap();
439					Some(unsafe { cast.assume_init() })
440				}
441				Err(_) => {
442					panic!("Attempted to downcast element to wrong type");
443				}
444			}
445		} else {
446			panic!("Attempted to pop elements in order different from insertion");
447		}
448	}
449
450	pub fn pop_front_vec(&mut self) -> Option<Vec<u8, DeviceAlloc>> {
451		if self.elems.is_empty() {
452			return None;
453		}
454
455		let elem = if self.elems.len() <= 2 {
456			self.elems.swap_remove(0)
457		} else {
458			self.elems.remove(0)
459		};
460
461		if let BufferElem::Vector(mut vector) = elem {
462			let new_len = u32::min(
463				vector.capacity().try_into().unwrap(),
464				self.remaining_written_len,
465			);
466			self.remaining_written_len -= new_len;
467			unsafe { vector.set_len(new_len.try_into().unwrap()) };
468			Some(vector)
469		} else {
470			panic!("Attempted to pop elements in order different from insertion");
471		}
472	}
473
474	/// It is possible for devices to use descriptors for a type other than what they were meant.
475	/// (e.g. for a portion of the received frame in the network driver when [virtio::net::F::MRG_RXBUF] is negotiated).
476	/// In that case, we hand over the popped box directly with the used length.
477	///
478	/// We may not return a vector as its layout would be different and deallocation would not be correct
479	/// (see the information for [Box::into_non_null_with_allocator]).
480	pub fn pop_front_raw(&mut self) -> Option<(Box<dyn Any + Send, DeviceAlloc>, usize)> {
481		let elem = if self.elems.len() <= 2 {
482			self.elems.swap_remove(0)
483		} else {
484			self.elems.remove(0)
485		};
486
487		if let BufferElem::Sized(sized) = elem {
488			let capacity = u32::try_from(size_of_val(sized.as_ref())).unwrap();
489			let len = u32::min(capacity, self.remaining_written_len);
490			self.remaining_written_len -= len;
491			Some((sized, len.try_into().unwrap()))
492		} else {
493			panic!("This function is meant for the Sized variant of the BufferElem enum.");
494		}
495	}
496}

523// Public interface of BufferToken
524impl AvailBufferToken {
525	/// **Parameters**
526	/// * send: The slices that will make up the elements of the driver-writable buffer.
527	/// * recv: The slices that will make up the elements of the device-writable buffer.
528	///
529	/// **Reasons for Failure:**
530	/// * Both `send` and `recv` are empty, which is not allowed by Virtio.
531	///
532	/// * If one wants to have a structure in the style of:
533	/// ```
534	/// struct send_recv_struct {
535	///     // send_part: ...
536	///     // recv_part: ...
537	/// }
538	/// ```
539	/// they must split the structure after the send part and provide the respective part via the send argument and the respective other
540	/// part via the recv argument.
541	pub fn new(
542		send_buff: SmallVec<[BufferElem; 2]>,
543		recv_buff: SmallVec<[BufferElem; 2]>,
544	) -> Result<Self, VirtqError> {
545		if send_buff.is_empty() && recv_buff.is_empty() {
546			return Err(VirtqError::BufferNotSpecified);
547		}
548
549		Ok(Self {
550			send_buff,
551			recv_buff,
552		})
553	}
554}

556pub enum BufferType {
557	/// As many descriptors get consumed in the descriptor table as the sum of the numbers of slices in [AvailBufferToken::send_buff] and [AvailBufferToken::recv_buff].
558	Direct,
559	/// Results in one descriptor in the queue, hence consumes one element in the main descriptor table. The queue will merge the send and recv buffers as follows:
560	/// ```text
561	/// //+++++++++++++++++++++++
562	/// //+        Queue        +
563	/// //+++++++++++++++++++++++
564	/// //+ Indirect descriptor + -> refers to a descriptor list in the form of ->  ++++++++++++++++++++++++++
565	/// //+         ...         +                                                   +  Descriptors for send  +
566	/// //+++++++++++++++++++++++                                                   +  Descriptors for recv  +
567	/// //                                                                          ++++++++++++++++++++++++++
568	/// ```
569	/// As a result indirect descriptors result in a single descriptor consumption in the actual queue.
570	Indirect,
571}

577/// MemPool allows to easily control, request and provide memory for Virtqueues.
578///
579/// The struct is initialized with a limit of free running "tracked"
580/// memory descriptor ids. As Virtqueus do only allow a limited amount of descriptors in their queue,
581/// the independent queues, can control the number of descriptors by this.
582struct MemPool {
583	pool: Vec<MemDescrId>,
584	limit: u16,
585}
586
587impl MemPool {
588	/// Returns a given id to the id pool
589	fn ret_id(&mut self, id: MemDescrId) {
590		self.pool.push(id);
591	}
592
593	/// Returns a new instance, with a pool of the specified size.
594	fn new(size: u16) -> MemPool {
595		MemPool {
596			pool: (0..size).map(MemDescrId).collect(),
597			limit: size,
598		}
599	}

602/// Virtqeueus error module.
603///
604/// This module unifies errors provided to useres of a virtqueue, independent of the underlying
605/// virtqueue implementation, realized via the different enum variants.
606pub mod error {
607	use crate::errno::Errno;
608
609	#[derive(Debug)]
610	// Internal Error Handling for Buffers
611	pub enum BufferError {
612		WriteToLarge,
613		ToManyWrites,
614	}
615
616	// External Error Handling for users of the virtqueue.
617	pub enum VirtqError {
618		General,
619		/// Call to create a BufferToken or TransferToken without
620		/// any buffers to be inserted
621		BufferNotSpecified,
622		/// Selected queue does not exist or
623		/// is not known to the device and hence can not be used
624		QueueNotExisting(u16),
625		/// Signals, that the queue does not have any free descriptors
626		/// left.
627		/// Typically this means, that the driver either has to provide
628		/// "unsend" `TransferToken` to the queue (see Docs for details)
629		/// or the device needs to process available descriptors in the queue.
630		NoDescrAvail,
631		/// Indicates that a Bytes::new() call failed or generally that a buffer is to large to
632		/// be transferred as one. The Maximum size is u32::MAX. This also is the maximum for indirect
633		/// descriptors (both the one placed in the queue, as also the ones the indirect descriptor is
634		/// referring to).
635		BufferToLarge,
636		QueueSizeNotAllowed(u16),
637		FeatureNotSupported(virtio::F),
638		AllocationError,
639		IncompleteWrite,
640		NoNewUsed,
641	}
642
643	impl core::fmt::Debug for VirtqError {
644		fn fmt(&self, f: &mut core::fmt::Formatter<'_>) -> core::fmt::Result {
645			match self {
646				VirtqError::General => write!(f, "Virtq failure due to unknown reasons!"),
647				VirtqError::BufferNotSpecified => {
648					write!(f, "Virtq detected creation of Token, without a BuffSpec")
649				}
650				VirtqError::QueueNotExisting(_) => {
651					write!(f, "Virtq does not exist and can not be used!")
652				}
653				VirtqError::NoDescrAvail => write!(f, "Virtqs memory pool is exhausted!"),
654				VirtqError::BufferToLarge => {
655					write!(f, "Buffer to large for queue! u32::MAX exceeded.")
656				}
657				VirtqError::QueueSizeNotAllowed(_) => {
658					write!(f, "The requested queue size is not valid.")
659				}
660				VirtqError::FeatureNotSupported(_) => {
661					write!(f, "An unsupported feature was requested from the queue.")
662				}
663				VirtqError::AllocationError => write!(
664					f,
665					"An error was encountered during the allocation of the queue structures."
666				),
667				VirtqError::IncompleteWrite => {
668					write!(f, "A sized object was partially initialized.")
669				}
670				VirtqError::NoNewUsed => {
671					write!(f, "The queue does not contain any new used buffers.")
672				}
673			}
674		}
675	}
676
677	impl core::convert::From<VirtqError> for Errno {
678		fn from(_: VirtqError) -> Self {
679			Errno::Io
680		}
681	}
