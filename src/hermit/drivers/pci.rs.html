96	/// Returns the bar at bar-register `slot`.
97	pub fn get_bar(&self, slot: u8) -> Option<Bar> {
98		let header = self.header();
99		if let Some(endpoint) = EndpointHeader::from_header(header, &self.access) {
100			return endpoint.bar(slot, &self.access);
101		}
102
103		None
104	}

106	/// Configure the bar at register `slot`
107	pub fn set_bar(&self, slot: u8, bar: Bar) {
108		let value = match bar {
109			Bar::Io { port } => (port | 1) as usize,
110			Bar::Memory32 {
111				address,
112				size: _,
113				prefetchable,
114			} => {
115				if prefetchable {
116					(address | (1 << 3)) as usize
117				} else {
118					address as usize
119				}
120			}
121			Bar::Memory64 {
122				address,
123				size: _,
124				prefetchable,
125			} => {
126				if prefetchable {
127					(address | (2 << 1) | (1 << 3)) as usize
128				} else {
129					(address | (2 << 1)) as usize
130				}
131			}
132		};
133		let mut header = EndpointHeader::from_header(self.header(), &self.access).unwrap();
134		unsafe {
135			header.write_bar(slot, &self.access, value).unwrap();
136		}
137	}

139	/// Memory maps pci bar with specified index to identical location in virtual memory.
140	/// no_cache determines if we set the `Cache Disable` flag in the page-table-entry.
141	/// Returns (virtual-pointer, size) if successful, else None (if bar non-existent or IOSpace)
142	pub fn memory_map_bar(&self, index: u8, no_cache: bool) -> Option<(VirtAddr, usize)> {
143		let (address, size, prefetchable, width) = match self.get_bar(index) {
144			Some(Bar::Io { .. }) => {
145				warn!("Cannot map IOBar!");
146				return None;
147			}
148			Some(Bar::Memory32 {
149				address,
150				size,
151				prefetchable,
152			}) => (
153				u64::from(address),
154				usize::try_from(size).unwrap(),
155				prefetchable,
156				32,
157			),
158			Some(Bar::Memory64 {
159				address,
160				size,
161				prefetchable,
162			}) => (address, usize::try_from(size).unwrap(), prefetchable, 64),
163			_ => {
164				return None;
165			}
166		};
167
168		if address == 0 {
169			return None;
170		}
171
172		debug!("Mapping bar {index} at {address:#x} with length {size:#x}");
173
174		if width != 64 {
175			warn!("Currently only mapping of 64 bit bars is supported!");
176			return None;
177		}
178		if !prefetchable {
179			warn!("Currently only mapping of prefetchable bars is supported!");
180		}
181
182		// Since the bios/bootloader manages the physical address space, the address got from the bar is unique and not overlapping.
183		// We therefore do not need to reserve any additional memory in our kernel.
184		// Map bar into RW^X virtual memory
185		let physical_address = address;
186		let virtual_address =
187			crate::mm::map(PhysAddr::new(physical_address), size, true, true, no_cache);
188
189		Some((virtual_address, size))
190	}

192	pub fn get_irq(&self) -> Option<InterruptLine> {
193		let header = self.header();
194		if let Some(endpoint) = EndpointHeader::from_header(header, &self.access) {
195			let (_pin, line) = endpoint.interrupt(&self.access);
196			Some(line)
197		} else {
198			None
199		}
200	}
201
202	pub fn set_irq(&self, pin: InterruptPin, line: InterruptLine) {
203		let mut header = EndpointHeader::from_header(self.header(), &self.access).unwrap();
204		header.update_interrupt(&self.access, |(_pin, _line)| (pin, line));
205	}

208		let (_vendor_id, device_id) = self.header().id(&self.access);
209		device_id

226impl<T: ConfigRegionAccess> fmt::Display for PciDevice<T> {
227	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
228		let header = self.header();
229		let header_type = header.header_type(&self.access);
230		let (vendor_id, device_id) = header.id(&self.access);
231		let (_dev_rev, class_id, subclass_id, _interface) = header.revision_and_class(&self.access);
232
233		if let Some(endpoint) = EndpointHeader::from_header(header, &self.access) {
234			#[cfg(feature = "pci-ids")]
235			let (class_name, vendor_name, device_name) = {
236				use pci_ids::{Class, Device, FromId, Subclass};
237
238				let class_name = Class::from_id(class_id).map_or("Unknown Class", |class| {
239					class
240						.subclasses()
241						.find(|s| s.id() == subclass_id)
242						.map(Subclass::name)
243						.unwrap_or_else(|| class.name())
244				});
245
246				let (vendor_name, device_name) = Device::from_vid_pid(vendor_id, device_id)
247					.map(|device| (device.vendor().name(), device.name()))
248					.unwrap_or(("Unknown Vendor", "Unknown Device"));
249
250				(class_name, vendor_name, device_name)
251			};
252
253			#[cfg(not(feature = "pci-ids"))]
254			let (class_name, vendor_name, device_name) =
255				("Unknown Class", "Unknown Vendor", "Unknown Device");
256
257			// Output detailed readable information about this device.
258			write!(
259				f,
260				"{:02X}:{:02X} {} [{:02X}{:02X}]: {} {} [{:04X}:{:04X}]",
261				self.address.bus(),
262				self.address.device(),
263				class_name,
264				class_id,
265				subclass_id,
266				vendor_name,
267				device_name,
268				vendor_id,
269				device_id
270			)?;
271
272			// If the devices uses an IRQ, output this one as well.
273			let (_, irq) = endpoint.interrupt(&self.access);
274			if irq != 0 && irq != u8::MAX {
275				write!(f, ", IRQ {irq}")?;
276			}
277
278			let mut slot: u8 = 0;
279			while usize::from(slot) < MAX_BARS {
280				if let Some(pci_bar) = endpoint.bar(slot, &self.access) {
281					match pci_bar {
282						Bar::Memory64 {
283							address,
284							size,
285							prefetchable,
286						} => {
287							write!(
288								f,
289								", BAR{slot} Memory64 {{ address: {address:#X}, size: {size:#X}, prefetchable: {prefetchable} }}"
290							)?;
291							slot += 1;
292						}
293						Bar::Memory32 {
294							address,
295							size,
296							prefetchable,
297						} => {
298							write!(
299								f,
300								", BAR{slot} Memory32 {{ address: {address:#X}, size: {size:#X}, prefetchable: {prefetchable} }}"
301							)?;
302						}
303						Bar::Io { port } => {
304							write!(f, ", BAR{slot} IO {{ port: {port:#X} }}")?;
305						}
306					}
307				}
308				slot += 1;
309			}
310		} else {
311			// Output detailed readable information about this device.
312			write!(
313				f,
314				"{:02X}:{:02X} {:?} [{:04X}:{:04X}]",
315				self.address.bus(),
316				self.address.device(),
317				header_type,
318				vendor_id,
319				device_id
320			)?;
321		}
322
323		Ok(())
324	}
325}

422
423				let irq_number = drv.lock().get_interrupt_number();

425				(irq_number, vsock_handler)
426			}
427			#[cfg(all(
428				target_arch = "x86_64",
429				feature = "rtl8139",
430				any(feature = "tcp", feature = "udp")
431			))]
432			Self::RTL8139Net(drv) => {
433				fn rtl8139_handler() {
434					if let Some(driver) = get_network_driver() {
435						driver.lock().handle_interrupt();
436					}
437				}
438
439				let irq_number = drv.lock().get_interrupt_number();
440
441				(irq_number, rtl8139_handler)
442			}
443			#[cfg(all(
444				not(all(target_arch = "x86_64", feature = "rtl8139")),
445				any(feature = "tcp", feature = "udp")
446			))]
447			Self::VirtioNet(drv) => {
448				fn network_handler() {
449					if let Some(driver) = get_network_driver() {
450						driver.lock().handle_interrupt();
451					}
452				}
453
454				let irq_number = drv.lock().get_interrupt_number();
455
456				(irq_number, network_handler)
457			}
458			#[cfg(feature = "fuse")]
459			Self::VirtioFs(drv) => {
460				fn fuse_handler() {}
461
462				let irq_number = drv.lock().get_interrupt_number();

464				(irq_number, fuse_handler)
465			}
466			#[cfg(feature = "console")]
467			Self::VirtioConsole(drv) => {

475				(irq_number, console_handler)
476			}
477			_ => todo!(),
478		}
479	}

486pub(crate) fn get_interrupt_handlers() -> HashMap<InterruptLine, InterruptHandlerQueue, RandomState>
487{
488	let mut handlers: HashMap<InterruptLine, InterruptHandlerQueue, RandomState> =
489		HashMap::with_hasher(RandomState::with_seeds(0, 0, 0, 0));
490
491	for drv in PCI_DRIVERS.finalize().iter() {
492		let (irq_number, handler) = drv.get_interrupt_handler();
493
494		if let Some(map) = handlers.get_mut(&irq_number) {
495			map.push_back(handler);
496		} else {
497			let mut map: InterruptHandlerQueue = VecDeque::new();
498			map.push_back(handler);
499			handlers.insert(irq_number, map);
500		}
501	}
502
503	#[cfg(target_arch = "x86_64")]
504	{
505		use crate::kernel::serial::get_serial_handler;
506		let (irq_number, handler) = get_serial_handler();
507
508		if let Some(map) = handlers.get_mut(&irq_number) {
509			map.push_back(handler);
510		} else {
511			let mut map: InterruptHandlerQueue = VecDeque::new();
512			map.push_back(handler);
513			handlers.insert(irq_number, map);
514		}
515	}
516
517	handlers
518}

567pub(crate) fn init() {
568	// virtio: 4.1.2 PCI Device Discovery
569	without_interrupts(|| {
570		for adapter in PCI_DEVICES.finalize().iter().filter(|x| {
571			let (vendor_id, device_id) = x.id();
572			vendor_id == 0x1af4 && (0x1000..=0x107f).contains(&device_id)
573		}) {
574			info!(
575				"Found virtio device with device id {:#x}",
576				adapter.device_id()
577			);
578
579			#[cfg(any(
580				all(
581					any(feature = "tcp", feature = "udp"),
582					not(all(target_arch = "x86_64", feature = "rtl8139"))
583				),
584				feature = "fuse",
585				feature = "vsock",
586				feature = "console"
587			))]
588			match pci_virtio::init_device(adapter) {
589				#[cfg(all(
590					not(all(target_arch = "x86_64", feature = "rtl8139")),
591					any(feature = "tcp", feature = "udp")
592				))]
593				Ok(VirtioDriver::Network(drv)) => {
594					register_driver(PciDriver::VirtioNet(InterruptTicketMutex::new(drv)));
595				}
596				#[cfg(feature = "console")]
597				Ok(VirtioDriver::Console(drv)) => {
598					register_driver(PciDriver::VirtioConsole(InterruptTicketMutex::new(*drv)));
599					info!("Switch to virtio console");
600					crate::console::CONSOLE
601						.lock()
602						.replace_device(IoDevice::Virtio(VirtioUART::new()));
603				}
604				#[cfg(feature = "vsock")]
605				Ok(VirtioDriver::Vsock(drv)) => {
606					register_driver(PciDriver::VirtioVsock(InterruptTicketMutex::new(*drv)));
607				}
608				#[cfg(feature = "fuse")]
609				Ok(VirtioDriver::FileSystem(drv)) => {
610					register_driver(PciDriver::VirtioFs(InterruptTicketMutex::new(drv)));
611				}
612				_ => {}
613			}
614		}
615
616		// Searching for Realtek RTL8139, which is supported by Qemu
617		#[cfg(all(target_arch = "x86_64", feature = "rtl8139"))]
618		for adapter in PCI_DEVICES.finalize().iter().filter(|x| {
619			let (vendor_id, device_id) = x.id();
620			vendor_id == 0x10ec && (0x8138..=0x8139).contains(&device_id)
621		}) {
622			info!(
623				"Found Realtek network device with device id {:#x}",
624				adapter.device_id()
625			);
626
627			if let Ok(drv) = rtl8139::init_device(adapter) {
628				register_driver(PciDriver::RTL8139Net(InterruptTicketMutex::new(drv)));
629			}
630		}
631	});
632}
