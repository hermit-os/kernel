46pub(crate) struct NetDevCfg {

48	pub dev_id: u16,
49	pub features: virtio::net::F,
50}

52fn determine_mtu(dev_cfg: &NetDevCfg) -> u16 {
53	// If VIRTIO_NET_F_MTU is negotiated, "the driver uses mtu as the maximum MTU value"
54	// (VirtIO specification, 5.1.3, "Feature bits")
55	if dev_cfg.features.contains(virtio::net::F::MTU) {
56		dev_cfg.raw.as_ptr().mtu().read().to_ne()
57	} else {
58		// Otherwise, we can just use the MTU we want to use
59		mtu()
60	}
61}

63fn determine_rx_buf_size(dev_cfg: &NetDevCfg) -> u32 {
64	// See Virtio specification v1.1 - 5.1.6.3.1 and 5.1.4.2
65
66	// Our desired minimum buffer size - we want it to be at least the MTU generally
67	let mut min_buf_size = determine_mtu(dev_cfg).into();
68
69	// If VIRTIO_NET_F_MRG_RXBUF is negotiated, each buffer MUST be at least the size of the struct virtio_net_hdr.
70	// We just use MTU in that case, but otherwise...
71	if dev_cfg.features.contains(virtio::net::F::MRG_RXBUF)
72		&& let Some(my_mrg_rxbuf_size) = hermit_var!("HERMIT_MRG_RXBUF_SIZE")
73	{
74		let my_mrg_rxbuf_size = u32::from_str(&my_mrg_rxbuf_size).unwrap();
75		assert!(
76			my_mrg_rxbuf_size > 0,
77			"VIRTIO does not allow buffer elements of size 0."
78		);
79		min_buf_size = my_mrg_rxbuf_size;
80	} else {
81		// If [...] are negotiated, the driver SHOULD populate the receive queue(s) with buffers of at least 65562 bytes.
82		if dev_cfg.features.contains(virtio::net::F::GUEST_TSO4)
83			|| dev_cfg.features.contains(virtio::net::F::GUEST_TSO6)
84			|| dev_cfg.features.contains(virtio::net::F::GUEST_UFO)
85		{
86			min_buf_size = u32::max(min_buf_size, 65562 - size_of::<Hdr>() as u32);
87		} else {
88			// Otherwise, the driver SHOULD populate the receive queue(s) with buffers of at least 1526 bytes.
89			min_buf_size = u32::max(min_buf_size, 1526 - size_of::<Hdr>() as u32);
90		}
91	}
92
93	min_buf_size
94}

96pub struct RxQueues {
97	vqs: Vec<VirtQueue>,
98	buf_size: u32,
99}

101impl RxQueues {
102	pub fn new(vqs: Vec<VirtQueue>, dev_cfg: &NetDevCfg) -> Self {
103		Self {
104			vqs,
105			buf_size: determine_rx_buf_size(dev_cfg),
106		}
107	}
108
109	/// Adds a given queue to the underlying vector and populates the queue with RecvBuffers.
110	///
111	/// Queues are all populated according to Virtio specification v1.1. - 5.1.6.3.1
112	fn add(&mut self, mut vq: VirtQueue) {
113		let num_bufs: u16 = u16::from(vq.size()) / constants::BUFF_PER_PACKET;
114		fill_queue(&mut vq, num_bufs, self.buf_size);
115		self.vqs.push(vq);
116	}
117
118	fn get_next(&mut self) -> Option<UsedBufferToken> {
119		self.vqs[0].try_recv().ok()
120	}
121
122	fn enable_notifs(&mut self) {
123		for vq in &mut self.vqs {
124			vq.enable_notifs();
125		}
126	}
127
128	fn disable_notifs(&mut self) {
129		for vq in &mut self.vqs {
130			vq.disable_notifs();
131		}
132	}
133
134	fn has_packet(&self) -> bool {
135		self.vqs.iter().any(|vq| vq.has_used_buffers())
136	}
137}

139fn buffer_token_from_hdr(
140	hdr: Box<MaybeUninit<Hdr>, DeviceAlloc>,
141	buf_size: u32,
142) -> AvailBufferToken {
143	AvailBufferToken::new(SmallVec::new(), {
144		SmallVec::from_buf([
145			BufferElem::Sized(hdr),
146			BufferElem::Vector(Vec::with_capacity_in(
147				buf_size.try_into().unwrap(),
148				DeviceAlloc,
149			)),
150		])
151	})
152	.unwrap()
153}
154
155fn fill_queue(vq: &mut VirtQueue, num_bufs: u16, buf_size: u32) {
156	for _ in 0..num_bufs {
157		let buff_tkn = buffer_token_from_hdr(Box::<Hdr, _>::new_uninit_in(DeviceAlloc), buf_size);
158
159		// BufferTokens are directly provided to the queue
160		// TransferTokens are directly dispatched
161		// Transfers will be awaited at the queue
162		if let Err(err) = vq.dispatch(buff_tkn, false, BufferType::Direct) {
163			error!("{err:#?}");
164			break;
165		}
166	}
167}

169/// Structure which handles transmission of packets and delegation
170/// to the respective queue structures.
171pub struct TxQueues {
172	vqs: Vec<VirtQueue>,
173	buf_size: u32,
174}
175
176impl TxQueues {
177	pub fn new(vqs: Vec<VirtQueue>, dev_cfg: &NetDevCfg) -> Self {
178		Self {
179			vqs,
180			buf_size: determine_mtu(dev_cfg).into(),
181		}
182	}
183
184	#[allow(dead_code)]
185	fn enable_notifs(&mut self) {
186		for vq in &mut self.vqs {
187			vq.enable_notifs();
188		}
189	}
190
191	#[allow(dead_code)]
192	fn disable_notifs(&mut self) {
193		for vq in &mut self.vqs {
194			vq.disable_notifs();
195		}
196	}
197
198	/// Polls all queues for buffers whose transmission has been completed and returns the number of such buffers.
199	fn poll(&mut self) -> u32 {
200		let mut released_buffers = 0u32;
201		for vq in &mut self.vqs {
202			// We don't do anything with the buffers but we need to receive them for the
203			// ring slots to be emptied and the memory from the previous transfers to be freed.
204			while vq.try_recv().is_ok() {
205				released_buffers += 1;
206			}
207		}
208		released_buffers
209	}
210
211	fn add(&mut self, vq: VirtQueue) {
212		// Currently we are doing nothing with the additional queues. They are inactive and might be used in the
213		// future
214		self.vqs.push(vq);
215	}
216}

251impl Drop for TxToken<'_> {
252	fn drop(&mut self) {
253		*self.send_capacity += u32::from(BUFF_PER_PACKET);
254	}

257impl smoltcp::phy::TxToken for TxToken<'_> {
258	fn consume<R, F>(self, len: usize, f: F) -> R
259	where
260		F: FnOnce(&mut [u8]) -> R,
261	{
262		// When the token is consumed, the capacity cannot be returned until its buffer is marked by the device as used.
263		// Thus, we bypass the Drop implementation that would do that prematurely and let the call to poll in the next
264		// call to this function return the capacity.
265		let mut token = ManuallyDrop::new(self);
266		assert!(len <= usize::try_from(token.send_vqs.buf_size).unwrap());
267		let mut packet = Vec::with_capacity_in(len, DeviceAlloc);
268		let result = unsafe {
269			let result = f(packet.spare_capacity_mut().assume_init_mut());
270			packet.set_len(len);
271			result
272		};
273
274		let mut header = Box::new_in(<Hdr as Default>::default(), DeviceAlloc);
275
276		// If a checksum calculation by the host is necessary, we have to inform the host within the header
277		// see Virtio specification 5.1.6.2
278		if let Some((ip_header_len, csum_offset)) =
279			VirtioNetDriver::should_request_checksum(&token.checksums, &mut packet)
280		{
281			header.flags = HdrF::NEEDS_CSUM;
282			header.csum_start =
283				(u16::try_from(ETHERNET_HEADER_LEN).unwrap() + ip_header_len).into();
284			header.csum_offset = csum_offset.into();
285		}
286
287		let buff_tkn = AvailBufferToken::new(
288			SmallVec::from_buf([BufferElem::Sized(header), BufferElem::Vector(packet)]),
289			SmallVec::new(),
290		)
291		.unwrap();
292
293		token.send_vqs.vqs[0]
294			.dispatch(buff_tkn, false, BufferType::Direct)
295			.unwrap();
296
297		result
298	}
299}

302	recv_vqs: &'a mut RxQueues,
303	is_mrg_rxbuf_enabled: bool,
304}
305
306impl smoltcp::phy::RxToken for RxToken<'_> {
307	fn consume<R, F>(self, f: F) -> R
308	where
309		F: FnOnce(&[u8]) -> R,
310	{
311		let Some(mut buffer_tkn) = self.recv_vqs.get_next() else {
312			// We overpromised a frame. The best we can do is to provide an empty frame to smoltcp and let it handle it as a faulty reception.
313			return f(&[]);
314		};
315		// Safety: any buffers that do not start with a `Hdr` must have been consumed by the previous call
316		// to this function.
317		let first_header = unsafe {
318			buffer_tkn
319				.used_recv_buff
320				.pop_front_downcast::<Hdr>()
321				.unwrap()
322		};
323		let first_packet = buffer_tkn.used_recv_buff.pop_front_vec().unwrap();
324
325		// According to VIRTIO spec v1.2 sec. 5.1.6.3.2, "num_buffers will always be 1 if VIRTIO_NET_F_MRG_RXBUF is not negotiated."
326		// Unfortunately, NVIDIA MLX5 does not comply with this requirement and we have to manually set the value to the correct one.
327		let num_buffers = if self.is_mrg_rxbuf_enabled {
328			first_header.num_buffers.to_ne()
329		} else {
330			1
331		};
332
333		let mut combined_packets = first_packet;
334
335		let first_tkn = buffer_token_from_hdr(
336			// SAFETY: Box<T> -> Box<MaybeUninit<T>> is sound
337			unsafe {
338				transmute::<Box<Hdr, DeviceAlloc>, Box<MaybeUninit<Hdr>, DeviceAlloc>>(first_header)
339			},
340			self.recv_vqs.buf_size,
341		);
342		self.recv_vqs.vqs[0]
343			.dispatch(first_tkn, false, BufferType::Direct)
344			.unwrap();
345
346		for _ in 1..num_buffers {
347			let mut buffer_tkn = self.recv_vqs.get_next().unwrap();
348			// The descriptor that was meant for the header of another frame was used for a portion of the current frame's contents.
349			// Thus, we cannot cast it to a Hdr.
350			let (header_descriptor, used_len) = buffer_tkn.used_recv_buff.pop_front_raw().unwrap();
351			combined_packets.extend_from_slice(unsafe {
352				core::slice::from_raw_parts((&raw const *header_descriptor).cast::<u8>(), used_len)
353			});
354
355			let packet = buffer_tkn.used_recv_buff.pop_front_vec().unwrap();
356			combined_packets.extend_from_slice(&packet);
357
358			let header = header_descriptor.downcast::<MaybeUninit<Hdr>>().unwrap();
359
360			let tkn = buffer_token_from_hdr(
361				// SAFETY: Box<T> -> Box<MaybeUninit<T>> is sound
362				header,
363				self.recv_vqs.buf_size,
364			);
365			self.recv_vqs.vqs[0]
366				.dispatch(tkn, false, BufferType::Direct)
367				.unwrap();
368		}
369
370		f(&combined_packets)
371	}
372}

406			todo!("Implement possibility to change config on the fly...")

411			info!("Configuration changes are not possible! Aborting");
412			todo!("Implement possibility to change config on the fly...")
413		}
414
415		self.isr_stat.acknowledge();
416	}

419impl smoltcp::phy::Device for VirtioNetDriver {
420	type TxToken<'a> = TxToken<'a>;
421	type RxToken<'a> = RxToken<'a>;
422
423	fn capabilities(&self) -> DeviceCapabilities {
424		let mut device_capabilities = DeviceCapabilities::default();
425		device_capabilities.medium = smoltcp::phy::Medium::Ethernet;
426		device_capabilities.max_transmission_unit = self.inner.mtu.into();
427		device_capabilities.max_burst_size =
428			Some(usize::try_from(self.inner.send_capacity).unwrap() / usize::from(BUFF_PER_PACKET));
429		device_capabilities.checksum = self.checksums.clone();
430		device_capabilities
431	}
432
433	fn receive(
434		&mut self,
435		_timestamp: smoltcp::time::Instant,
436	) -> Option<(Self::RxToken<'_>, Self::TxToken<'_>)> {
437		if self.inner.recv_vqs.has_packet() && {
438			self.free_up_send_capacity();
439			self.inner.send_capacity >= u32::from(BUFF_PER_PACKET)
440		} {
441			self.inner.send_capacity -= u32::from(BUFF_PER_PACKET);
442			Some((
443				RxToken {
444					recv_vqs: &mut self.inner.recv_vqs,
445					is_mrg_rxbuf_enabled: self.dev_cfg.features.contains(virtio::net::F::MRG_RXBUF),
446				},
447				TxToken {
448					send_vqs: &mut self.inner.send_vqs,
449					checksums: self.checksums.clone(),
450					send_capacity: &mut self.inner.send_capacity,
451				},
452			))
453		} else {
454			None
455		}
456	}
457
458	fn transmit(&mut self, _timestamp: smoltcp::time::Instant) -> Option<Self::TxToken<'_>> {
459		self.free_up_send_capacity();
460		if self.inner.send_capacity >= u32::from(BUFF_PER_PACKET) {
461			self.inner.send_capacity -= u32::from(BUFF_PER_PACKET);
462			Some(TxToken {
463				send_vqs: &mut self.inner.send_vqs,
464				checksums: self.checksums.clone(),
465				send_capacity: &mut self.inner.send_capacity,
466			})
467		} else {
468			None
469		}
470	}
471}

561
562	/// If necessary, sets the TCP or UDP checksum field to the checksum of the
563	/// pseudo-header and returns the IP header length and the checksum offset.
564	/// Otherwise, returns None.
565	fn should_request_checksum<T: AsRef<[u8]> + AsMut<[u8]>>(
566		checksums: &ChecksumCapabilities,
567		frame: T,
568	) -> Option<(u16, u16)> {
569		if checksums.tcp.tx() && checksums.udp.tx() {
570			return None;
571		}
572
573		let ip_header_len: u16;
574		let ip_packet_len: usize;
575		let protocol;
576		let pseudo_header_checksum;
577		let mut ethernet_frame = EthernetFrame::new_unchecked(frame);
578		match ethernet_frame.ethertype() {
579			smoltcp::wire::EthernetProtocol::Ipv4 => {
580				let ip_packet = Ipv4Packet::new_unchecked(&*ethernet_frame.payload_mut());
581				ip_header_len = ip_packet.header_len().into();
582				ip_packet_len = ip_packet.total_len().into();
583				protocol = ip_packet.next_header();
584				pseudo_header_checksum =
585					partial_checksum::ipv4_pseudo_header_partial_checksum(&ip_packet);
586			}
587			smoltcp::wire::EthernetProtocol::Ipv6 => {
588				let ip_packet = Ipv6Packet::new_unchecked(&*ethernet_frame.payload_mut());
589				ip_header_len = ip_packet.header_len().try_into().expect(
590					"VIRTIO does not support IP headers that are longer than u16::MAX bytes.",
591				);
592				ip_packet_len = ip_packet.total_len();
593				protocol = ip_packet.next_header();
594				pseudo_header_checksum =
595					partial_checksum::ipv6_pseudo_header_partial_checksum(&ip_packet);
596			}
597			// If the Ethernet protocol is not one of these two above, for which we know there may be a checksum field,
598			// we default to not asking for checksum, as otherwise the frame will be corrupted by the device trying
599			// to write the checksum.
600			_ => return None,
601		};
602
603		let csum_offset;
604		let ip_payload = &mut ethernet_frame.payload_mut()[ip_header_len.into()..ip_packet_len];
605		// Like the Ethernet protocol check, we check for IP protocols for which we know the location of the checksum field.
606		if protocol == smoltcp::wire::IpProtocol::Tcp && !checksums.tcp.tx() {
607			let mut tcp_packet = smoltcp::wire::TcpPacket::new_unchecked(ip_payload);
608			tcp_packet.set_checksum(pseudo_header_checksum);
609			csum_offset = 16;
610		} else if protocol == smoltcp::wire::IpProtocol::Udp && !checksums.udp.tx() {
611			let mut udp_packet = smoltcp::wire::UdpPacket::new_unchecked(ip_payload);
612			udp_packet.set_checksum(pseudo_header_checksum);
613			csum_offset = 6;
614		} else {
615			return None;
616		};
617
618		Some((ip_header_len, csum_offset))
619	}
620
621	fn free_up_send_capacity(&mut self) {
622		// We need to poll to get the queue to remove elements from the table and open up capacity if possible.
623		self.inner.send_capacity += self.inner.send_vqs.poll() * u32::from(BUFF_PER_PACKET);
624	}
625}

627impl VirtioNetDriver<Uninit> {
628	/// Initializes the device in adherence to specification. Returns Some(VirtioNetError)
629	/// upon failure and None in case everything worked as expected.
630	///
631	/// See Virtio specification v1.1. - 3.1.1.
632	///                      and v1.1. - 5.1.5
633	pub fn init_dev(mut self) -> Result<VirtioNetDriver<Init>, VirtioNetError> {
634		// Reset
635		self.com_cfg.reset_dev();
636
637		// Indicate device, that OS noticed it
638		self.com_cfg.ack_dev();
639
640		// Indicate device, that driver is able to handle it
641		self.com_cfg.set_drv();
642
643		let minimal_features = virtio::net::F::VERSION_1 | virtio::net::F::MAC;
644
645		// If wanted, push new features into feats here:
646		let mut features = minimal_features
647			// Indirect descriptors can be used
648			| virtio::net::F::INDIRECT_DESC
649			// Packed Vq can be used
650			| virtio::net::F::RING_PACKED
651			| virtio::net::F::NOTIFICATION_DATA
652			// MTU setting can be used
653			| virtio::net::F::MTU
654			// Driver can merge receive buffers
655			| virtio::net::F::MRG_RXBUF
656			// the link status can be announced
657			| virtio::net::F::STATUS
658			// control queue support
659			| virtio::net::F::CTRL_VQ
660			// Multiqueue support
661			| virtio::net::F::MQ
662			// Checksum calculation can partially be offloaded to the device
663			| virtio::net::F::CSUM;
664
665		// Currently the driver does NOT support the features below.
666		// In order to provide functionality for these, the driver
667		// needs to take care of calculating checksum.
668		// | virtio::net::F::GUEST_TSO4
669		// | virtio::net::F::GUEST_TSO6
670
671		// Negotiate features with device. Automatically reduces selected feats in order to meet device capabilities.
672		// Aborts in case incompatible features are selected by the driver or the device does not support min_feat_set.
673		match self.negotiate_features(features) {
674			Ok(()) => info!(
675				"Driver found a subset of features for virtio device {:x}. Features are: {features:?}",
676				self.dev_cfg.dev_id
677			),
678			Err(vnet_err) => {
679				match vnet_err {
680					VirtioNetError::FeatureRequirementsNotMet(features) => {
681						error!(
682							"Network drivers feature set {features:?} does not satisfy rules in section 5.1.3.1 of specification v1.1. Aborting!"
683						);
684						return Err(vnet_err);
685					}
686					VirtioNetError::IncompatibleFeatureSets(drv_feats, dev_feats) => {
687						// Create a new matching feature set for device and driver if the minimal set is met!
688						if !dev_feats.contains(minimal_features) {
689							error!(
690								"Device features set, does not satisfy minimal features needed. Aborting!"
691							);
692							return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
693						}
694
695						let common_features = drv_feats & dev_feats;
696						if common_features.is_empty() {
697							error!(
698								"Feature negotiation failed with minimal feature set. Aborting!"
699							);
700							return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
701						}
702						features = common_features;
703
704						match self.negotiate_features(features) {
705							Ok(()) => info!(
706								"Driver found a subset of features for virtio device {:x}. Features are: {features:?}",
707								self.dev_cfg.dev_id
708							),
709							Err(vnet_err) => match vnet_err {
710								VirtioNetError::FeatureRequirementsNotMet(features) => {
711									error!(
712										"Network device offers a feature set {features:?} when used completely does not satisfy rules in section 5.1.3.1 of specification v1.1. Aborting!"
713									);
714									return Err(vnet_err);
715								}
716								_ => {
717									error!(
718										"Feature Set after reduction still not usable. Set: {features:?}. Aborting!"
719									);
720									return Err(vnet_err);
721								}
722							},
723						}
724					}
725					VirtioNetError::FailFeatureNeg(_) => {
726						error!(
727							"Wanted set of features is NOT supported by device. Set: {features:?}"
728						);
729						return Err(vnet_err);
730					}
731					#[cfg(feature = "pci")]
732					VirtioNetError::NoDevCfg(_) => {
733						error!("No device config found.");
734						return Err(vnet_err);
735					}
736				}
737			}
738		}
739
740		// Indicates the device, that the current feature set is final for the driver
741		// and will not be changed.
742		self.com_cfg.features_ok();
743
744		// Checks if the device has accepted final set. This finishes feature negotiation.
745		if self.com_cfg.check_features() {
746			info!(
747				"Features have been negotiated between virtio network device {:x} and driver.",
748				self.dev_cfg.dev_id
749			);
750			// Set feature set in device config fur future use.
751			self.dev_cfg.features = features;
752		} else {
753			return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
754		}
755
756		let mut inner = Init {
757			mtu: determine_mtu(&self.dev_cfg),
758			ctrl_vq: None,
759			recv_vqs: RxQueues::new(Vec::new(), &self.dev_cfg),
760			send_vqs: TxQueues::new(Vec::new(), &self.dev_cfg),
761			send_capacity: 0,
762		};
763
764		debug!("Using RX buffer size of {}", inner.recv_vqs.buf_size);
765
766		self.dev_spec_init(&mut inner)?;
767		info!(
768			"Device specific initialization for Virtio network device {:x} finished",
769			self.dev_cfg.dev_id
770		);
771
772		// At this point the device is "live"
773		self.com_cfg.drv_ok();
774
775		if self.dev_cfg.features.contains(virtio::net::F::CSUM)
776			&& self.dev_cfg.features.contains(virtio::net::F::GUEST_CSUM)
777		{
778			self.checksums.udp = Checksum::None;
779			self.checksums.tcp = Checksum::None;
780		} else if self.dev_cfg.features.contains(virtio::net::F::CSUM) {
781			self.checksums.udp = Checksum::Rx;
782			self.checksums.tcp = Checksum::Rx;
783		} else if self.dev_cfg.features.contains(virtio::net::F::GUEST_CSUM) {
784			self.checksums.udp = Checksum::Tx;
785			self.checksums.tcp = Checksum::Tx;
786		}
787		debug!("{:?}", self.checksums);
788
789		Ok(VirtioNetDriver {
790			dev_cfg: self.dev_cfg,
791			com_cfg: self.com_cfg,
792			isr_stat: self.isr_stat,
793			notif_cfg: self.notif_cfg,
794			inner,
795			num_vqs: self.num_vqs,
796			irq: self.irq,
797			checksums: self.checksums,
798		})
799	}
800
801	/// Negotiates a subset of features, understood and wanted by both the OS
802	/// and the device.
803	fn negotiate_features(
804		&mut self,
805		driver_features: virtio::net::F,
806	) -> Result<(), VirtioNetError> {
807		let device_features = virtio::net::F::from(self.com_cfg.dev_features());
808
809		if device_features.requirements_satisfied() {
810			info!("Feature set wanted by network driver are in conformance with specification.");
811		} else {
812			return Err(VirtioNetError::FeatureRequirementsNotMet(device_features));
813		}
814
815		if device_features.contains(driver_features) {
816			// If device supports subset of features write feature set to common config
817			self.com_cfg.set_drv_features(driver_features.into());
818			Ok(())
819		} else {
820			Err(VirtioNetError::IncompatibleFeatureSets(
821				driver_features,
822				device_features,
823			))
824		}
825	}
826
827	/// Device Specific initialization according to Virtio specifictation v1.1. - 5.1.5
828	fn dev_spec_init(&mut self, inner: &mut Init) -> Result<(), VirtioNetError> {
829		self.virtqueue_init(inner)?;
830		info!("Network driver successfully initialized virtqueues.");
831
832		// Add a control if feature is negotiated
833		if self.dev_cfg.features.contains(virtio::net::F::CTRL_VQ) {
834			let mut ctrl_vq = if self.dev_cfg.features.contains(virtio::net::F::RING_PACKED) {
835				VirtQueue::Packed(
836					PackedVq::new(
837						&mut self.com_cfg,
838						&self.notif_cfg,
839						VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
840						VqIndex::from(self.num_vqs),
841						self.dev_cfg.features.into(),
842					)
843					.unwrap(),
844				)
845			} else {
846				VirtQueue::Split(
847					SplitVq::new(
848						&mut self.com_cfg,
849						&self.notif_cfg,
850						VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
851						VqIndex::from(self.num_vqs),
852						self.dev_cfg.features.into(),
853					)
854					.unwrap(),
855				)
856			};
857
858			ctrl_vq.enable_notifs();
859			inner.ctrl_vq = Some(ctrl_vq);
860		}
861
862		Ok(())
863	}
864
865	/// Initialize virtqueues via the queue interface and populates receiving queues
866	fn virtqueue_init(&mut self, inner: &mut Init) -> Result<(), VirtioNetError> {
867		// We are assuming here, that the device single source of truth is the
868		// device specific configuration. Hence we do NOT check if
869		//
870		// max_virtqueue_pairs + 1 < num_queues
871		//
872		// - the plus 1 is due to the possibility of an existing control queue
873		// - the num_queues is found in the ComCfg struct of the device and defines the maximal number
874		// of supported queues.
875		if self.dev_cfg.features.contains(virtio::net::F::MQ) {
876			if self
877				.dev_cfg
878				.raw
879				.as_ptr()
880				.max_virtqueue_pairs()
881				.read()
882				.to_ne() * 2 >= MAX_NUM_VQ
883			{
884				self.num_vqs = MAX_NUM_VQ;
885			} else {
886				self.num_vqs = self
887					.dev_cfg
888					.raw
889					.as_ptr()
890					.max_virtqueue_pairs()
891					.read()
892					.to_ne() * 2;
893			}
894		} else {
895			// Minimal number of virtqueues defined in the standard v1.1. - 5.1.5 Step 1
896			self.num_vqs = 2;
897		}
898
899		// The loop is running from 0 to num_vqs and the indexes are provided to the VqIndex::from function in this way
900		// in order to allow the indexes of the queues to be in a form of:
901		//
902		// index i for receive queue
903		// index i+1 for send queue
904		//
905		// as it is wanted by the network network device.
906		// see Virtio specification v1.1. - 5.1.2
907		// Assure that we have always an even number of queues (i.e. pairs of queues).
908		assert_eq!(self.num_vqs % 2, 0);
909
910		for i in 0..(self.num_vqs / 2) {
911			if self.dev_cfg.features.contains(virtio::net::F::RING_PACKED) {
912				let mut vq = PackedVq::new(
913					&mut self.com_cfg,
914					&self.notif_cfg,
915					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
916					VqIndex::from(2 * i),
917					self.dev_cfg.features.into(),
918				)
919				.unwrap();
920				// Interrupt for receiving packets is wanted
921				vq.enable_notifs();
922
923				inner.recv_vqs.add(VirtQueue::Packed(vq));
924
925				let mut vq = PackedVq::new(
926					&mut self.com_cfg,
927					&self.notif_cfg,
928					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
929					VqIndex::from(2 * i + 1),
930					self.dev_cfg.features.into(),
931				)
932				.unwrap();
933				// Interrupt for communicating that a sent packet left, is not needed
934				vq.disable_notifs();
935
936				inner.send_capacity += u32::from(u16::from(vq.size()));
937				inner.send_vqs.add(VirtQueue::Packed(vq));
938			} else {
939				let mut vq = SplitVq::new(
940					&mut self.com_cfg,
941					&self.notif_cfg,
942					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
943					VqIndex::from(2 * i),
944					self.dev_cfg.features.into(),
945				)
946				.unwrap();
947				// Interrupt for receiving packets is wanted
948				vq.enable_notifs();
949
950				inner.recv_vqs.add(VirtQueue::Split(vq));
951
952				let mut vq = SplitVq::new(
953					&mut self.com_cfg,
954					&self.notif_cfg,
955					VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
956					VqIndex::from(2 * i + 1),
957					self.dev_cfg.features.into(),
958				)
959				.unwrap();
960				// Interrupt for communicating that a sent packet left, is not needed
961				vq.disable_notifs();
962				inner.send_capacity += u32::from(u16::from(vq.size()));
963				inner.send_vqs.add(VirtQueue::Split(vq));
964			}
965		}
966
967		Ok(())
968	}

995/// The checksum functions in this module only calculate the one's complement sum for the pseudo-header
996/// and their results are meant to be combined with the TCP payload to calculate the real checksum.
997/// They are only useful for the VIRTIO driver with the checksum offloading feature.
998///
999/// The calculations here can theoretically be made faster by exploiting the properties described in
1000/// [RFC 1071 section 2](https://www.rfc-editor.org/rfc/rfc1071).
1001mod partial_checksum {
1002	use smoltcp::wire::{Ipv4Packet, Ipv6Packet};
1003
1004	fn addr_sum<const N: usize>(addr: &[u8; N]) -> u16 {
1005		let mut sum = 0;
1006		const CHUNK_SIZE: usize = size_of::<u16>();
1007		for i in 0..(N / CHUNK_SIZE) {
1008			sum = ones_complement_add(
1009				sum,
1010				(u16::from(addr[CHUNK_SIZE * i]) << 8) | u16::from(addr[CHUNK_SIZE * i + 1]),
1011			);
1012		}
1013		sum
1014	}
1015
1016	/// Calculates the checksum for the IPv4 pseudo-header as described in
1017	/// [RFC 9293 subsection 3.1](https://www.rfc-editor.org/rfc/rfc9293.html#section-3.1-6.18.1) WITHOUT the final inversion.
1018	pub(super) fn ipv4_pseudo_header_partial_checksum<T: AsRef<[u8]>>(
1019		packet: &Ipv4Packet<T>,
1020	) -> u16 {
1021		let padded_protocol = u16::from(u8::from(packet.next_header()));
1022		let payload_len = packet.total_len() - u16::from(packet.header_len());
1023
1024		let mut sum = addr_sum(&packet.src_addr().octets());
1025		sum = ones_complement_add(sum, addr_sum(&packet.dst_addr().octets()));
1026		sum = ones_complement_add(sum, padded_protocol);
1027		ones_complement_add(sum, payload_len)
1028	}
1029
1030	/// Calculates the checksum for the IPv6 pseudo-header as described in
1031	/// [RFC 8200 subsection 8.1](https://www.rfc-editor.org/rfc/rfc8200.html#section-8.1) WITHOUT the final inversion.
1032	pub(super) fn ipv6_pseudo_header_partial_checksum<T: AsRef<[u8]>>(
1033		packet: &Ipv6Packet<T>,
1034	) -> u16 {
1035		warn!("The IPv6 partial checksum implementation is untested!");
1036		let padded_protocol = u16::from(u8::from(packet.next_header()));
1037
1038		let mut sum = addr_sum(&packet.src_addr().octets());
1039		sum = ones_complement_add(sum, addr_sum(&packet.dst_addr().octets()));
1040		sum = ones_complement_add(sum, packet.payload_len());
1041		ones_complement_add(sum, padded_protocol)
1042	}
1043
1044	/// Implements one's complement checksum as described in [RFC 1071 section 1](https://www.rfc-editor.org/rfc/rfc1071#section-1).
1045	fn ones_complement_add(lhs: u16, rhs: u16) -> u16 {
1046		let (sum, overflow) = u16::overflowing_add(lhs, rhs);
1047		sum + u16::from(overflow)
1048	}
