17/// Memory at this physical address is supposed to contain a pointer to the Extended BIOS Data Area (EBDA).
18const EBDA_PTR_LOCATION: PhysAddr = PhysAddr::new(0x0000_040e);
19/// Minimum physical address where a valid EBDA must be located.

22const EBDA_WINDOW_SIZE: usize = 1024;

24const RSDP_SEARCH_ADDRESS_LOW: PhysAddr = PhysAddr::new(0xe_0000);

26const RSDP_SEARCH_ADDRESS_HIGH: PhysAddr = PhysAddr::new(0xf_ffff);

30const RSDP_XCHECKSUM_LENGTH: usize = 36;

99/// A convenience structure to work with an ACPI table.
100/// Maps a single table to memory and frees the memory when a variable of this structure goes out of scope.
101#[derive(Debug)]
102pub struct AcpiTable<'a> {
103	header: &'a AcpiSdtHeader,
104	allocated_virtual_address: VirtAddr,
105	allocated_length: usize,
106}
107
108impl AcpiTable<'_> {
109	fn map(physical_address: PhysAddr) -> Self {
110		if env::is_uefi() {
111			// For UEFI Systems, the tables are already mapped so we only need to return a proper reference to the table
112			let allocated_virtual_address = VirtAddr::new(physical_address.as_u64());
113			let header = unsafe {
114				allocated_virtual_address
115					.as_ptr::<AcpiSdtHeader>()
116					.as_ref()
117					.unwrap()
118			};
119			let allocated_length = usize::try_from(header.length).unwrap();
120
121			return Self {
122				header,
123				allocated_virtual_address,
124				allocated_length,
125			};
126		}
127
128		let mut flags = PageTableEntryFlags::empty();
129		flags.normal().read_only().execute_disable();
130
131		// Allocate two 4 KiB pages for the table and map it.
132		// This guarantees that we can access at least the "length" field of the table header when its physical address
133		// crosses a page boundary.
134		let mut allocated_length = 2 * BasePageSize::SIZE as usize;
135		let mut count = allocated_length / BasePageSize::SIZE as usize;
136
137		let physical_map_address = physical_address.align_down(BasePageSize::SIZE);
138		let offset = (physical_address - physical_map_address) as usize;
139		let layout = PageLayout::from_size(allocated_length).unwrap();
140		let page_range = PageAlloc::allocate(layout).unwrap();
141		let mut virtual_address = VirtAddr::from(page_range.start());
142		paging::map::<BasePageSize>(virtual_address, physical_map_address, count, flags);
143
144		// Get a pointer to the header and query the table length.
145		let mut header_ptr: *const AcpiSdtHeader = (virtual_address + offset).as_ptr();
146		let table_length = unsafe { (*header_ptr).length } as usize;
147
148		// Remap if the length exceeds what we've allocated.
149		if table_length > allocated_length - offset {
150			let range =
151				PageRange::from_start_len(virtual_address.as_usize(), allocated_length).unwrap();
152			unsafe {
153				PageAlloc::deallocate(range);
154			}
155
156			allocated_length = (table_length + offset).align_up(BasePageSize::SIZE as usize);
157			count = allocated_length / BasePageSize::SIZE as usize;
158
159			let layout = PageLayout::from_size(allocated_length).unwrap();
160			let page_range = PageAlloc::allocate(layout).unwrap();
161			virtual_address = VirtAddr::from(page_range.start());
162			paging::map::<BasePageSize>(virtual_address, physical_map_address, count, flags);
163
164			header_ptr = (virtual_address + offset).as_ptr();
165		}
166
167		// Return the table.
168		Self {
169			header: unsafe { &*header_ptr },
170			allocated_virtual_address: virtual_address,
171			allocated_length,
172		}
173	}
174
175	pub fn header_start_address(&self) -> usize {
176		ptr::from_ref(self.header).addr()
177	}
178
179	pub fn table_start_address(&self) -> usize {
180		self.header_start_address() + mem::size_of::<AcpiSdtHeader>()
181	}
182
183	pub fn table_end_address(&self) -> usize {
184		self.header_start_address() + self.header.length as usize
185	}
186}

277/// Verifies the checksum of an ACPI table.
278/// Tables supporting this feature contain a "checksum" field. The value of this field is chosen, so that a
279/// (wrapping) sum over all table fields equals zero.
280fn verify_checksum(start_address: usize, length: usize) -> Result<(), ()> {
281	// Get a slice over all bytes of the structure that are considered for the checksum.
282	let slice =
283		unsafe { slice::from_raw_parts(ptr::with_exposed_provenance(start_address), length) };
284
285	// Perform a wrapping sum over these bytes.
286	let checksum = slice.iter().fold(0, |acc: u8, x| acc.wrapping_add(*x));
287
288	// This sum must equal to zero to be valid.
289	if checksum == 0 { Ok(()) } else { Err(()) }
290}

292/// Tries to find the ACPI RSDP within the specified address range.
293/// Returns a reference to it within the Ok() if successful or an empty Err() on failure.
294fn detect_rsdp(start_address: PhysAddr, end_address: PhysAddr) -> Result<&'static AcpiRsdp, ()> {
295	// Trigger page mapping in the first iteration!
296	let mut current_page = 0;
297
298	// Look for the ACPI RSDP in all possible 16-byte aligned addresses within this range.
299	for current_address in (start_address.as_usize()..end_address.as_usize()).step_by(16) {
300		// Have we crossed a page boundary in the last iteration?
301		if current_address / BasePageSize::SIZE as usize > current_page {
302			// Identity-map this possible page of the RSDP.
303			let frame = PhysFrame::<BasePageSize>::containing_address(x86_64::PhysAddr::new(
304				current_address as u64,
305			));
306			paging::identity_map::<BasePageSize>(frame.start_address().into());
307			current_page = current_address / BasePageSize::SIZE as usize;
308		}
309
310		// Verify the signature to find out if this is really an ACPI RSDP.
311		let rsdp = unsafe { &*(ptr::with_exposed_provenance::<AcpiRsdp>(current_address)) };
312		if &rsdp.signature != b"RSD PTR " {
313			continue;
314		}
315
316		// Verify the basic checksum.
317		if verify_checksum(current_address, RSDP_CHECKSUM_LENGTH).is_err() {
318			debug!("Found an ACPI table at {current_address:#X}, but its RSDP checksum is invalid");
319			continue;
320		}
321
322		// Verify the extended checksum if this is an ACPI 2.0-compliant table.
323		if rsdp.revision >= 2 && verify_checksum(current_address, RSDP_XCHECKSUM_LENGTH).is_err() {
324			debug!(
325				"Found an ACPI table at {current_address:#X}, but its RSDP extended checksum is invalid"
326			);
327			continue;
328		}
329
330		// We were successful! Return a pointer to the RSDT (whose 64-bit address is called XSDT in this structure).
331		info!(
332			"Found an ACPI revision {} table at {:#X} with OEM ID \"{}\"",
333			rsdp.revision,
334			current_address,
335			rsdp.oem_id()
336		);
337		return Ok(rsdp);
338	}
339
340	// We found no valid ACPI RSDP.
341	Err(())
342}

344/// Detects ACPI support of the computer system.
345/// Returns a reference to the ACPI RSDP within the Ok() if successful or an empty Err() on failure.
346fn detect_acpi() -> Result<&'static AcpiRsdp, ()> {
347	if let Some(rsdp) = env::rsdp() {
348		trace!("RSDP detected successfully at {rsdp:#x?}");
349		let rsdp = unsafe {
350			ptr::with_exposed_provenance::<AcpiRsdp>(rsdp.get())
351				.as_ref()
352				.unwrap()
353		};
354		assert!(&rsdp.signature == b"RSD PTR ", "RSDP Address not valid!");
355		return Ok(rsdp);
356	}
357
358	// Get the address of the EBDA.
359	let frame = PhysFrame::<BasePageSize>::containing_address(EBDA_PTR_LOCATION.into());
360	paging::identity_map::<BasePageSize>(frame.start_address().into());
361	let ebda_ptr_location: &u16 =
362		unsafe { &*(VirtAddr::from(EBDA_PTR_LOCATION.as_u64()).as_ptr()) };
363	let ebda_address = PhysAddr::new(u64::from(*ebda_ptr_location) << 4);
364
365	// Check if the pointed address is valid. This check is also done in ACPICA.
366	if ebda_address > EBDA_MINIMUM_ADDRESS {
367		// Try to find an RSDP within the 1 KiB window of the EBDA.
368		if let Ok(rsdp) = detect_rsdp(ebda_address, ebda_address + EBDA_WINDOW_SIZE) {
369			return Ok(rsdp);
370		}
371	}
372
373	// If we didn't find anything above, check the other memory range for an RSDP.
374	if let Ok(rsdp) = detect_rsdp(RSDP_SEARCH_ADDRESS_LOW, RSDP_SEARCH_ADDRESS_HIGH) {
375		return Ok(rsdp);
376	}
377
378	// We didn't find any ACPI tables.
379	Err(())
380}

382fn search_s5_in_table(table: AcpiTable<'_>) {
383	// Get the AML code.
384	// As we do not implement an AML interpreter, we search through the bytecode.
385	let aml = unsafe {
386		slice::from_ptr_range(
387			ptr::with_exposed_provenance(table.table_start_address())
388				..ptr::with_exposed_provenance(table.table_end_address()),
389		)
390	};
391
392	// Find the "_S5_" object in the bytecode.
393	let s5 = [b'_', b'S', b'5', b'_', AML_PACKAGEOP];
394	let s5_position = aml.windows(s5.len()).position(|window| window == s5);
395	if let Some(i) = s5_position {
396		// We have found an "_S5_" object that looks valid.
397		// To be sure, verify that it begins with an AML_NAMEOP or an AML_NAMEOP and a backslash.
398		if i > 2 && (aml[i - 1] == AML_NAMEOP || (aml[i - 2] == AML_NAMEOP && aml[i - 1] == b'\\'))
399		{
400			// This is a valid "_S5_" object.
401			// It should be followed by this structure:
402			//    - single byte for PkgLength (index 5)
403			//    - single byte for NumElements (index 6)
404			let pkg_length = aml[i + 5];
405			let num_elements = aml[i + 6];
406
407			// Bits 6-7 of PkgLength are non-zero for larger packages, resulting in a different structure.
408			// This mustn't be the case for the "_S5_" object.
409			if pkg_length & 0b1100_0000 == 0 && num_elements > 0 {
410				// The next byte is an opcode describing the data.
411				// It is usually the byte prefix, indicating that the actual data is the single byte following the opcode.
412				// However, if the data is a zero or one byte, this may also be indicated by the opcode.
413				let op = aml[i + 7];
414				let slp_typa = match op {
415					AML_ZEROOP => 0,
416					AML_ONEOP => 1,
417					AML_BYTEPREFIX => aml[i + 8],
418					_ => return,
419				};
420
421				// All assumptions are correct, so slp_typa is supposed to contain valid information.
422				// Now we have all information we need for powering off through ACPI.
423				//
424				// Note that Power Off may also be controlled through PM1B_CNT_BLK / SLP_TYPB
425				// according to the ACPI Specification. However, this has not yet been observed on real computers
426				// and therefore not implemented.
427				SLP_TYPA.set(slp_typa).unwrap();
428			}
429		}
430	}
431}

433fn parse_fadt(fadt: AcpiTable<'_>) {
434	// Get us a reference to the actual fields of the FADT table.
435	// Note that not all fields may be accessible depending on the ACPI revision of the computer.
436	// Always check fadt.table_end_address() when accessing an optional field!
437	let fadt_table =
438		unsafe { &*ptr::with_exposed_provenance::<AcpiFadt>(fadt.table_start_address()) };
439
440	// Check if the FADT is large enough to hold an x_pm1a_cnt_blk field and if this field is non-zero.
441	// In that case, it shall be preferred over the I/O port specified in pm1a_cnt_blk.
442	// As all PM1 control registers are supposed to be in I/O space, we can simply check the address_space field
443	// of x_pm1a_cnt_blk to determine the validity of x_pm1a_cnt_blk.
444	let x_pm1a_cnt_blk_field_address = ptr::from_ref(&fadt_table.x_pm1a_cnt_blk).addr();
445	let pm1a_cnt_blk = if x_pm1a_cnt_blk_field_address < fadt.table_end_address()
446		&& fadt_table.x_pm1a_cnt_blk.address_space == GENERIC_ADDRESS_IO_SPACE
447	{
448		fadt_table.x_pm1a_cnt_blk.address as u16
449	} else {
450		fadt_table.pm1a_cnt_blk as u16
451	};
452	PM1A_CNT_BLK.set(Port::new(pm1a_cnt_blk)).unwrap();
453
454	// Map the "Differentiated System Description Table" (DSDT).
455	let x_dsdt_field_address = ptr::addr_of!(fadt_table.x_dsdt) as usize;
456	let dsdt_address = if x_dsdt_field_address < fadt.table_end_address() && fadt_table.x_dsdt > 0 {
457		PhysAddr::new(fadt_table.x_dsdt)
458	} else {
459		PhysAddr::new(fadt_table.dsdt.into())
460	};
461	let dsdt = AcpiTable::map(dsdt_address);
462
463	// Check it.
464	assert!(
465		dsdt.header.signature() == "DSDT",
466		"DSDT at {:p} has invalid signature \"{}\"",
467		dsdt_address,
468		dsdt.header.signature()
469	);
470	assert!(
471		verify_checksum(dsdt.header_start_address(), dsdt.header.length as usize).is_ok(),
472		"DSDT at {dsdt_address:p} has invalid checksum"
473	);
474
475	// Try to find the "_S5_" object for SLP_TYPA in the DSDT AML bytecode.
476	// It may also be in an SSDT though.
477	search_s5_in_table(dsdt);
478}

480fn parse_ssdt(ssdt: AcpiTable<'_>) {
481	// We don't need to parse the SSDT if we already have information about the "_S5_" object
482	// (e.g. from the DSDT or a previous SSDT).
483	if SLP_TYPA.get().is_some() {
484		return;
485	}
486
487	// Otherwise, just try to find "_S5_" information in the AML bytecode of this SSDT.
488	search_s5_in_table(ssdt);
489}

500pub fn poweroff() {
501	if let (Some(mut pm1a_cnt_blk), Some(&slp_typa)) = (PM1A_CNT_BLK.get().cloned(), SLP_TYPA.get())
502	{
503		let bits = (u16::from(slp_typa) << 10) | SLP_EN;
504		debug!("Powering Off through ACPI (port {pm1a_cnt_blk:?}, bitmask {bits:#X})");
505		unsafe {
506			pm1a_cnt_blk.write(bits);
507		}
508	} else {
509		warn!("ACPI Power Off is not available");
510	}
511}

513pub fn init() {
514	// Detect the RSDP and get a pointer to either the XSDT (64-bit) or RSDT (32-bit), whichever is available.
515	// Both are called RSDT in the following.
516	let rsdp = detect_acpi().expect("Hermit requires an ACPI-compliant system");
517	let rsdt_physical_address = if rsdp.revision >= 2 {
518		PhysAddr::new(rsdp.xsdt_physical_address)
519	} else {
520		PhysAddr::new(rsdp.rsdt_physical_address.into())
521	};
522
523	// Map the RSDT.
524	let rsdt = AcpiTable::map(rsdt_physical_address);
525
526	// The RSDT contains pointers to all available ACPI tables.
527	// Iterate through them.
528	let mut current_address = rsdt.table_start_address();
529	while current_address < rsdt.table_end_address() {
530		// Depending on the RSDP revision, either an XSDT or an RSDT has been chosen above.
531		// The XSDT contains 64-bit pointers whereas the RSDT has 32-bit pointers.
532		let table_physical_address = if rsdp.revision >= 2 {
533			let address = unsafe {
534				PhysAddr::new(ptr::with_exposed_provenance::<u64>(current_address).read_unaligned())
535			};
536			current_address += mem::size_of::<u64>();
537			address
538		} else {
539			let address = unsafe {
540				PhysAddr::new(
541					ptr::with_exposed_provenance::<u32>(current_address)
542						.read_unaligned()
543						.into(),
544				)
545			};
546			current_address += mem::size_of::<u32>();
547			address
548		};
549
550		let table = AcpiTable::map(table_physical_address);
551		debug!("Found ACPI table: {}", table.header.signature());
552
553		if table.header.signature() == "APIC" {
554			// The "Multiple APIC Description Table" (MADT) aka "APIC Table" (APIC)
555			// Check and save the entire APIC table for the get_apic_table() call.
556			assert!(
557				verify_checksum(table.header_start_address(), table.header.length as usize).is_ok(),
558				"MADT at {table_physical_address:p} has invalid checksum"
559			);
560			MADT.set(table).unwrap();
561		} else if table.header.signature() == "FACP" {
562			// The "Fixed ACPI Description Table" (FADT) aka "Fixed ACPI Control Pointer" (FACP)
563			// Check and parse this table for the poweroff() call.
564			assert!(
565				verify_checksum(table.header_start_address(), table.header.length as usize).is_ok(),
566				"FADT at {table_physical_address:p} has invalid checksum"
567			);
568			parse_fadt(table);
569		} else if table.header.signature() == "SSDT" {
570			assert!(
571				verify_checksum(table.header_start_address(), table.header.length as usize).is_ok(),
572				"SSDT at {table_physical_address:p} has invalid checksum"
573			);
574			parse_ssdt(table);
575		} else if table.header.signature() == "MCFG" {
576			#[cfg(feature = "pci")]
577			{
578				assert!(
579					verify_checksum(table.header_start_address(), table.header.length as usize)
580						.is_ok(),
581					"MCFG at {table_physical_address:p} has invalid checksum"
582				);
583				MCFG.set(table).unwrap();
584			}
585		}
586	}
