72pub struct CommonStack {
73	/// start address of allocated virtual memory region
74	virt_addr: VirtAddr,
75	/// start address of allocated virtual memory region
76	phys_addr: PhysAddr,
77	/// total size of all stacks
78	total_size: usize,
79}

81pub enum TaskStacks {
82	Boot(BootStack),
83	Common(CommonStack),
84}
85
86impl TaskStacks {
87	/// Size of the debug marker at the very top of each stack.
88	///
89	/// We have a marker at the very top of the stack for debugging (`0xdeadbeef`), which should not be overridden.
90	pub const MARKER_SIZE: usize = 0x10;
91
92	pub fn new(size: usize) -> TaskStacks {
93		let user_stack_size = if size < KERNEL_STACK_SIZE {
94			KERNEL_STACK_SIZE
95		} else {
96			size.align_up(BasePageSize::SIZE as usize)
97		};
98		let total_size = user_stack_size + DEFAULT_STACK_SIZE + IST_SIZE;
99		let layout = PageLayout::from_size(total_size + 4 * BasePageSize::SIZE as usize).unwrap();
100		let page_range = PageAlloc::allocate(layout).unwrap();
101		let virt_addr = VirtAddr::from(page_range.start());
102
103		let frame_layout = PageLayout::from_size(total_size).unwrap();
104		let frame_range = FrameAlloc::allocate(frame_layout)
105			.expect("Failed to allocate Physical Memory for TaskStacks");
106		let phys_addr = PhysAddr::from(frame_range.start());
107
108		debug!(
109			"Create stacks at {:p} with a size of {} KB",
110			virt_addr,
111			total_size >> 10
112		);
113
114		let mut flags = PageTableEntryFlags::empty();
115		flags.normal().writable().execute_disable();
116
117		// map IST1 into the address space
118		crate::arch::mm::paging::map::<BasePageSize>(
119			virt_addr + BasePageSize::SIZE,
120			phys_addr,
121			IST_SIZE / BasePageSize::SIZE as usize,
122			flags,
123		);
124
125		// map kernel stack into the address space
126		crate::arch::mm::paging::map::<BasePageSize>(
127			virt_addr + IST_SIZE + 2 * BasePageSize::SIZE,
128			phys_addr + IST_SIZE,
129			DEFAULT_STACK_SIZE / BasePageSize::SIZE as usize,
130			flags,
131		);
132
133		// map user stack into the address space
134		crate::arch::mm::paging::map::<BasePageSize>(
135			virt_addr + IST_SIZE + DEFAULT_STACK_SIZE + 3 * BasePageSize::SIZE,
136			phys_addr + IST_SIZE + DEFAULT_STACK_SIZE,
137			user_stack_size / BasePageSize::SIZE as usize,
138			flags,
139		);
140
141		// clear user stack
142		unsafe {
143			ptr::write_bytes(
144				(virt_addr + IST_SIZE + DEFAULT_STACK_SIZE + 3 * BasePageSize::SIZE)
145					.as_mut_ptr::<u8>(),
146				0,
147				user_stack_size,
148			);
149		}
150
151		TaskStacks::Common(CommonStack {
152			virt_addr,
153			phys_addr,
154			total_size,
155		})
156	}
157
158	pub fn from_boot_stacks() -> TaskStacks {
159		let tss = unsafe { &*CoreLocal::get().tss.get() };
160		let stack = VirtAddr::new(
161			tss.privilege_stack_table[0].as_u64() + Self::MARKER_SIZE as u64
162				- KERNEL_STACK_SIZE as u64,
163		);
164		debug!("Using boot stack {stack:p}");
165		let ist1 = VirtAddr::new(
166			tss.interrupt_stack_table[0].as_u64() + Self::MARKER_SIZE as u64 - IST_SIZE as u64,
167		);
168		debug!("IST1 is located at {ist1:p}");
169
170		TaskStacks::Boot(BootStack { stack, ist1 })
171	}
172
173	pub fn get_user_stack_size(&self) -> usize {
174		match self {
175			TaskStacks::Boot(_) => 0,
176			TaskStacks::Common(stacks) => stacks.total_size - DEFAULT_STACK_SIZE - IST_SIZE,
177		}
178	}
179
180	pub fn get_user_stack(&self) -> VirtAddr {
181		match self {
182			TaskStacks::Boot(_) => VirtAddr::zero(),
183			TaskStacks::Common(stacks) => {
184				stacks.virt_addr + IST_SIZE + DEFAULT_STACK_SIZE + 3 * BasePageSize::SIZE
185			}
186		}
187	}
188
189	pub fn get_kernel_stack(&self) -> VirtAddr {
190		match self {
191			TaskStacks::Boot(stacks) => stacks.stack,
192			TaskStacks::Common(stacks) => stacks.virt_addr + IST_SIZE + 2 * BasePageSize::SIZE,
193		}
194	}
195
196	pub fn get_kernel_stack_size(&self) -> usize {
197		match self {
198			TaskStacks::Boot(_) => KERNEL_STACK_SIZE,
199			TaskStacks::Common(_) => DEFAULT_STACK_SIZE,
200		}
201	}
202
203	pub fn get_interrupt_stack(&self) -> VirtAddr {
204		match self {
205			TaskStacks::Boot(stacks) => stacks.ist1,
206			TaskStacks::Common(stacks) => stacks.virt_addr + BasePageSize::SIZE,
207		}
208	}
209
210	pub fn get_interrupt_stack_size(&self) -> usize {
211		IST_SIZE
212	}
213}

215impl Drop for TaskStacks {
216	fn drop(&mut self) {
217		// we should never deallocate a boot stack
218		match self {
219			TaskStacks::Boot(_) => {}
220			TaskStacks::Common(stacks) => {
221				debug!(
222					"Deallocating stacks at {:p} with a size of {} KB",
223					stacks.virt_addr,
224					stacks.total_size >> 10,
225				);
226
227				if !env::is_uefi() {
228					crate::arch::mm::paging::unmap::<BasePageSize>(
229						stacks.virt_addr,
230						stacks.total_size / BasePageSize::SIZE as usize + 4,
231					);
232				}
233				let range = PageRange::from_start_len(
234					stacks.virt_addr.as_usize(),
235					stacks.total_size + 4 * BasePageSize::SIZE as usize,
236				)
237				.unwrap();
238				unsafe {
239					PageAlloc::deallocate(range);
240				}
241
242				let range =
243					PageRange::from_start_len(stacks.phys_addr.as_usize(), stacks.total_size)
244						.unwrap();
245				unsafe {
246					FrameAlloc::deallocate(range);
247				}
248			}
249		}
250	}
251}

258#[cfg(target_os = "none")]
259#[unsafe(naked)]
260extern "C" fn task_start(_f: extern "C" fn(usize), _arg: usize, _user_stack: u64) -> ! {
261	// `f` is in the `rdi` register
262	// `arg` is in the `rsi` register
263	// `user_stack` is in the `rdx` register
264
265	naked_asm!(
266		"mov rsp, rdx",
267		"sti",
268		"jmp {task_entry}",
269		task_entry = sym task_entry,
270	)
271}
272
273extern "C" fn task_entry(func: extern "C" fn(usize), arg: usize) -> ! {
274	// Call the actual entry point of the task.
275	func(arg);
276
277	// Exit task
278	debug!("Exit thread with error code 0!");
279	core_scheduler().exit(0)
280}

282impl TaskFrame for Task {
283	fn create_stack_frame(&mut self, func: unsafe extern "C" fn(usize), arg: usize) {
284		// Check if TLS is allocated already and if the task uses thread-local storage.
285		#[cfg(not(feature = "common-os"))]
286		if self.tls.is_none() {
287			use crate::scheduler::task::tls::Tls;
288
289			self.tls = Tls::from_env();
290		}
291
292		unsafe {
293			// Set a marker for debugging at the very top.
294			let mut stack = self.stacks.get_kernel_stack() + self.stacks.get_kernel_stack_size()
295				- TaskStacks::MARKER_SIZE;
296			*stack.as_mut_ptr::<u64>() = 0xdead_beefu64;
297
298			// Put the State structure expected by the ASM switch() function on the stack.
299			stack -= mem::size_of::<State>();
300
301			let state = stack.as_mut_ptr::<State>();
302			#[cfg(not(feature = "common-os"))]
303			if let Some(tls) = &self.tls {
304				(*state).fs = tls.thread_ptr().addr() as u64;
305			}
306			(*state).rip = task_start as usize as u64;
307			(*state).rdi = func as usize as u64;
308			(*state).rsi = arg as u64;
309
310			// per default we disable interrupts
311			(*state).rflags = 0x1202u64;
312
313			// Set the task's stack pointer entry to the stack we have just crafted.
314			self.last_stack_pointer = stack;
315			self.user_stack_pointer = self.stacks.get_user_stack()
316				+ self.stacks.get_user_stack_size()
317				- TaskStacks::MARKER_SIZE;
318
319			// rdx is required to initialize the stack
320			(*state).rdx = self.user_stack_pointer.as_u64() - mem::size_of::<u64>() as u64;
321		}
322	}
323}

325extern "x86-interrupt" fn timer_handler(_stack_frame: interrupts::ExceptionStackFrame) {
326	increment_irq_counter(apic::TIMER_INTERRUPT_NUMBER);
327	core_scheduler().handle_waiting_tasks();
328	apic::eoi();
329	core_scheduler().reschedule();
330}
331
332pub fn install_timer_handler() {
333	unsafe {
334		let mut idt = IDT.lock();
335		idt[apic::TIMER_INTERRUPT_NUMBER]
336			.set_handler_fn(timer_handler)
337			.set_stack_index(0);
338	}
339	interrupts::add_irq_name(apic::TIMER_INTERRUPT_NUMBER - 32, "Timer");
340}
