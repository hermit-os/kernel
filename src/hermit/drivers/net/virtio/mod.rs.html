55fn determine_mtu(dev_cfg: &NetDevCfg) -> u16 {
56	// If VIRTIO_NET_F_MTU is negotiated, "the driver uses mtu as the maximum MTU value"
57	// (VirtIO specification, 5.1.3, "Feature bits")
58	if dev_cfg.features.contains(virtio::net::F::MTU) {
59		dev_cfg.raw.as_ptr().mtu().read().to_ne()
60	} else {
61		// Otherwise, we can just use the MTU we want to use
62		mtu()
63	}
64}
65
66fn determine_rx_buf_size(dev_cfg: &NetDevCfg) -> u32 {
67	// See Virtio specification v1.1 - 5.1.6.3.1 and 5.1.4.2
68
69	// Our desired minimum buffer size - we want it to be at least the MTU generally
70	let mut min_buf_size = determine_mtu(dev_cfg).into();
71
72	// If VIRTIO_NET_F_MRG_RXBUF is negotiated, each buffer MUST be at least the size of the struct virtio_net_hdr.
73	// We just use MTU in that case, but otherwise...
74	if dev_cfg.features.contains(virtio::net::F::MRG_RXBUF)
75		&& let Some(my_mrg_rxbuf_size) = hermit_var!("HERMIT_MRG_RXBUF_SIZE")
76	{
77		let my_mrg_rxbuf_size = u32::from_str(&my_mrg_rxbuf_size).unwrap();
78		assert!(
79			my_mrg_rxbuf_size > 0,
80			"VIRTIO does not allow buffer elements of size 0."
81		);
82		min_buf_size = my_mrg_rxbuf_size;
83	} else {
84		// If [...] are negotiated, the driver SHOULD populate the receive queue(s) with buffers of at least 65562 bytes.
85		if dev_cfg.features.contains(virtio::net::F::GUEST_TSO4)
86			|| dev_cfg.features.contains(virtio::net::F::GUEST_TSO6)
87			|| dev_cfg.features.contains(virtio::net::F::GUEST_UFO)
88		{
89			min_buf_size = u32::max(min_buf_size, 65562 - size_of::<Hdr>() as u32);
90		} else {
91			// Otherwise, the driver SHOULD populate the receive queue(s) with buffers of at least 1526 bytes.
92			min_buf_size = u32::max(min_buf_size, 1526 - size_of::<Hdr>() as u32);
93		}
94	}
95
96	min_buf_size
97}

99pub struct RxQueues {
100	vqs: Vec<VirtQueue>,
101	buf_size: u32,
102}
103
104impl RxQueues {
105	pub fn new(vqs: Vec<VirtQueue>, dev_cfg: &NetDevCfg) -> Self {
106		Self {
107			vqs,
108			buf_size: determine_rx_buf_size(dev_cfg),
109		}
110	}
111
112	/// Adds a given queue to the underlying vector and populates the queue with RecvBuffers.
113	///
114	/// Queues are all populated according to Virtio specification v1.1. - 5.1.6.3.1
115	fn add(&mut self, mut vq: VirtQueue) {
116		let num_bufs = vq.size() / constants::BUFF_PER_PACKET;
117		fill_queue(&mut vq, num_bufs, self.buf_size);
118		self.vqs.push(vq);
119	}
120
121	fn get_next(&mut self) -> Option<UsedBufferToken> {
122		self.vqs[0].try_recv().ok()
123	}
124
125	fn enable_notifs(&mut self) {
126		for vq in &mut self.vqs {
127			vq.enable_notifs();
128		}
129	}
130
131	fn disable_notifs(&mut self) {
132		for vq in &mut self.vqs {
133			vq.disable_notifs();
134		}
135	}
136
137	fn has_packet(&self) -> bool {
138		self.vqs.iter().any(|vq| vq.has_used_buffers())
139	}
140}
141
142fn buffer_token_from_hdr(
143	hdr: Box<MaybeUninit<Hdr>, DeviceAlloc>,
144	buf_size: u32,
145) -> AvailBufferToken {
146	AvailBufferToken::new(SmallVec::new(), {
147		SmallVec::from_buf([
148			BufferElem::Sized(hdr),
149			BufferElem::Vector(Vec::with_capacity_in(
150				buf_size.try_into().unwrap(),
151				DeviceAlloc,
152			)),
153		])
154	})
155	.unwrap()
156}
157
158fn fill_queue(vq: &mut VirtQueue, num_bufs: u16, buf_size: u32) {
159	for _ in 0..num_bufs {
160		let buff_tkn = buffer_token_from_hdr(Box::<Hdr, _>::new_uninit_in(DeviceAlloc), buf_size);
161
162		// BufferTokens are directly provided to the queue
163		// TransferTokens are directly dispatched
164		// Transfers will be awaited at the queue
165		if let Err(err) = vq.dispatch(buff_tkn, false, BufferType::Direct) {
166			error!("{err:#?}");
167			break;
168		}
169	}
170}

172/// Structure which handles transmission of packets and delegation
173/// to the respective queue structures.
174pub struct TxQueues {
175	vqs: Vec<VirtQueue>,
176	buf_size: u32,
177}
178
179impl TxQueues {
180	pub fn new(vqs: Vec<VirtQueue>, dev_cfg: &NetDevCfg) -> Self {
181		Self {
182			vqs,
183			buf_size: determine_mtu(dev_cfg).into(),
184		}
185	}
186
187	#[allow(dead_code)]
188	fn enable_notifs(&mut self) {
189		for vq in &mut self.vqs {
190			vq.enable_notifs();
191		}
192	}
193
194	#[allow(dead_code)]
195	fn disable_notifs(&mut self) {
196		for vq in &mut self.vqs {
197			vq.disable_notifs();
198		}
199	}
200
201	/// Polls all queues for buffers whose transmission has been completed and returns the number of such buffers.
202	fn poll(&mut self) -> u32 {
203		let mut released_buffers = 0u32;
204		for vq in &mut self.vqs {
205			// We don't do anything with the buffers but we need to receive them for the
206			// ring slots to be emptied and the memory from the previous transfers to be freed.
207			while vq.try_recv().is_ok() {
208				released_buffers += 1;
209			}
210		}
211		released_buffers
212	}
213
214	fn add(&mut self, vq: VirtQueue) {
215		// Currently we are doing nothing with the additional queues. They are inactive and might be used in the
216		// future
217		self.vqs.push(vq);
218	}
219}

254impl Drop for TxToken<'_> {
255	fn drop(&mut self) {
256		*self.send_capacity += u32::from(BUFF_PER_PACKET);
257	}
258}
259
260impl smoltcp::phy::TxToken for TxToken<'_> {
261	fn consume<R, F>(self, len: usize, f: F) -> R
262	where
263		F: FnOnce(&mut [u8]) -> R,
264	{
265		// When the token is consumed, the capacity cannot be returned until its buffer is marked by the device as used.
266		// Thus, we bypass the Drop implementation that would do that prematurely and let the call to poll in the next
267		// call to this function return the capacity.
268		let mut token = ManuallyDrop::new(self);
269		assert!(len <= usize::try_from(token.send_vqs.buf_size).unwrap());
270		let mut packet = Vec::with_capacity_in(len, DeviceAlloc);
271		let result = unsafe {
272			let result = f(packet.spare_capacity_mut().assume_init_mut());
273			packet.set_len(len);
274			result
275		};
276
277		let mut header = Box::new_in(<Hdr as Default>::default(), DeviceAlloc);
278
279		// If a checksum calculation by the host is necessary, we have to inform the host within the header
280		// see Virtio specification 5.1.6.2
281		if let Some((ip_header_len, csum_offset)) =
282			VirtioNetDriver::should_request_checksum(&token.checksums, &mut packet)
283		{
284			header.flags = HdrF::NEEDS_CSUM;
285			header.csum_start =
286				(u16::try_from(ETHERNET_HEADER_LEN).unwrap() + ip_header_len).into();
287			header.csum_offset = csum_offset.into();
288		}
289
290		let buff_tkn = AvailBufferToken::new(
291			SmallVec::from_buf([BufferElem::Sized(header), BufferElem::Vector(packet)]),
292			SmallVec::new(),
293		)
294		.unwrap();
295
296		token.send_vqs.vqs[0]
297			.dispatch(buff_tkn, false, BufferType::Direct)
298			.unwrap();
299
300		result
301	}
302}

304pub struct RxToken<'a> {
305	recv_vqs: &'a mut RxQueues,
306	is_mrg_rxbuf_enabled: bool,
307}
308
309impl smoltcp::phy::RxToken for RxToken<'_> {
310	fn consume<R, F>(self, f: F) -> R
311	where
312		F: FnOnce(&[u8]) -> R,
313	{
314		let Some(mut buffer_tkn) = self.recv_vqs.get_next() else {
315			// We overpromised a frame. The best we can do is to provide an empty frame to smoltcp and let it handle it as a faulty reception.
316			return f(&[]);
317		};
318		// Safety: any buffers that do not start with a `Hdr` must have been consumed by the previous call
319		// to this function.
320		let first_header = unsafe {
321			buffer_tkn
322				.used_recv_buff
323				.pop_front_downcast::<Hdr>()
324				.unwrap()
325		};
326		let first_packet = buffer_tkn.used_recv_buff.pop_front_vec().unwrap();
327
328		// According to VIRTIO spec v1.2 sec. 5.1.6.3.2, "num_buffers will always be 1 if VIRTIO_NET_F_MRG_RXBUF is not negotiated."
329		// Unfortunately, NVIDIA MLX5 does not comply with this requirement and we have to manually set the value to the correct one.
330		let num_buffers = if self.is_mrg_rxbuf_enabled {
331			first_header.num_buffers.to_ne()
332		} else {
333			1
334		};
335
336		let mut combined_packets = first_packet;
337
338		let first_tkn = buffer_token_from_hdr(
339			// SAFETY: Box<T> -> Box<MaybeUninit<T>> is sound
340			unsafe {
341				transmute::<Box<Hdr, DeviceAlloc>, Box<MaybeUninit<Hdr>, DeviceAlloc>>(first_header)
342			},
343			self.recv_vqs.buf_size,
344		);
345		self.recv_vqs.vqs[0]
346			.dispatch(first_tkn, false, BufferType::Direct)
347			.unwrap();
348
349		for _ in 1..num_buffers {
350			let mut buffer_tkn = self.recv_vqs.get_next().unwrap();
351			// The descriptor that was meant for the header of another frame was used for a portion of the current frame's contents.
352			// Thus, we cannot cast it to a Hdr.
353			let (header_descriptor, used_len) = buffer_tkn.used_recv_buff.pop_front_raw().unwrap();
354			combined_packets.extend_from_slice(unsafe {
355				core::slice::from_raw_parts((&raw const *header_descriptor).cast::<u8>(), used_len)
356			});
357
358			let packet = buffer_tkn.used_recv_buff.pop_front_vec().unwrap();
359			combined_packets.extend_from_slice(&packet);
360
361			let header = header_descriptor.downcast::<MaybeUninit<Hdr>>().unwrap();
362
363			let tkn = buffer_token_from_hdr(
364				// SAFETY: Box<T> -> Box<MaybeUninit<T>> is sound
365				header,
366				self.recv_vqs.buf_size,
367			);
368			self.recv_vqs.vqs[0]
369				.dispatch(tkn, false, BufferType::Direct)
370				.unwrap();
371		}
372
373		f(&combined_packets)
374	}
375}

404		let status = self.isr_stat.is_queue_interrupt();

406		#[cfg(not(feature = "pci"))]
407		if status.contains(virtio::mmio::InterruptStatus::CONFIGURATION_CHANGE_NOTIFICATION) {
408			info!("Configuration changes are not possible! Aborting");
409			todo!("Implement possibility to change config on the fly...")
410		}
411
412		#[cfg(feature = "pci")]
413		if status.contains(virtio::pci::IsrStatus::DEVICE_CONFIGURATION_INTERRUPT) {
414			info!("Configuration changes are not possible! Aborting");
415			todo!("Implement possibility to change config on the fly...")
416		}
417
418		self.isr_stat.acknowledge();
419	}

422impl smoltcp::phy::Device for VirtioNetDriver {
423	type TxToken<'a> = TxToken<'a>;
424	type RxToken<'a> = RxToken<'a>;
425
426	fn capabilities(&self) -> DeviceCapabilities {
427		let mut device_capabilities = DeviceCapabilities::default();
428		device_capabilities.medium = smoltcp::phy::Medium::Ethernet;
429		device_capabilities.max_transmission_unit = self.inner.mtu.into();
430		device_capabilities.max_burst_size =
431			Some(usize::try_from(self.inner.send_capacity).unwrap() / usize::from(BUFF_PER_PACKET));
432		device_capabilities.checksum = self.checksums.clone();
433		device_capabilities
434	}
435
436	fn receive(
437		&mut self,
438		_timestamp: smoltcp::time::Instant,
439	) -> Option<(Self::RxToken<'_>, Self::TxToken<'_>)> {
440		if self.inner.recv_vqs.has_packet() && {
441			self.free_up_send_capacity();
442			self.inner.send_capacity >= u32::from(BUFF_PER_PACKET)
443		} {
444			self.inner.send_capacity -= u32::from(BUFF_PER_PACKET);
445			Some((
446				RxToken {
447					recv_vqs: &mut self.inner.recv_vqs,
448					is_mrg_rxbuf_enabled: self.dev_cfg.features.contains(virtio::net::F::MRG_RXBUF),
449				},
450				TxToken {
451					send_vqs: &mut self.inner.send_vqs,
452					checksums: self.checksums.clone(),
453					send_capacity: &mut self.inner.send_capacity,
454				},
455			))
456		} else {
457			None
458		}
459	}
460
461	fn transmit(&mut self, _timestamp: smoltcp::time::Instant) -> Option<Self::TxToken<'_>> {
462		self.free_up_send_capacity();
463		if self.inner.send_capacity >= u32::from(BUFF_PER_PACKET) {
464			self.inner.send_capacity -= u32::from(BUFF_PER_PACKET);
465			Some(TxToken {
466				send_vqs: &mut self.inner.send_vqs,
467				checksums: self.checksums.clone(),
468				send_capacity: &mut self.inner.send_capacity,
469			})
470		} else {
471			None
472		}
473	}
474}

564
565	/// If necessary, sets the TCP or UDP checksum field to the checksum of the
566	/// pseudo-header and returns the IP header length and the checksum offset.
567	/// Otherwise, returns None.
568	fn should_request_checksum<T: AsRef<[u8]> + AsMut<[u8]>>(
569		checksums: &ChecksumCapabilities,
570		frame: T,
571	) -> Option<(u16, u16)> {
572		if checksums.tcp.tx() && checksums.udp.tx() {
573			return None;
574		}
575
576		let ip_header_len: u16;
577		let ip_packet_len: usize;
578		let protocol;
579		let pseudo_header_checksum;
580		let mut ethernet_frame = EthernetFrame::new_unchecked(frame);
581		match ethernet_frame.ethertype() {
582			smoltcp::wire::EthernetProtocol::Ipv4 => {
583				let ip_packet = Ipv4Packet::new_unchecked(&*ethernet_frame.payload_mut());
584				ip_header_len = ip_packet.header_len().into();
585				ip_packet_len = ip_packet.total_len().into();
586				protocol = ip_packet.next_header();
587				pseudo_header_checksum =
588					partial_checksum::ipv4_pseudo_header_partial_checksum(&ip_packet);
589			}
590			smoltcp::wire::EthernetProtocol::Ipv6 => {
591				let ip_packet = Ipv6Packet::new_unchecked(&*ethernet_frame.payload_mut());
592				ip_header_len = ip_packet.header_len().try_into().expect(
593					"VIRTIO does not support IP headers that are longer than u16::MAX bytes.",
594				);
595				ip_packet_len = ip_packet.total_len();
596				protocol = ip_packet.next_header();
597				pseudo_header_checksum =
598					partial_checksum::ipv6_pseudo_header_partial_checksum(&ip_packet);
599			}
600			// If the Ethernet protocol is not one of these two above, for which we know there may be a checksum field,
601			// we default to not asking for checksum, as otherwise the frame will be corrupted by the device trying
602			// to write the checksum.
603			_ => return None,
604		};
605
606		let csum_offset;
607		let ip_payload = &mut ethernet_frame.payload_mut()[ip_header_len.into()..ip_packet_len];
608		// Like the Ethernet protocol check, we check for IP protocols for which we know the location of the checksum field.
609		if protocol == smoltcp::wire::IpProtocol::Tcp && !checksums.tcp.tx() {
610			let mut tcp_packet = smoltcp::wire::TcpPacket::new_unchecked(ip_payload);
611			tcp_packet.set_checksum(pseudo_header_checksum);
612			csum_offset = 16;
613		} else if protocol == smoltcp::wire::IpProtocol::Udp && !checksums.udp.tx() {
614			let mut udp_packet = smoltcp::wire::UdpPacket::new_unchecked(ip_payload);
615			udp_packet.set_checksum(pseudo_header_checksum);
616			csum_offset = 6;
617		} else {
618			return None;
619		};
620
621		Some((ip_header_len, csum_offset))
622	}
623
624	fn free_up_send_capacity(&mut self) {
625		// We need to poll to get the queue to remove elements from the table and open up capacity if possible.
626		self.inner.send_capacity += self.inner.send_vqs.poll() * u32::from(BUFF_PER_PACKET);
627	}
628}

630impl VirtioNetDriver<Uninit> {
631	/// Initializes the device in adherence to specification. Returns Some(VirtioNetError)
632	/// upon failure and None in case everything worked as expected.
633	///
634	/// See Virtio specification v1.1. - 3.1.1.
635	///                      and v1.1. - 5.1.5
636	pub fn init_dev(mut self) -> Result<VirtioNetDriver<Init>, VirtioNetError> {
637		// Reset
638		self.com_cfg.reset_dev();
639
640		// Indicate device, that OS noticed it
641		self.com_cfg.ack_dev();
642
643		// Indicate device, that driver is able to handle it
644		self.com_cfg.set_drv();
645
646		let minimal_features = virtio::net::F::VERSION_1 | virtio::net::F::MAC;
647
648		// If wanted, push new features into feats here:
649		let mut features = minimal_features
650			// Indirect descriptors can be used
651			| virtio::net::F::INDIRECT_DESC
652			// Packed Vq can be used
653			| virtio::net::F::RING_PACKED
654			| virtio::net::F::NOTIFICATION_DATA
655			// MTU setting can be used
656			| virtio::net::F::MTU
657			// Driver can merge receive buffers
658			| virtio::net::F::MRG_RXBUF
659			// the link status can be announced
660			| virtio::net::F::STATUS
661			// control queue support
662			| virtio::net::F::CTRL_VQ
663			// Multiqueue support
664			| virtio::net::F::MQ
665			// Checksum calculation can partially be offloaded to the device
666			| virtio::net::F::CSUM;
667
668		// Currently the driver does NOT support the features below.
669		// In order to provide functionality for these, the driver
670		// needs to take care of calculating checksum.
671		// | virtio::net::F::GUEST_TSO4
672		// | virtio::net::F::GUEST_TSO6
673
674		// Negotiate features with device. Automatically reduces selected feats in order to meet device capabilities.
675		// Aborts in case incompatible features are selected by the driver or the device does not support min_feat_set.
676		match self.negotiate_features(features) {
677			Ok(()) => info!(
678				"Driver found a subset of features for virtio device {:x}. Features are: {features:?}",
679				self.dev_cfg.dev_id
680			),
681			Err(vnet_err) => {
682				match vnet_err {
683					VirtioNetError::FeatureRequirementsNotMet(features) => {
684						error!(
685							"Network drivers feature set {features:?} does not satisfy rules in section 5.1.3.1 of specification v1.1. Aborting!"
686						);
687						return Err(vnet_err);
688					}
689					VirtioNetError::IncompatibleFeatureSets(drv_feats, dev_feats) => {
690						// Create a new matching feature set for device and driver if the minimal set is met!
691						if !dev_feats.contains(minimal_features) {
692							error!(
693								"Device features set, does not satisfy minimal features needed. Aborting!"
694							);
695							return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
696						}
697
698						let common_features = drv_feats & dev_feats;
699						if common_features.is_empty() {
700							error!(
701								"Feature negotiation failed with minimal feature set. Aborting!"
702							);
703							return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
704						}
705						features = common_features;
706
707						match self.negotiate_features(features) {
708							Ok(()) => info!(
709								"Driver found a subset of features for virtio device {:x}. Features are: {features:?}",
710								self.dev_cfg.dev_id
711							),
712							Err(vnet_err) => match vnet_err {
713								VirtioNetError::FeatureRequirementsNotMet(features) => {
714									error!(
715										"Network device offers a feature set {features:?} when used completely does not satisfy rules in section 5.1.3.1 of specification v1.1. Aborting!"
716									);
717									return Err(vnet_err);
718								}
719								_ => {
720									error!(
721										"Feature Set after reduction still not usable. Set: {features:?}. Aborting!"
722									);
723									return Err(vnet_err);
724								}
725							},
726						}
727					}
728					VirtioNetError::FailFeatureNeg(_) => {
729						error!(
730							"Wanted set of features is NOT supported by device. Set: {features:?}"
731						);
732						return Err(vnet_err);
733					}
734					#[cfg(feature = "pci")]
735					VirtioNetError::NoDevCfg(_) => {
736						error!("No device config found.");
737						return Err(vnet_err);
738					}
739				}
740			}
741		}
742
743		// Indicates the device, that the current feature set is final for the driver
744		// and will not be changed.
745		self.com_cfg.features_ok();
746
747		// Checks if the device has accepted final set. This finishes feature negotiation.
748		if self.com_cfg.check_features() {
749			info!(
750				"Features have been negotiated between virtio network device {:x} and driver.",
751				self.dev_cfg.dev_id
752			);
753			// Set feature set in device config fur future use.
754			self.dev_cfg.features = features;
755		} else {
756			return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
757		}
758
759		let mut inner = Init {
760			mtu: determine_mtu(&self.dev_cfg),
761			ctrl_vq: None,
762			recv_vqs: RxQueues::new(Vec::new(), &self.dev_cfg),
763			send_vqs: TxQueues::new(Vec::new(), &self.dev_cfg),
764			send_capacity: 0,
765		};
766
767		debug!("Using RX buffer size of {}", inner.recv_vqs.buf_size);
768
769		self.dev_spec_init(&mut inner)?;
770		info!(
771			"Device specific initialization for Virtio network device {:x} finished",
772			self.dev_cfg.dev_id
773		);
774
775		// At this point the device is "live"
776		self.com_cfg.drv_ok();
777
778		if self.dev_cfg.features.contains(virtio::net::F::CSUM)
779			&& self.dev_cfg.features.contains(virtio::net::F::GUEST_CSUM)
780		{
781			self.checksums.udp = Checksum::None;
782			self.checksums.tcp = Checksum::None;
783		} else if self.dev_cfg.features.contains(virtio::net::F::CSUM) {
784			self.checksums.udp = Checksum::Rx;
785			self.checksums.tcp = Checksum::Rx;
786		} else if self.dev_cfg.features.contains(virtio::net::F::GUEST_CSUM) {
787			self.checksums.udp = Checksum::Tx;
788			self.checksums.tcp = Checksum::Tx;
789		}
790		debug!("{:?}", self.checksums);
791
792		Ok(VirtioNetDriver {
793			dev_cfg: self.dev_cfg,
794			com_cfg: self.com_cfg,
795			isr_stat: self.isr_stat,
796			notif_cfg: self.notif_cfg,
797			inner,
798			num_vqs: self.num_vqs,
799			irq: self.irq,
800			checksums: self.checksums,
801		})
802	}
803
804	/// Negotiates a subset of features, understood and wanted by both the OS
805	/// and the device.
806	fn negotiate_features(
807		&mut self,
808		driver_features: virtio::net::F,
809	) -> Result<(), VirtioNetError> {
810		let device_features = virtio::net::F::from(self.com_cfg.dev_features());
811
812		if device_features.requirements_satisfied() {
813			info!("Feature set wanted by network driver are in conformance with specification.");
814		} else {
815			return Err(VirtioNetError::FeatureRequirementsNotMet(device_features));
816		}
817
818		if device_features.contains(driver_features) {
819			// If device supports subset of features write feature set to common config
820			self.com_cfg.set_drv_features(driver_features.into());
821			Ok(())
822		} else {
823			Err(VirtioNetError::IncompatibleFeatureSets(
824				driver_features,
825				device_features,
826			))
827		}
828	}
829
830	/// Device Specific initialization according to Virtio specifictation v1.1. - 5.1.5
831	fn dev_spec_init(&mut self, inner: &mut Init) -> Result<(), VirtioNetError> {
832		self.virtqueue_init(inner)?;
833		info!("Network driver successfully initialized virtqueues.");
834
835		// Add a control if feature is negotiated
836		if self.dev_cfg.features.contains(virtio::net::F::CTRL_VQ) {
837			let mut ctrl_vq = if self.dev_cfg.features.contains(virtio::net::F::RING_PACKED) {
838				VirtQueue::Packed(
839					PackedVq::new(
840						&mut self.com_cfg,
841						&self.notif_cfg,
842						VIRTIO_MAX_QUEUE_SIZE,
843						self.num_vqs,
844						self.dev_cfg.features.into(),
845					)
846					.unwrap(),
847				)
848			} else {
849				VirtQueue::Split(
850					SplitVq::new(
851						&mut self.com_cfg,
852						&self.notif_cfg,
853						VIRTIO_MAX_QUEUE_SIZE,
854						self.num_vqs,
855						self.dev_cfg.features.into(),
856					)
857					.unwrap(),
858				)
859			};
860
861			ctrl_vq.enable_notifs();
862			inner.ctrl_vq = Some(ctrl_vq);
863		}
864
865		Ok(())
866	}
867
868	/// Initialize virtqueues via the queue interface and populates receiving queues
869	fn virtqueue_init(&mut self, inner: &mut Init) -> Result<(), VirtioNetError> {
870		// We are assuming here, that the device single source of truth is the
871		// device specific configuration. Hence we do NOT check if
872		//
873		// max_virtqueue_pairs + 1 < num_queues
874		//
875		// - the plus 1 is due to the possibility of an existing control queue
876		// - the num_queues is found in the ComCfg struct of the device and defines the maximal number
877		// of supported queues.
878		if self.dev_cfg.features.contains(virtio::net::F::MQ) {
879			if self
880				.dev_cfg
881				.raw
882				.as_ptr()
883				.max_virtqueue_pairs()
884				.read()
885				.to_ne() * 2 >= MAX_NUM_VQ
886			{
887				self.num_vqs = MAX_NUM_VQ;
888			} else {
889				self.num_vqs = self
890					.dev_cfg
891					.raw
892					.as_ptr()
893					.max_virtqueue_pairs()
894					.read()
895					.to_ne() * 2;
896			}
897		} else {
898			// Minimal number of virtqueues defined in the standard v1.1. - 5.1.5 Step 1
899			self.num_vqs = 2;
900		}
901
902		// The loop is running from 0 to num_vqs and the indexes are provided in this way
903		// in order to allow the indexes of the queues to be in a form of:
904		//
905		// index i for receive queue
906		// index i+1 for send queue
907		//
908		// as it is wanted by the network network device.
909		// see Virtio specification v1.1. - 5.1.2
910		// Assure that we have always an even number of queues (i.e. pairs of queues).
911		assert_eq!(self.num_vqs % 2, 0);
912
913		for i in 0..(self.num_vqs / 2) {
914			if self.dev_cfg.features.contains(virtio::net::F::RING_PACKED) {
915				let mut vq = PackedVq::new(
916					&mut self.com_cfg,
917					&self.notif_cfg,
918					VIRTIO_MAX_QUEUE_SIZE,
919					2 * i,
920					self.dev_cfg.features.into(),
921				)
922				.unwrap();
923				// Interrupt for receiving packets is wanted
924				vq.enable_notifs();
925
926				inner.recv_vqs.add(VirtQueue::Packed(vq));
927
928				let mut vq = PackedVq::new(
929					&mut self.com_cfg,
930					&self.notif_cfg,
931					VIRTIO_MAX_QUEUE_SIZE,
932					2 * i + 1,
933					self.dev_cfg.features.into(),
934				)
935				.unwrap();
936				// Interrupt for communicating that a sent packet left, is not needed
937				vq.disable_notifs();
938
939				inner.send_capacity += u32::from(vq.size());
940				inner.send_vqs.add(VirtQueue::Packed(vq));
941			} else {
942				let mut vq = SplitVq::new(
943					&mut self.com_cfg,
944					&self.notif_cfg,
945					VIRTIO_MAX_QUEUE_SIZE,
946					2 * i,
947					self.dev_cfg.features.into(),
948				)
949				.unwrap();
950				// Interrupt for receiving packets is wanted
951				vq.enable_notifs();
952
953				inner.recv_vqs.add(VirtQueue::Split(vq));
954
955				let mut vq = SplitVq::new(
956					&mut self.com_cfg,
957					&self.notif_cfg,
958					VIRTIO_MAX_QUEUE_SIZE,
959					2 * i + 1,
960					self.dev_cfg.features.into(),
961				)
962				.unwrap();
963				// Interrupt for communicating that a sent packet left, is not needed
964				vq.disable_notifs();
965				inner.send_capacity += u32::from(vq.size());
966				inner.send_vqs.add(VirtQueue::Split(vq));
967			}
968		}
969
970		Ok(())
971	}

1013/// The checksum functions in this module only calculate the one's complement sum for the pseudo-header
1014/// and their results are meant to be combined with the TCP payload to calculate the real checksum.
1015/// They are only useful for the VIRTIO driver with the checksum offloading feature.
1016///
1017/// The calculations here can theoretically be made faster by exploiting the properties described in
1018/// [RFC 1071 section 2](https://www.rfc-editor.org/rfc/rfc1071).
1019mod partial_checksum {
1020	use smoltcp::wire::{Ipv4Packet, Ipv6Packet};
1021
1022	fn addr_sum<const N: usize>(addr: &[u8; N]) -> u16 {
1023		let mut sum = 0;
1024		const CHUNK_SIZE: usize = size_of::<u16>();
1025		for i in 0..(N / CHUNK_SIZE) {
1026			sum = ones_complement_add(
1027				sum,
1028				(u16::from(addr[CHUNK_SIZE * i]) << 8) | u16::from(addr[CHUNK_SIZE * i + 1]),
1029			);
1030		}
1031		sum
1032	}
1033
1034	/// Calculates the checksum for the IPv4 pseudo-header as described in
1035	/// [RFC 9293 subsection 3.1](https://www.rfc-editor.org/rfc/rfc9293.html#section-3.1-6.18.1) WITHOUT the final inversion.
1036	pub(super) fn ipv4_pseudo_header_partial_checksum<T: AsRef<[u8]>>(
1037		packet: &Ipv4Packet<T>,
1038	) -> u16 {
1039		let padded_protocol = u16::from(u8::from(packet.next_header()));
1040		let payload_len = packet.total_len() - u16::from(packet.header_len());
1041
1042		let mut sum = addr_sum(&packet.src_addr().octets());
1043		sum = ones_complement_add(sum, addr_sum(&packet.dst_addr().octets()));
1044		sum = ones_complement_add(sum, padded_protocol);
1045		ones_complement_add(sum, payload_len)
1046	}
1047
1048	/// Calculates the checksum for the IPv6 pseudo-header as described in
1049	/// [RFC 8200 subsection 8.1](https://www.rfc-editor.org/rfc/rfc8200.html#section-8.1) WITHOUT the final inversion.
1050	pub(super) fn ipv6_pseudo_header_partial_checksum<T: AsRef<[u8]>>(
1051		packet: &Ipv6Packet<T>,
1052	) -> u16 {
1053		warn!("The IPv6 partial checksum implementation is untested!");
1054		let padded_protocol = u16::from(u8::from(packet.next_header()));
1055
1056		let mut sum = addr_sum(&packet.src_addr().octets());
1057		sum = ones_complement_add(sum, addr_sum(&packet.dst_addr().octets()));
1058		sum = ones_complement_add(sum, packet.payload_len());
1059		ones_complement_add(sum, padded_protocol)
1060	}
1061
1062	/// Implements one's complement checksum as described in [RFC 1071 section 1](https://www.rfc-editor.org/rfc/rfc1071#section-1).
1063	fn ones_complement_add(lhs: u16, rhs: u16) -> u16 {
1064		let (sum, overflow) = u16::overflowing_add(lhs, rhs);
1065		sum + u16::from(overflow)
1066	}
