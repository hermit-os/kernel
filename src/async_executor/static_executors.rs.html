13impl<RM, RRL> Executor<'static, RM, RRL>
14where
15    RM: RawMutex + Send + Sync + 'static,
16    RRL: RawRwLock + Send + Sync + 'static,
17{
18    /// Consumes the [`Executor`] and intentionally leaks it.
19    ///
20    /// Largely equivalent to calling `Box::leak(Box::new(executor))`, but the produced
21    /// [`StaticExecutor`]'s functions are optimized to require fewer synchronizing operations
22    /// when spawning, running, and finishing tasks.
23    ///
24    /// `StaticExecutor` cannot be converted back into a `Executor`, so this operation is
25    /// irreversible without the use of unsafe.
26    ///
27    /// # Example
28    ///
29    /// ```
30    /// use async_executor::Executor;
31    /// use futures_lite::future;
32    ///
33    /// let ex = Executor::new().leak();
34    ///
35    /// let task = ex.spawn(async {
36    ///     println!("Hello world");
37    /// });
38    ///
39    /// future::block_on(ex.run(task));
40    /// ```
41    pub fn leak(self) -> &'static StaticExecutor<RM, RRL> {
42        let ptr = self.state_ptr();
43        // SAFETY: So long as an Executor lives, it's state pointer will always be valid
44        // when accessed through state_ptr. This executor will live for the full 'static
45        // lifetime so this isn't an arbitrary lifetime extension.
46        let state: &'static State<RM, RRL> = unsafe { &*ptr };
47
48        core::mem::forget(self);
49
50        let mut active = state.active.lock();
51        if !active.is_empty() {
52            // Reschedule all of the active tasks.
53            for waker in active.drain() {
54                waker.wake();
55            }
56            // Overwrite to ensure that the slab is deallocated.
57            *active = Slab::new();
58        }
59
60        // SAFETY: StaticExecutor has the same memory layout as State as it's repr(transparent).
61        // The lifetime is not altered: 'static -> 'static.
62        let static_executor: &'static StaticExecutor<RM, RRL> =
63            unsafe { core::mem::transmute(state) };
64        static_executor
65    }
66}

123/// A static-lifetimed async [`Executor`].
124///
125/// This is primarily intended to be used in [`static`] variables, or types intended to be used, or can be created in non-static
126/// contexts via [`Executor::leak`].
127///
128/// Spawning, running, and finishing tasks are optimized with the assumption that the executor will never be `Drop`'ed.
129/// A static executor may require signficantly less overhead in both single-threaded and mulitthreaded use cases.
130///
131/// As this type does not implement `Drop`, losing the handle to the executor or failing
132/// to consistently drive the executor with [`StaticExecutor::tick`] or
133/// [`StaticExecutor::run`] will cause the all spawned tasks to permanently leak. Any
134/// tasks at the time will not be cancelled.
135///
136/// [`static`]: https://doc.rust-lang.org/std/keyword.static.html
137#[repr(transparent)]
138pub struct StaticExecutor<RM: RawMutex, RRL: RawRwLock> {
139    state: State<RM, RRL>,
140}
141
142// SAFETY: Executor stores no thread local state that can be accessed via other thread.
143unsafe impl<RM: RawMutex, RRL: RawRwLock> Send for StaticExecutor<RM, RRL> {}
144// SAFETY: Executor internally synchronizes all of it's operations internally.
145unsafe impl<RM: RawMutex, RRL: RawRwLock> Sync for StaticExecutor<RM, RRL> {}
146
147impl<RM: RawMutex, RRL: RawRwLock> UnwindSafe for StaticExecutor<RM, RRL> {}
148impl<RM: RawMutex, RRL: RawRwLock> RefUnwindSafe for StaticExecutor<RM, RRL> {}

150impl<RM: RawMutex, RRL: RawRwLock> fmt::Debug for StaticExecutor<RM, RRL> {
151    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
152        debug_state(&self.state, "StaticExecutor", f)
153    }
154}

161    /// Creates a new StaticExecutor.
162    ///
163    /// # Examples
164    ///
165    /// ```
166    /// use async_executor::StaticExecutor;
167    ///
168    /// static EXECUTOR: StaticExecutor = StaticExecutor::new();
169    /// ```
170    pub const fn new() -> Self {
171        Self {
172            state: State::new(),
173        }
174    }

176    /// Spawns a task onto the executor.
177    ///
178    /// Note: unlike [`Executor::spawn`], this function requires being called with a `'static`
179    /// borrow on the executor.
180    ///
181    /// # Examples
182    ///
183    /// ```
184    /// use async_executor::StaticExecutor;
185    ///
186    /// static EXECUTOR: StaticExecutor = StaticExecutor::new();
187    ///
188    /// let task = EXECUTOR.spawn(async {
189    ///     println!("Hello world");
190    /// });
191    /// ```
192    pub fn spawn<T: Send + 'static>(
193        &'static self,
194        future: impl Future<Output = T> + Send + 'static,
195    ) -> Task<T> {
196        let builder = Builder::new();
197        #[cfg(feature = "std")]
198        let builder = builder.propagate_panic(true);
199        let (runnable, task) = builder.spawn(|()| future, self.schedule());
200        runnable.schedule();
201        task
202    }
203
204    /// Spawns a non-`'static` task onto the executor.
205    ///
206    /// ## Safety
207    ///
208    /// The caller must ensure that the returned task terminates
209    /// or is cancelled before the end of 'a.
210    pub unsafe fn spawn_scoped<'a, T: Send + 'a>(
211        &'static self,
212        future: impl Future<Output = T> + Send + 'a,
213    ) -> Task<T> {
214        let builder = Builder::new();
215        #[cfg(feature = "std")]
216        let builder = builder.propagate_panic(true);
217        // SAFETY:
218        //
219        // - `future` is `Send`
220        // - `future` is not `'static`, but the caller guarantees that the
221        //    task, and thus its `Runnable` must not live longer than `'a`.
222        // - `self.schedule()` is `Send`, `Sync` and `'static`, as checked below.
223        //    Therefore we do not need to worry about what is done with the
224        //    `Waker`.
225        let (runnable, task) = unsafe { builder.spawn_unchecked(|()| future, self.schedule()) };
226        runnable.schedule();
227        task
228    }

230    /// Attempts to run a task if at least one is scheduled.
231    ///
232    /// Running a scheduled task means simply polling its future once.
233    ///
234    /// # Examples
235    ///
236    /// ```
237    /// use async_executor::StaticExecutor;
238    ///
239    /// static EXECUTOR: StaticExecutor = StaticExecutor::new();
240    ///
241    /// assert!(!EXECUTOR.try_tick()); // no tasks to run
242    ///
243    /// let task = EXECUTOR.spawn(async {
244    ///     println!("Hello world");
245    /// });
246    ///
247    /// assert!(EXECUTOR.try_tick()); // a task was found
248    /// ```
249    pub fn try_tick(&self) -> bool {
250        self.state.try_tick()
251    }

253    /// Runs a single task.
254    ///
255    /// Running a task means simply polling its future once.
256    ///
257    /// If no tasks are scheduled when this method is called, it will wait until one is scheduled.
258    ///
259    /// # Examples
260    ///
261    /// ```
262    /// use async_executor::StaticExecutor;
263    /// use futures_lite::future;
264    ///
265    /// static EXECUTOR: StaticExecutor = StaticExecutor::new();
266    ///
267    /// let task = EXECUTOR.spawn(async {
268    ///     println!("Hello world");
269    /// });
270    ///
271    /// future::block_on(EXECUTOR.tick()); // runs the task
272    /// ```
273    pub async fn tick(&self) {
274        self.state.tick().await;
275    }
276
277    /// Runs the executor until the given future completes.
278    ///
279    /// # Examples
280    ///
281    /// ```
282    /// use async_executor::StaticExecutor;
283    /// use futures_lite::future;
284    ///
285    /// static EXECUTOR: StaticExecutor = StaticExecutor::new();
286    ///
287    /// let task = EXECUTOR.spawn(async { 1 + 2 });
288    /// let res = future::block_on(EXECUTOR.run(async { task.await * 2 }));
289    ///
290    /// assert_eq!(res, 6);
291    /// ```
292    pub async fn run<T>(&self, future: impl Future<Output = T>) -> T {
293        self.state.run(future).await
294    }
295
296    /// Returns a function that schedules a runnable task when it gets woken up.
297    fn schedule(&'static self) -> impl Fn(Runnable) + Send + Sync + 'static {
298        let state: &'static State<RM, RRL> = &self.state;
299        // TODO: If possible, push into the current local queue and notify the ticker.
300        move |runnable| {
301            state.queue.push(runnable).unwrap();
302            state.notify();
303        }
304    }

369    /// Spawns a task onto the executor.
370    ///
371    /// Note: unlike [`LocalExecutor::spawn`], this function requires being called with a `'static`
372    /// borrow on the executor.
373    ///
374    /// # Examples
375    ///
376    /// ```
377    /// use async_executor::LocalExecutor;
378    ///
379    /// let ex = LocalExecutor::new().leak();
380    ///
381    /// let task = ex.spawn(async {
382    ///     println!("Hello world");
383    /// });
384    /// ```
385    pub fn spawn<T: 'static>(&'static self, future: impl Future<Output = T> + 'static) -> Task<T> {
386        #[cfg(feature = "std")]
387        let (runnable, task) = Builder::new()
388            .propagate_panic(true)
389            .spawn_local(|()| future, self.schedule());
390        // SAFETY:
391        //
392        // - `future` is not `Send` but `StaticLocalExecutor` is `!Sync`,
393        //   `try_tick`, `tick` and `run` can only be called from the origin
394        //    thread of the `StaticLocalExecutor`. Similarly, `spawn_scoped` can only
395        //    be called from the origin thread, ensuring that `future` and the executor
396        //    share the same origin thread. The `Runnable` can be scheduled from other
397        //    threads, but because of the above `Runnable` can only be called or
398        //    dropped on the origin thread.
399        // - `future` is not `'static`, but the caller guarantees that the
400        //    task, and thus its `Runnable` must not live longer than `'a`.
401        // - `self.schedule()` is `Send`, `Sync` and `'static`, as checked below.
402        //    Therefore we do not need to worry about what is done with the
403        //    `Waker`.
404        #[cfg(not(feature = "std"))]
405        let (runnable, task) =
406            unsafe { Builder::new().spawn_unchecked(|()| future, self.schedule()) };
407        runnable.schedule();
408        task
409    }
