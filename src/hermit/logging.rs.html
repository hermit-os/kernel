9/// Data structure to filter kernel messages
10pub struct KernelLogger {
11	time: AtomicBool,
12}
13
14impl KernelLogger {
15	pub const fn new() -> Self {
16		Self {
17			time: AtomicBool::new(false),
18		}
19	}
20
21	pub fn time(&self) -> bool {
22		self.time.load(Ordering::Relaxed)
23	}
24
25	pub fn set_time(&self, time: bool) {
26		self.time.store(time, Ordering::Relaxed);
27	}
28}
29
30impl log::Log for KernelLogger {
31	fn enabled(&self, _: &Metadata<'_>) -> bool {
32		true
33	}
34
35	fn flush(&self) {
36		// nothing to do
37	}
38
39	fn log(&self, record: &Record<'_>) {
40		if !self.enabled(record.metadata()) {
41			return;
42		}
43
44		// FIXME: Use `super let` once stable
45		let time;
46		let format_time = if self.time() {
47			time = Microseconds(crate::processor::get_timer_ticks());
48			format_args!("[{time}]")
49		} else {
50			format_args!("[            ]")
51		};
52		let core_id = crate::arch::core_local::core_id();
53		let level = ColorLevel(record.level());
54
55		let target = record.target();
56		let (crate_, modules) = target.split_once("::").unwrap_or((target, ""));
57		let (_modules, module) = modules.rsplit_once("::").unwrap_or(("", modules));
58		let target = if !module.is_empty() && crate_ == "hermit" {
59			module
60		} else {
61			crate_
62		};
63		let format_target = format_args!(" {target:<10}");
64
65		let args = record.args();
66		println!("{format_time}[{core_id}][{level}{format_target}] {args}");
67	}
68}

70struct Microseconds(u64);
71
72impl fmt::Display for Microseconds {
73	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
74		let seconds = self.0 / 1_000_000;
75		let microseconds = self.0 % 1_000_000;
76		write!(f, "{seconds:5}.{microseconds:06}")
77	}
78}

80struct ColorLevel(Level);
81
82impl fmt::Display for ColorLevel {
83	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
84		let level = self.0;
85
86		if no_color() {
87			write!(f, "{level:<5}")
88		} else {
89			let color = match level {
90				Level::Trace => AnsiColor::Magenta,
91				Level::Debug => AnsiColor::Blue,
92				Level::Info => AnsiColor::Green,
93				Level::Warn => AnsiColor::Yellow,
94				Level::Error => AnsiColor::Red,
95			};
96
97			let style = anstyle::Style::new().fg_color(Some(color.into()));
98			write!(f, "{style}{level:<5}{style:#}")
99		}
100	}
101}
102
103fn no_color() -> bool {
104	option_env!("NO_COLOR").is_some_and(|val| !val.is_empty())
105}

107pub unsafe fn init() {
108	log::set_logger(&KERNEL_LOGGER).expect("Can't initialize logger");
109	// Determines LevelFilter at compile time
110	let log_level: Option<&'static str> = option_env!("HERMIT_LOG_LEVEL_FILTER");
111	let mut max_level = LevelFilter::Info;
112
113	if let Some(log_level) = log_level {
114		max_level = if log_level.eq_ignore_ascii_case("off") {
115			LevelFilter::Off
116		} else if log_level.eq_ignore_ascii_case("error") {
117			LevelFilter::Error
118		} else if log_level.eq_ignore_ascii_case("warn") {
119			LevelFilter::Warn
120		} else if log_level.eq_ignore_ascii_case("info") {
121			LevelFilter::Info
122		} else if log_level.eq_ignore_ascii_case("debug") {
123			LevelFilter::Debug
124		} else if log_level.eq_ignore_ascii_case("trace") {
125			LevelFilter::Trace
126		} else {
127			error!("Could not parse HERMIT_LOG_LEVEL_FILTER, falling back to `info`.");
128			LevelFilter::Info
129		};
130	}
131
132	log::set_max_level(max_level);
133}
