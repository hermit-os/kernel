42/// Maps a given device specific pci configuration structure and
43/// returns a static reference to it.
44pub fn map_dev_cfg<T>(cap: &PciCap) -> Option<&'static mut T> {
45	if cap.cap.cfg_type != CapCfgType::Device {
46		error!("Capability of device config has wrong id. Mapping not possible...");
47		return None;
48	};
49
50	if cap.bar_len() < cap.len() + cap.offset() {
51		error!(
52			"Device config of device {:x}, does not fit into memory specified by bar!",
53			cap.dev_id(),
54		);
55		return None;
56	}
57
58	// Drivers MAY do this check. See Virtio specification v1.1. - 4.1.4.1
59	if cap.len() < u64::try_from(mem::size_of::<T>()).unwrap() {
60		error!(
61			"Device specific config from device {:x}, does not represent actual structure specified by the standard!",
62			cap.dev_id()
63		);
64		return None;
65	}
66
67	let virt_addr_raw = cap.bar_addr() + cap.offset();
68
69	// Create mutable reference to the PCI structure in PCI memory
70	let dev_cfg: &'static mut T =
71		unsafe { &mut *(ptr::with_exposed_provenance_mut(virt_addr_raw.try_into().unwrap())) };
72
73	Some(dev_cfg)
74}

76/// Virtio's PCI capabilities structure.
77/// See Virtio specification v.1.1 - 4.1.4
78///
79/// Indicating: Where the capability field is mapped in memory and
80/// Which id (sometimes also indicates priority for multiple
81/// capabilities of same type) it holds.
82///
83/// This structure does NOT represent the structure in the standard,
84/// as it is not directly mapped into address space from PCI device
85/// configuration space.
86/// Therefore the struct only contains necessary information to map
87/// corresponding config type into address space.
88#[derive(Clone)]
89pub struct PciCap {
90	bar: PciBar,
91	dev_id: u16,
92	cap: CapData,
93}
94
95impl PciCap {
96	pub fn offset(&self) -> u64 {
97		self.cap.offset.to_ne()
98	}
99
100	pub fn len(&self) -> u64 {
101		self.cap.length.to_ne()
102	}
103
104	pub fn bar_len(&self) -> u64 {
105		self.bar.length
106	}
107
108	pub fn bar_addr(&self) -> u64 {
109		self.bar.mem_addr
110	}
111
112	pub fn dev_id(&self) -> u16 {
113		self.dev_id
114	}
115
116	/// Returns a reference to the actual structure inside the PCI devices memory space.
117	fn map_common_cfg(&self) -> Option<VolatileRef<'static, CommonCfg>> {
118		if self.bar.length < self.len() + self.offset() {
119			let dev_id = self.dev_id;
120			let index = self.bar.index;
121			error!(
122				"Common config of the capability of device {dev_id:x} does not fit into memory specified by bar {index:x}!"
123			);
124			return None;
125		}
126
127		// `CommonCfg::queue_notify_data` and `CommonCfg::queue_reset` are optional.
128		const MIN_SIZE: usize = mem::size_of::<CommonCfg>() - mem::size_of::<[le16; 2]>();
129		if self.len() < u64::try_from(MIN_SIZE).unwrap() {
130			error!("Common config does not represent actual structure specified by the standard!");
131			return None;
132		}
133
134		let virt_addr_raw = self.bar.mem_addr + self.offset();
135		let ptr = NonNull::new(ptr::with_exposed_provenance_mut::<CommonCfg>(
136			virt_addr_raw.try_into().unwrap(),
137		))
138		.unwrap();
139
140		// Create mutable reference to the PCI structure in PCI memory
141		let com_cfg_raw = unsafe { VolatileRef::new(ptr) };
142
143		Some(com_cfg_raw)
144	}
145
146	fn map_isr_status(&self) -> Option<VolatileRef<'static, IsrStatusRaw>> {
147		if self.bar.length < self.len() + self.offset() {
148			let dev_id = self.dev_id;
149			let index = self.bar.index;
150			error!(
151				"ISR status config of device {dev_id:x}, does not fit into memory specified by bar {index:x}!"
152			);
153			return None;
154		}
155
156		let virt_addr_raw = self.bar.mem_addr + self.offset();
157		let ptr = NonNull::new(ptr::with_exposed_provenance_mut::<IsrStatusRaw>(
158			virt_addr_raw.try_into().unwrap(),
159		))
160		.unwrap();
161
162		// Create mutable reference to the PCI structure in the devices memory area
163		let isr_stat_raw = unsafe { VolatileRef::new(ptr) };
164
165		Some(isr_stat_raw)
166	}
167}

169/// Universal Caplist Collections holds all universal capability structures for
170/// a given Virtio PCI device.
171///
172/// As Virtio's PCI devices are allowed to present multiple capability
173/// structures of the same config type, the structure
174/// provides a driver with all capabilities, sorted in descending priority,
175/// allowing the driver to choose.
176/// The structure contains a special dev_cfg_list field, a vector holding
177/// [PciCap] objects, to allow the driver to map its
178/// device specific configurations independently.
179pub struct UniCapsColl {
180	pub(crate) com_cfg: ComCfg,
181	pub(crate) notif_cfg: NotifCfg,
182	pub(crate) isr_cfg: IsrStatus,
183	pub(crate) dev_cfg_list: Vec<PciCap>,
184}

203pub struct VqCfgHandler<'a> {
204	vq_index: u16,
205	raw: VolatileRef<'a, CommonCfg>,
206}
207
208impl VqCfgHandler<'_> {
209	// TODO: Create type for queue selected invariant to get rid of `self.select_queue()` everywhere.
210	fn select_queue(&mut self) {
211		self.raw
212			.as_mut_ptr()
213			.queue_select()
214			.write(self.vq_index.into());
215	}
216
217	/// Sets the size of a given virtqueue. In case the provided size exceeds the maximum allowed
218	/// size, the size is set to this maximum instead. Else size is set to the provided value.
219	///
220	/// Returns the set size in form of a `u16`.
221	pub fn set_vq_size(&mut self, max_size: u16) -> u16 {
222		self.select_queue();
223		let queue_size = self.raw.as_mut_ptr().queue_size();
224
225		let dev_queue_size = queue_size.read().to_ne();
226		if dev_queue_size >= max_size {
227			queue_size.write(max_size.into());
228			debug_assert_eq!(queue_size.read().to_ne(), max_size);
229			max_size
230		} else {
231			dev_queue_size
232		}
233	}
234
235	pub fn set_ring_addr(&mut self, addr: PhysAddr) {
236		self.select_queue();
237		self.raw
238			.as_mut_ptr()
239			.queue_desc()
240			.write(addr.as_u64().into());
241	}
242
243	pub fn set_drv_ctrl_addr(&mut self, addr: PhysAddr) {
244		self.select_queue();
245		self.raw
246			.as_mut_ptr()
247			.queue_driver()
248			.write(addr.as_u64().into());
249	}
250
251	pub fn set_dev_ctrl_addr(&mut self, addr: PhysAddr) {
252		self.select_queue();
253		self.raw
254			.as_mut_ptr()
255			.queue_device()
256			.write(addr.as_u64().into());
257	}
258
259	pub fn notif_off(&mut self) -> u16 {
260		self.select_queue();
261		self.raw.as_mut_ptr().queue_notify_off().read().to_ne()
262	}
263
264	pub fn enable_queue(&mut self) {
265		self.select_queue();
266		self.raw.as_mut_ptr().queue_enable().write(1.into());
267	}

367/// Notification Structure to handle virtqueue notification settings.
368/// See Virtio specification v1.1 - 4.1.4.4
369pub struct NotifCfg {
370	/// Start addr, from where the notification addresses for the virtqueues are computed
371	base_addr: u64,
372	notify_off_multiplier: u32,
373	/// defines the maximum size of the notification space, starting from base_addr.
374	#[cfg(debug_assertions)]
375	length: u64,
376}
377
378impl NotifCfg {
379	fn new(cap: &PciCap) -> Option<Self> {
380		if cap.bar.length < cap.len() + cap.offset() {
381			let dev_id = cap.dev_id;
382			let index = cap.bar.index;
383			error!(
384				"Notification config of device {dev_id:x}, does not fit into memory specified by bar {index:x}!"
385			);
386			return None;
387		}
388
389		let notify_off_multiplier = cap.cap.notify_off_multiplier?.to_ne();
390
391		// define base memory address from which the actual Queue Notify address can be derived via
392		// base_addr + queue_notify_off * notify_off_multiplier.
393		//
394		// Where queue_notify_off is taken from the respective common configuration struct.
395		// See Virtio specification v1.1. - 4.1.4.4
396		//
397		// Base address here already includes offset!
398		let base_addr = cap.bar.mem_addr + cap.offset();
399
400		Some(NotifCfg {
401			base_addr,
402			notify_off_multiplier,
403			#[cfg(debug_assertions)]
404			length: cap.len(),
405		})
406	}
407
408	pub fn notification_location(&self, vq_cfg_handler: &mut VqCfgHandler<'_>) -> *mut le32 {
409		let addend = u32::from(vq_cfg_handler.notif_off()) * self.notify_off_multiplier;
410
411		// TODO: This should be
412		// cap.length >= queue_notify_off * notify_off_multiplier + 4
413		// if VIRTIO_F_NOTIFICATION_DATA has been negotiated.
414		// Knowing this here requires a larger refactoring.
415		#[cfg(debug_assertions)]
416		assert!(self.length >= u64::from(addend + 2));
417
418		let addr = self.base_addr + u64::from(addend);
419		ptr::with_exposed_provenance_mut(addr.try_into().unwrap())
420	}
421}

423/// Control structure, allowing to notify a device via PCI bus.
424/// Typically hold by a virtqueue.
425pub struct NotifCtrl {
426	/// Indicates if VIRTIO_F_NOTIFICATION_DATA has been negotiated
427	f_notif_data: bool,
428	/// Where to write notification
429	notif_addr: *mut le32,
430}
431
432// FIXME: make `notif_addr` implement `Send` instead
433unsafe impl Send for NotifCtrl {}
434
435impl NotifCtrl {
436	/// Returns a new controller. By default MSI-X capabilities and VIRTIO_F_NOTIFICATION_DATA
437	/// are disabled.
438	pub fn new(notif_addr: *mut le32) -> Self {
439		NotifCtrl {
440			f_notif_data: false,
441			notif_addr,
442		}
443	}
444
445	/// Enables VIRTIO_F_NOTIFICATION_DATA. This changes which data is provided to the device. ONLY a good idea if Feature has been negotiated.
446	pub fn enable_notif_data(&mut self) {
447		self.f_notif_data = true;
448	}
449
450	pub fn notify_dev(&self, data: NotificationData) {
451		// See Virtio specification v.1.1. - 4.1.5.2
452		// Depending in the feature negotiation, we write either only the
453		// virtqueue index or the index and the next position inside the queue.
454
455		if self.f_notif_data {
456			unsafe {
457				self.notif_addr.write_volatile(data.into_bits());
458			}
459		} else {
460			unsafe {
461				self.notif_addr
462					.cast::<le16>()
463					.write_volatile(data.vqn().into());
464			}
465		}
466	}
467}

512/// Reads all PCI capabilities, starting at the capabilities list pointer from the
513/// PCI device.
514///
515/// Returns ONLY Virtio specific capabilities, which allow to locate the actual capability
516/// structures inside the memory areas, indicated by the BaseAddressRegisters (BAR's).
517fn read_caps(device: &PciDevice<PciConfigRegion>) -> Result<Vec<PciCap>, PciError> {
518	let device_id = device.device_id();
519
520	let capabilities = device
521		.capabilities()
522		.unwrap()
523		.filter_map(|capability| match capability {
524			PciCapability::Vendor(capability) => Some(capability),
525			_ => None,
526		})
527		.map(|addr| CapData::read(addr, device.access()).unwrap())
528		.filter(|cap| cap.cfg_type != CapCfgType::Pci)
529		.flat_map(|cap| {
530			let slot = cap.bar;
531			device
532				.memory_map_bar(slot, true)
533				.map(|(addr, size)| PciCap {
534					bar: VirtioPciBar::new(slot, addr.as_u64(), size.try_into().unwrap()),
535					dev_id: device_id,
536					cap,
537				})
538		})
539		.collect::<Vec<_>>();
540
541	if capabilities.is_empty() {
542		error!("No virtio capability found for device {device_id:x}");
543		return Err(PciError::NoVirtioCaps(device_id));
544	}
545
546	Ok(capabilities)
547}

549pub(crate) fn map_caps(device: &PciDevice<PciConfigRegion>) -> Result<UniCapsColl, VirtioError> {
550	let device_id = device.device_id();
551
552	// In case caplist pointer is not used, abort as it is essential
553	if !device.status().has_capability_list() {
554		error!("Found virtio device without capability list. Aborting!");
555		return Err(VirtioError::FromPci(PciError::NoCapPtr(device_id)));
556	}
557
558	// Get list of PciCaps pointing to capabilities
559	let cap_list = match read_caps(device) {
560		Ok(list) => list,
561		Err(pci_error) => return Err(VirtioError::FromPci(pci_error)),
562	};
563
564	let mut com_cfg = None;
565	let mut notif_cfg = None;
566	let mut isr_cfg = None;
567	let mut dev_cfg_list = Vec::new();
568	// Map Caps in virtual memory
569	for pci_cap in cap_list {
570		match pci_cap.cap.cfg_type {
571			CapCfgType::Common => {
572				if com_cfg.is_none() {
573					match pci_cap.map_common_cfg() {
574						Some(cap) => com_cfg = Some(ComCfg::new(cap)),
575						None => error!(
576							"Common config capability of device {device_id:x} could not be mapped!"
577						),
578					}
579				}
580			}
581			CapCfgType::Notify => {
582				if notif_cfg.is_none() {
583					match NotifCfg::new(&pci_cap) {
584						Some(notif) => notif_cfg = Some(notif),
585						None => error!(
586							"Notification config capability of device {device_id:x} could not be used!"
587						),
588					}
589				}
590			}
591			CapCfgType::Isr => {
592				if isr_cfg.is_none() {
593					match pci_cap.map_isr_status() {
594						Some(isr_stat) => isr_cfg = Some(IsrStatus::new(isr_stat)),
595						None => error!(
596							"ISR status config capability of device {device_id:x} could not be used!"
597						),
598					}
599				}
600			}
601			CapCfgType::SharedMemory => {
602				let cap_id = pci_cap.cap.id;
603				error!(
604					"Shared Memory config capability with id {cap_id} of device {device_id:x} could not be used!"
605				);
606			}
607			CapCfgType::Device => dev_cfg_list.push(pci_cap),
608
609			// PCI's configuration space is allowed to hold other structures, which are not virtio specific and are therefore ignored
610			// in the following
611			_ => continue,
612		}
613	}
614
615	Ok(UniCapsColl {
616		com_cfg: com_cfg.ok_or(VirtioError::NoComCfg(device_id))?,
617		notif_cfg: notif_cfg.ok_or(VirtioError::NoNotifCfg(device_id))?,
618		isr_cfg: isr_cfg.ok_or(VirtioError::NoIsrCfg(device_id))?,
619		dev_cfg_list,
620	})
621}

623/// Checks existing drivers for support of given device. Upon match, provides
624/// driver with a [`PciDevice<PciConfigRegion>`] reference, allowing access to the capabilities
625/// list of the given device through [map_caps].
626pub(crate) fn init_device(
627	device: &PciDevice<PciConfigRegion>,
628) -> Result<VirtioDriver, DriverError> {
629	let device_id = device.device_id();
630
631	if device_id < 0x1040 {
632		warn!(
633			"Legacy/transitional Virtio device, with id: {device_id:#x} is NOT supported, skipping!"
634		);
635
636		// Return Driver error inidacting device is not supported
637		return Err(DriverError::InitVirtioDevFail(
638			VirtioError::DevNotSupported(device_id),
639		));
640	}
641
642	let id = virtio::Id::from(u8::try_from(device_id - 0x1040).unwrap());
643
644	match id {
645		#[cfg(feature = "virtio-console")]
646		virtio::Id::Console => match VirtioConsoleDriver::init(device) {
647			Ok(virt_console_drv) => {
648				info!("Virtio console driver initialized.");
649
650				let irq = device.get_irq().unwrap();
651				crate::arch::interrupts::add_irq_name(irq, "virtio");
652				info!("Virtio interrupt handler at line {irq}");
653
654				Ok(VirtioDriver::Console(alloc::boxed::Box::new(
655					virt_console_drv,
656				)))
657			}
658			Err(virtio_error) => {
659				error!("Virtio console driver could not be initialized with device: {device_id:x}");
660				Err(DriverError::InitVirtioDevFail(virtio_error))
661			}
662		},
663		#[cfg(feature = "virtio-fs")]
664		virtio::Id::Fs => {
665			// TODO: check subclass
666			// TODO: proper error handling on driver creation fail
667			match VirtioFsDriver::init(device) {
668				Ok(virt_fs_drv) => {
669					info!("Virtio filesystem driver initialized.");
670					let irq = device.get_irq().unwrap();
671					crate::arch::interrupts::add_irq_name(irq, "virtio");
672					Ok(VirtioDriver::Fs(alloc::boxed::Box::new(virt_fs_drv)))
673				}
674				Err(virtio_error) => {
675					error!(
676						"Virtio filesystem driver could not be initialized with device: {device_id:x}"
677					);
678					Err(DriverError::InitVirtioDevFail(virtio_error))
679				}
680			}
681		}
682		#[cfg(all(
683			not(all(target_arch = "riscv64", feature = "gem-net", not(feature = "pci"))),
684			not(feature = "rtl8139"),
685			feature = "virtio-net",
686		))]
687		virtio::Id::Net => match VirtioNetDriver::init(device) {
688			Ok(virt_net_drv) => {
689				info!("Virtio network driver initialized.");
690
691				let irq = device.get_irq().unwrap();
692				crate::arch::interrupts::add_irq_name(irq, "virtio");
693				info!("Virtio interrupt handler at line {irq}");
694
695				Ok(VirtioDriver::Net(alloc::boxed::Box::new(virt_net_drv)))
696			}
697			Err(virtio_error) => {
698				error!(
699					"Virtio networkd driver could not be initialized with device: {device_id:x}"
700				);
701				Err(DriverError::InitVirtioDevFail(virtio_error))
702			}
703		},
704		#[cfg(feature = "virtio-vsock")]
705		virtio::Id::Vsock => match VirtioVsockDriver::init(device) {
706			Ok(virt_sock_drv) => {
707				info!("Virtio sock driver initialized.");
708
709				let irq = device.get_irq().unwrap();
710				crate::arch::interrupts::add_irq_name(irq, "virtio");
711				info!("Virtio interrupt handler at line {irq}");
712
713				Ok(VirtioDriver::Vsock(alloc::boxed::Box::new(virt_sock_drv)))
714			}
715			Err(virtio_error) => {
716				error!("Virtio sock driver could not be initialized with device: {device_id:x}");
717				Err(DriverError::InitVirtioDevFail(virtio_error))
718			}
719		},
720		id => {
721			if let Some(feature) = id.as_feature() {
722				error!("Virtio driver {id:?} is currently not active.");
723				error!("To use the device, recompile the kernel with the {feature} feature.");
724			} else {
725				error!("Virtio device {id:?} is not supported!");
726			}
727
728			// Return driver error indicating device is not supported.
729			Err(DriverError::InitVirtioDevFail(
730				VirtioError::DevNotSupported(device_id),
731			))
732		}
733	}
734}

736pub(crate) enum VirtioDriver {
737	#[cfg(feature = "virtio-console")]
738	Console(alloc::boxed::Box<VirtioConsoleDriver>),
739	#[cfg(feature = "virtio-fs")]
740	Fs(alloc::boxed::Box<VirtioFsDriver>),
741	#[cfg(all(
742		not(all(target_arch = "riscv64", feature = "gem-net", not(feature = "pci"))),
743		not(feature = "rtl8139"),
744		feature = "virtio-net",
745	))]
746	Net(alloc::boxed::Box<VirtioNetDriver>),
747	#[cfg(feature = "virtio-vsock")]
748	Vsock(alloc::boxed::Box<VirtioVsockDriver>),
749}
