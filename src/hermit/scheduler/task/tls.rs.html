29/// Thread-local storage data structures.
30pub struct Tls {
31	_block: Allocation,
32	thread_ptr: *mut (),
33}

35/// Thread control block.
36#[repr(C)]
37struct Tcb {
38	/// Thread pointer.
39	#[cfg(target_arch = "x86_64")]
40	thread_ptr: *mut (),
41
42	/// Pointer to the dynamic thread vector (dtv).
43	///
44	/// Currently not needed on Hermit.
45	dtv: *mut (),
46
47	/// Implementation-defined TCB data.
48	///
49	/// Currently not used on Hermit.
50	tcb_data: *mut (),
51}
52
53impl Tls {
54	unsafe fn new(tls_info: TlsInfo) -> Self {
55		let start = usize::try_from(tls_info.start).unwrap();
56		let filesz = usize::try_from(tls_info.filesz).unwrap();
57		let memsz = usize::try_from(tls_info.memsz).unwrap();
58		let align = usize::try_from(tls_info.align).unwrap();
59
60		// Get TLS initialization image
61		let tls_init_image = {
62			let start = ptr::with_exposed_provenance(start);
63			unsafe { slice::from_raw_parts(start, filesz) }
64		};
65
66		let tcb_layout = Layout::new::<Tcb>().pad_to_align();
67		let data_layout = Layout::from_size_align(memsz, align)
68			.unwrap()
69			.pad_to_align();
70
71		let (layout, tls_offset, tcb_offset) =
72			if cfg!(any(target_arch = "aarch64", target_arch = "riscv64")) {
73				// AArch64 and 64-bit RISC-V use TLS data structures variant I.
74
75				// Variant I does not guarantee more than 16 bytes of space for the TCB.
76				assert_eq!(tcb_layout.size(), 16);
77
78				// Variant I requires the dtv pointer to be at the start of the TCB.
79				assert_eq!(mem::offset_of!(Tcb, dtv), 0);
80
81				// In variant I, the TLS data comes after the TCB.
82				let (tls_layout, data_offset) = tcb_layout.extend(data_layout).unwrap();
83				(tls_layout.pad_to_align(), data_offset, 0)
84			} else if cfg!(target_arch = "x86_64") {
85				// x86-64 uses TLS data structures variant II.
86
87				// Variant II (on GNU systems) requires the thread pointer to be at the start of the TCB:
88				// > For the implementation on GNU systems we can add one more requirement. The
89				// > address %gs:0 represents is actually the same as the thread pointer. I.e., the content of
90				// > the word addressed via %gs:0 is the address of the very same location.
91				#[cfg(target_arch = "x86_64")]
92				assert_eq!(mem::offset_of!(Tcb, thread_ptr), 0);
93
94				// In Variant II, the TCB comes after the TLS data.
95				let (tls_layout, tcb_offset) = data_layout.extend(tcb_layout).unwrap();
96				(tls_layout.pad_to_align(), 0, tcb_offset)
97			} else {
98				unimplemented!()
99			};
100
101		let mut block = Allocation::new(layout).unwrap();
102
103		// Initialize the beginning of the TLS block with the TLS initialization image.
104		block.as_mut_slice()[tls_offset..][..tls_init_image.len()].copy_from_slice(tls_init_image);
105
106		// Fill the rest of the TLS block with zeros.
107		block.as_mut_slice()[tls_offset..][tls_init_image.len()..data_layout.size()]
108			.fill(MaybeUninit::new(0));
109
110		let thread_ptr = if cfg!(target_arch = "riscv64") {
111			// On RISC-V, `tp` points to the address one past the end of the TCB.
112			unsafe { block.as_mut_ptr().add(tls_offset).cast() }
113		} else if cfg!(target_arch = "aarch64") {
114			// For variant I, `tp` points to the start of the block.
115			block.as_mut_ptr().cast()
116		} else if cfg!(target_arch = "x86_64") {
117			// For variant II, `tp` points to the TCB after the TLS data.
118			unsafe { block.as_mut_ptr().add(tcb_offset).cast() }
119		} else {
120			unimplemented!()
121		};
122
123		let tcb_ptr = unsafe { block.as_mut_ptr().add(tcb_offset).cast::<Tcb>() };
124		let tcb = Tcb {
125			#[cfg(target_arch = "x86_64")]
126			thread_ptr,
127			dtv: ptr::null_mut(),
128			tcb_data: ptr::null_mut(),
129		};
130		unsafe {
131			tcb_ptr.write(tcb);
132		}
133
134		Self {
135			_block: block,
136			thread_ptr,
137		}
138	}
139
140	pub fn from_env() -> Option<Self> {
141		let tls_info = crate::env::boot_info().load_info.tls_info?;
142		let this = unsafe { Self::new(tls_info) };
143		Some(this)
144	}
145
146	pub fn thread_ptr(&self) -> *mut () {
147		self.thread_ptr
148	}
149
150	/// Sets the thread pointer register to this TLS's thread pointer value.
151	///
152	/// This should only be used for the idle task. Since the idle task is
153	/// already running, we don't create a new stack frame that we would put
154	/// the thread pointer value into otherwise.
155	///
156	/// The idle task does not enter userspace. That's why it does not need a
157	/// TLS most of the time. In special situations such as instrumenting the
158	/// kernel, the tracer or profiler or `mcount` implementation such as
159	/// rftrace might use TLS for differentiating between the idle task and
160	/// other tasks.
161	pub fn set_thread_ptr(&self) {
162		cfg_if::cfg_if! {
163			if #[cfg(target_arch = "aarch64")] {
164				use aarch64_cpu::registers::{TPIDR_EL0, Writeable};
165
166				let addr = self.thread_ptr().expose_provenance();
167				TPIDR_EL0.set(addr.try_into().unwrap());
168			} else if #[cfg(target_arch = "riscv64")] {
169				unsafe {
170					core::arch::asm!(
171						"mv tp, {}",
172						in(reg) self.thread_ptr().expose_provenance(),
173						options(nomem, nostack, preserves_flags),
174					);
175				}
176			} else if #[cfg(target_arch = "x86_64")] {
177				use crate::arch::x86_64::kernel::processor;
178
179				let addr = self.thread_ptr().expose_provenance();
180				processor::writefs(addr);
181			}
182		}
183	}
184}

186mod allocation {
187	use core::alloc::Layout;
188	use core::mem::MaybeUninit;
189	use core::slice;
190
191	pub struct Allocation {
192		ptr: *mut u8,
193		layout: Layout,
194	}
195
196	impl Allocation {
197		pub fn new(layout: Layout) -> Option<Self> {
198			let ptr = unsafe { ::alloc::alloc::alloc(layout) };
199
200			if ptr.is_null() {
201				return None;
202			}
203
204			Some(Self { ptr, layout })
205		}
206
207		pub fn as_mut_ptr(&mut self) -> *mut u8 {
208			self.ptr
209		}
210
211		pub fn as_mut_slice(&mut self) -> &mut [MaybeUninit<u8>] {
212			unsafe { slice::from_raw_parts_mut(self.ptr.cast(), self.layout.size()) }
213		}
214	}
215
216	impl Drop for Allocation {
217		fn drop(&mut self) {
218			unsafe {
219				::alloc::alloc::dealloc(self.ptr, self.layout);
220			}
221		}
222	}
