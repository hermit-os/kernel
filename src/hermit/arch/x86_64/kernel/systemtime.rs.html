11const CMOS_COMMAND: Port<u8> = Port::new(0x70);
12const CMOS_DATA: Port<u8> = Port::new(0x71);
13
14const CMOS_DISABLE_NMI: u8 = 1 << 7;
15
16const CMOS_SECOND_REGISTER: u8 = 0x00;
17const CMOS_MINUTE_REGISTER: u8 = 0x02;
18const CMOS_HOUR_REGISTER: u8 = 0x04;
19const CMOS_DAY_REGISTER: u8 = 0x07;
20const CMOS_MONTH_REGISTER: u8 = 0x08;
21const CMOS_YEAR_REGISTER: u8 = 0x09;
22const CMOS_STATUS_REGISTER_A: u8 = 0x0a;
23const CMOS_STATUS_REGISTER_B: u8 = 0x0b;
24
25const CMOS_UPDATE_IN_PROGRESS_FLAG: u8 = 1 << 7;
26const CMOS_24_HOUR_FORMAT_FLAG: u8 = 1 << 1;
27const CMOS_BINARY_FORMAT_FLAG: u8 = 1 << 2;
28const CMOS_12_HOUR_PM_FLAG: u8 = 0x80;
29
30struct Rtc {
31	cmos_format: u8,
32}
33
34impl Rtc {
35	fn new() -> Self {
36		Self {
37			cmos_format: Self::read_cmos_register(CMOS_STATUS_REGISTER_B),
38		}
39	}
40
41	const fn is_24_hour_format(&self) -> bool {
42		self.cmos_format & CMOS_24_HOUR_FORMAT_FLAG > 0
43	}
44
45	const fn is_binary_format(&self) -> bool {
46		self.cmos_format & CMOS_BINARY_FORMAT_FLAG > 0
47	}
48
49	const fn time_is_pm(hour: u8) -> bool {
50		hour & CMOS_12_HOUR_PM_FLAG > 0
51	}
52
53	/// Returns the binary value for a given value in BCD (Binary-Coded Decimal).
54	const fn convert_bcd_value(value: u8) -> u8 {
55		((value / 16) * 10) + (value & 0xf)
56	}
57
58	/// Returns the number of microseconds since the epoch from a given date.
59	/// Inspired by Linux Kernel's mktime64(), see kernel/time/time.c.
60	fn microseconds_from_date(
61		year: u16,
62		month: u8,
63		day: u8,
64		hour: u8,
65		minute: u8,
66		second: u8,
67	) -> u64 {
68		let (m, y) = if month > 2 {
69			(u64::from(month - 2), u64::from(year))
70		} else {
71			(u64::from(month + 12 - 2), u64::from(year - 1))
72		};
73
74		let days_since_epoch =
75			(y / 4 - y / 100 + y / 400 + 367 * m / 12 + u64::from(day)) + y * 365 - 719_499;
76		let hours_since_epoch = days_since_epoch * 24 + u64::from(hour);
77		let minutes_since_epoch = hours_since_epoch * 60 + u64::from(minute);
78		let seconds_since_epoch = minutes_since_epoch * 60 + u64::from(second);
79
80		seconds_since_epoch * 1_000_000u64
81	}
82
83	fn read_cmos_register(register: u8) -> u8 {
84		let mut cmos_command = CMOS_COMMAND;
85		let mut cmos_data = CMOS_DATA;
86		unsafe {
87			cmos_command.write(CMOS_DISABLE_NMI | register);
88			cmos_data.read()
89		}
90	}
91
92	fn read_datetime_register(&self, register: u8) -> u8 {
93		let value = Self::read_cmos_register(register);
94
95		// Every date/time register may either be in binary or in BCD format.
96		// Convert BCD values if necessary.
97		if self.is_binary_format() {
98			value
99		} else {
100			Self::convert_bcd_value(value)
101		}
102	}
103
104	fn read_all_values(&self) -> u64 {
105		// Reading year, month, and day is straightforward.
106		let year = u16::from(self.read_datetime_register(CMOS_YEAR_REGISTER)) + 2000;
107		let month = self.read_datetime_register(CMOS_MONTH_REGISTER);
108		let day = self.read_datetime_register(CMOS_DAY_REGISTER);
109
110		// The hour register is a bitch.
111		// On top of being in either binary or BCD format, it may also be in 12-hour
112		// or 24-hour format.
113		let mut hour = Self::read_cmos_register(CMOS_HOUR_REGISTER);
114		let mut is_pm = false;
115
116		// Check and mask off a potential PM flag if the hour is given in 12-hour format.
117		if !self.is_24_hour_format() {
118			is_pm = Self::time_is_pm(hour);
119			hour &= !CMOS_12_HOUR_PM_FLAG;
120		}
121
122		// Now convert a BCD number to binary if necessary (after potentially masking off the PM flag above).
123		if !self.is_binary_format() {
124			hour = Self::convert_bcd_value(hour);
125		}
126
127		// If the hour is given in 12-hour format, do the necessary calculations to convert it into 24 hours.
128		if !self.is_24_hour_format() {
129			if hour == 12 {
130				// 12:00 AM is 00:00 and 12:00 PM is 12:00 (see is_pm below) in 24-hour format.
131				hour = 0;
132			}
133
134			if is_pm {
135				// {01:00 PM, 02:00 PM, ...} is {13:00, 14:00, ...} in 24-hour format.
136				hour += 12;
137			}
138		}
139
140		// The minute and second registers are straightforward again.
141		let minute = self.read_datetime_register(CMOS_MINUTE_REGISTER);
142		let second = self.read_datetime_register(CMOS_SECOND_REGISTER);
143
144		// Convert it all to microseconds and return the result.
145		Self::microseconds_from_date(year, month, day, hour, minute, second)
146	}
147
148	pub fn get_microseconds_since_epoch(&self) -> u64 {
149		loop {
150			// If a clock update is currently in progress, wait until it is finished.
151			while Self::read_cmos_register(CMOS_STATUS_REGISTER_A) & CMOS_UPDATE_IN_PROGRESS_FLAG
152				> 0
153			{
154				spin_loop();
155			}
156
157			// Get the current time in microseconds since the epoch.
158			let microseconds_since_epoch_1 = self.read_all_values();
159
160			// If the clock is already updating the time again, the read values may be inconsistent
161			// and we have to repeat this process.
162			if Self::read_cmos_register(CMOS_STATUS_REGISTER_A) & CMOS_UPDATE_IN_PROGRESS_FLAG > 0 {
163				continue;
164			}
165
166			// Get the current time again and verify that it's the same we last read.
167			let microseconds_since_epoch_2 = self.read_all_values();
168			if microseconds_since_epoch_1 == microseconds_since_epoch_2 {
169				// Both times are identical, so we have read consistent values and can exit the loop.
170				return microseconds_since_epoch_1;
171			}
172		}
173	}
174}

176static BOOT_TIME: OnceCell<u64> = OnceCell::new();
177
178pub fn init() {
179	let boot_time = match env::boot_info().platform_info {
180		PlatformInfo::Uhyve { boot_time, .. } => boot_time,
181		_ => {
182			// Get the current time in microseconds since the epoch (1970-01-01) from the x86 RTC.
183			// Subtract the timer ticks to get the actual time when Hermit was booted.
184			let current_time = without_interrupts(|| Rtc::new().get_microseconds_since_epoch());
185			let boot_time = current_time - processor::get_timer_ticks();
186			OffsetDateTime::from_unix_timestamp_nanos(i128::from(boot_time) * 1000).unwrap()
187		}
188	};
189	info!("Hermit booted on {boot_time}");
190
191	let micros = u64::try_from(boot_time.unix_timestamp_nanos() / 1000).unwrap();
192	BOOT_TIME.set(micros).unwrap();
193}

195/// Returns the current time in microseconds since UNIX epoch.
196pub fn now_micros() -> u64 {
197	*BOOT_TIME.get().unwrap() + super::processor::get_timer_ticks()
