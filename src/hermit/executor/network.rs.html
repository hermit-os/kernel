34pub(crate) enum NetworkState<'a> {
35	Missing,
36	// Never constructed if the kernel is configured for the loopback driver.

80fn start_endpoint() -> u16 {
81	((unsafe { core::arch::x86_64::_rdtsc() }) % u64::from(u16::MAX))
82		.try_into()
83		.unwrap()

86#[cfg(target_arch = "aarch64")]
87fn start_endpoint() -> u16 {
88	use core::arch::asm;
89	let value: u64;
90
91	unsafe {
92		asm!(
93			"mrs {value}, cntpct_el0",
94			value = out(reg) value,
95			options(nostack),
96		);
97	}
98
99	(value % u64::from(u16::MAX)).try_into().unwrap()
100}

103fn start_endpoint() -> u16 {

105		.try_into()

114#[cfg(feature = "dhcpv4")]
115async fn dhcpv4_run() {
116	future::poll_fn(|cx| {
117		let Some(mut guard) = NIC.try_lock() else {
118			// FIXME: only wake when progress can be made
119			cx.waker().wake_by_ref();
120			return Poll::Pending;
121		};
122
123		let nic = guard.as_nic_mut().unwrap();
124		let dhcp_handle = nic.dhcp_handle;
125		let socket = nic.sockets.get_mut::<dhcpv4::Socket<'_>>(dhcp_handle);
126
127		socket.register_waker(cx.waker());
128
129		match socket.poll() {
130			None => {}
131			Some(dhcpv4::Event::Configured(config)) => {
132				info!("DHCP config acquired!");
133				info!("IP address:   {}", config.address);
134				nic.iface.update_ip_addrs(|addrs| {
135					if let Some(dest) = addrs.iter_mut().next() {
136						*dest = IpCidr::Ipv4(config.address);
137					} else if addrs.push(IpCidr::Ipv4(config.address)).is_err() {
138						info!("Unable to update IP address");
139					}
140				});
141				if let Some(router) = config.router {
142					info!("Gateway:      {router}");
143					nic.iface
144						.routes_mut()
145						.add_default_ipv4_route(router)
146						.unwrap();
147				} else {
148					info!("Gateway:      None");
149					nic.iface.routes_mut().remove_default_ipv4_route();
150				}
151
152				#[cfg(feature = "dns")]
153				let mut dns_servers: Vec<IpAddress> = Vec::new();
154				for (i, s) in config.dns_servers.iter().enumerate() {
155					info!("DNS server {i}: {s}");
156					#[cfg(feature = "dns")]
157					dns_servers.push(IpAddress::Ipv4(*s));
158				}
159
160				#[cfg(feature = "dns")]
161				if !dns_servers.is_empty() {
162					let dns_socket = dns::Socket::new(dns_servers.as_slice(), vec![]);
163					nic.dns_handle = Some(nic.sockets.add(dns_socket));
164				}
165			}
166			Some(dhcpv4::Event::Deconfigured) => {
167				info!("DHCP lost config!");
168				let cidr = Ipv4Cidr::new(Ipv4Address::UNSPECIFIED, 0);
169				nic.iface.update_ip_addrs(|addrs| {
170					if let Some(dest) = addrs.iter_mut().next() {
171						*dest = IpCidr::Ipv4(cidr);
172					}
173				});
174				nic.iface.routes_mut().remove_default_ipv4_route();
175
176				#[cfg(feature = "dns")]
177				{
178					if let Some(dns_handle) = nic.dns_handle {
179						nic.sockets.remove(dns_handle);
180					}
181
182					nic.dns_handle = None;
183				}
184			}
185		};
186
187		Poll::<()>::Pending
188	})
189	.await;
190}

192async fn network_run() {
193	future::poll_fn(|cx| {
194		let Some(mut guard) = NIC.try_lock() else {
195			// FIXME: only wake when progress can be made
196			cx.waker().wake_by_ref();
197			// another task is already using the NIC => don't check
198			return Poll::Pending;
199		};
200
201		let NetworkState::Initialized(nic) = &mut *guard else {
202			return Poll::Ready(());
203		};
204
205		nic.poll_common(now());
206		// FIXME: only wake when progress can be made
207		cx.waker().wake_by_ref();
208		Poll::Pending
209	})
210	.await;
211}

213#[cfg(feature = "dns")]
214pub(crate) async fn get_query_result(query: QueryHandle) -> io::Result<Vec<IpAddress>> {
215	future::poll_fn(|cx| {
216		let Some(mut guard) = NIC.try_lock() else {
217			// FIXME: only wake when progress can be made
218			cx.waker().wake_by_ref();
219			return Poll::Pending;
220		};
221
222		let nic = guard.as_nic_mut().unwrap();
223		let socket = nic.get_mut_dns_socket()?;
224		match socket.get_query_result(query) {
225			Ok(addrs) => {
226				let mut ips = Vec::new();
227				for x in &addrs {
228					ips.push(*x);
229				}
230
231				Poll::Ready(Ok(ips))
232			}
233			Err(GetQueryResultError::Pending) => {
234				socket.register_query_waker(query, cx.waker());
235				Poll::Pending
236			}
237			Err(e) => {
238				warn!("DNS query failed: {e:?}");
239				Poll::Ready(Err(Errno::Noent))
240			}
241		}
242	})
243	.await
244}

246pub(crate) fn init() {
247	info!("Try to initialize network!");
248
249	// initialize variable, which contains the next local endpoint
250	LOCAL_ENDPOINT.store(start_endpoint(), Ordering::Relaxed);
251
252	let mut guard = NIC.lock();
253
254	*guard = NetworkInterface::create();
255
256	let NetworkState::Initialized(nic) = &mut *guard else {
257		return;
258	};
259
260	let time = now();
261	nic.poll_common(time);
262	let wakeup_time = nic
263		.poll_delay(time)
264		.map(|d| crate::arch::processor::get_timer_ticks() + d.total_micros());
265	crate::core_scheduler().add_network_timer(wakeup_time);
266
267	spawn(network_run());
268	#[cfg(feature = "dhcpv4")]
269	spawn(dhcpv4_run());
270}

272impl<'a> NetworkInterface<'a> {
273	#[cfg(feature = "udp")]
274	pub(crate) fn create_udp_handle(&mut self) -> Result<Handle, ()> {
275		let udp_rx_buffer =
276			udp::PacketBuffer::new(vec![udp::PacketMetadata::EMPTY; 4], vec![0; 0x10000]);
277		let udp_tx_buffer =
278			udp::PacketBuffer::new(vec![udp::PacketMetadata::EMPTY; 4], vec![0; 0x10000]);
279		let udp_socket = udp::Socket::new(udp_rx_buffer, udp_tx_buffer);
280		let udp_handle = self.sockets.add(udp_socket);
281
282		Ok(udp_handle)
283	}
284
285	#[cfg(feature = "tcp")]
286	pub(crate) fn create_tcp_handle(&mut self) -> Result<Handle, ()> {
287		let tcp_rx_buffer = tcp::SocketBuffer::new(vec![0; 0x10000]);
288		let tcp_tx_buffer = tcp::SocketBuffer::new(vec![0; 0x10000]);
289		let mut tcp_socket = tcp::Socket::new(tcp_rx_buffer, tcp_tx_buffer);
290		tcp_socket.set_nagle_enabled(true);
291		let tcp_handle = self.sockets.add(tcp_socket);
292
293		Ok(tcp_handle)
294	}

322	#[cfg(feature = "dns")]
323	pub(crate) fn start_query(
324		&mut self,
325		name: &str,
326		query_type: DnsQueryType,
327	) -> io::Result<QueryHandle> {
328		let dns_handle = self.dns_handle.ok_or(Errno::Inval)?;
329		let socket: &mut dns::Socket<'a> = self.sockets.get_mut(dns_handle);
330		socket
331			.start_query(self.iface.context(), name, query_type)
332			.map_err(|_| Errno::Io)
333	}

335	#[cfg(feature = "dns")]

337		let dns_handle = self.dns_handle.ok_or(Errno::Inval)?;
338		Ok(self.sockets.get_mut(dns_handle))
339	}
