29fn fill_queue(vq: &mut VirtQueue, num_packets: u16, packet_size: u32) {
30	for _ in 0..num_packets {
31		let buff_tkn = match AvailBufferToken::new(
32			SmallVec::new(),
33			SmallVec::from_buf([
34				BufferElem::Sized(Box::<Hdr, _>::new_uninit_in(DeviceAlloc)),
35				BufferElem::Vector(Vec::with_capacity_in(
36					packet_size.try_into().unwrap(),
37					DeviceAlloc,
38				)),
39			]),
40		) {
41			Ok(tkn) => tkn,
42			Err(_vq_err) => {
43				error!("Setup of network queue failed, which should not happen!");
44				panic!("setup of network queue failed!");
45			}
46		};
47
48		// BufferTokens are directly provided to the queue
49		// TransferTokens are directly dispatched
50		// Transfers will be awaited at the queue
51		if let Err(err) = vq.dispatch(buff_tkn, false, BufferType::Direct) {
52			error!("{err:#?}");
53			break;
54		}
55	}
56}

72	pub fn add(&mut self, mut vq: VirtQueue) {
73		const BUFF_PER_PACKET: u16 = 2;
74		let num_packets: u16 = u16::from(vq.size()) / BUFF_PER_PACKET;
75		info!("num_packets {num_packets}");
76		fill_queue(&mut vq, num_packets, self.packet_size);
77
78		self.vq = Some(vq);
79	}

84		}

88		if let Some(ref mut vq) = self.vq {
89			vq.disable_notifs();
90		}
91	}
92
93	fn get_next(&mut self) -> Option<UsedBufferToken> {
94		self.vq.as_mut().unwrap().try_recv().ok()
95	}
96
97	pub fn process_packet<F>(&mut self, mut f: F)
98	where
99		F: FnMut(&Hdr, &[u8]),
100	{
101		while let Some(mut buffer_tkn) = self.get_next() {
102			let header = buffer_tkn
103				.used_recv_buff
104				.pop_front_downcast::<Hdr>()
105				.unwrap();
106			let packet = buffer_tkn.used_recv_buff.pop_front_vec().unwrap();
107
108			if let Some(ref mut vq) = self.vq {
109				f(&header, &packet[..]);
110
111				fill_queue(vq, 1, self.packet_size);
112			} else {
113				panic!("Invalid length of receive queue");
114			}
115		}
116	}
117}

156	/// Provides a slice to copy the packet and transfer the packet
157	/// to the send queue. The caller has to create the header
158	/// for the vsock interface.
159	pub fn send_packet<R, F>(&mut self, len: usize, f: F) -> R
160	where
161		F: FnOnce(&mut [u8]) -> R,
162	{
163		// We need to poll to get the queue to remove elements from the table and make space for
164		// what we are about to add
165		self.poll();
166		if let Some(ref mut vq) = self.vq {
167			assert!(len < usize::try_from(self.packet_length).unwrap());
168			let mut packet = Vec::with_capacity_in(len, DeviceAlloc);
169			let result = unsafe {
170				let result = f(packet.spare_capacity_mut().assume_init_mut());
171				packet.set_len(len);
172				result
173			};
174
175			let buff_tkn = AvailBufferToken::new(
176				{
177					let mut vec = SmallVec::new();
178					vec.push(BufferElem::Vector(packet));
179					vec
180				},
181				SmallVec::new(),
182			)
183			.unwrap();
184
185			vq.dispatch(buff_tkn, false, BufferType::Direct).unwrap();
186
187			result
188		} else {
189			panic!("Unable to get send queue");
190		}
191	}
192}

307	/// Negotiates a subset of features, understood and wanted by both the OS
308	/// and the device.
309	fn negotiate_features(
310		&mut self,
311		driver_features: virtio::vsock::F,
312	) -> Result<(), VirtioVsockError> {
313		let device_features = virtio::vsock::F::from(self.com_cfg.dev_features());
314
315		if device_features.requirements_satisfied() {
316			info!("Feature set wanted by vsock driver are in conformance with specification.");
317		} else {
318			return Err(VirtioVsockError::FeatureRequirementsNotMet(device_features));
319		}
320
321		if device_features.contains(driver_features) {
322			// If device supports subset of features write feature set to common config
323			self.com_cfg.set_drv_features(driver_features.into());
324			Ok(())
325		} else {
326			Err(VirtioVsockError::IncompatibleFeatureSets(
327				driver_features,
328				device_features,
329			))
330		}
331	}
332
333	/// Initializes the device in adherence to specification. Returns Some(VirtioVsockError)
334	/// upon failure and None in case everything worked as expected.
335	///
336	/// See Virtio specification v1.1. - 3.1.1.
337	///                      and v1.1. - 5.10.6
338	pub fn init_dev(&mut self) -> Result<(), VirtioVsockError> {
339		// Reset
340		self.com_cfg.reset_dev();
341
342		// Indicate device, that OS noticed it
343		self.com_cfg.ack_dev();
344
345		// Indicate device, that driver is able to handle it
346		self.com_cfg.set_drv();
347
348		let features = virtio::vsock::F::VERSION_1;
349		self.negotiate_features(features)?;
350
351		// Indicates the device, that the current feature set is final for the driver
352		// and will not be changed.
353		self.com_cfg.features_ok();
354
355		// Checks if the device has accepted final set. This finishes feature negotiation.
356		if self.com_cfg.check_features() {
357			info!(
358				"Features have been negotiated between virtio socket device {:x} and driver.",
359				self.dev_cfg.dev_id
360			);
361			// Set feature set in device config fur future use.
362			self.dev_cfg.features = features;
363		} else {
364			return Err(VirtioVsockError::FailFeatureNeg(self.dev_cfg.dev_id));
365		}
366
367		// create the queues and tell device about them
368		self.recv_vq.add(VirtQueue::Split(
369			SplitVq::new(
370				&mut self.com_cfg,
371				&self.notif_cfg,
372				VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
373				VqIndex::from(0u16),
374				self.dev_cfg.features.into(),
375			)
376			.unwrap(),
377		));
378		// Interrupt for receiving packets is wanted
379		self.recv_vq.enable_notifs();
380
381		self.send_vq.add(VirtQueue::Split(
382			SplitVq::new(
383				&mut self.com_cfg,
384				&self.notif_cfg,
385				VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
386				VqIndex::from(1u16),
387				self.dev_cfg.features.into(),
388			)
389			.unwrap(),
390		));
391		// Interrupt for communicating that a sended packet left, is not needed
392		self.send_vq.disable_notifs();
393
394		// create the queues and tell device about them
395		self.event_vq.add(VirtQueue::Split(
396			SplitVq::new(
397				&mut self.com_cfg,
398				&self.notif_cfg,
399				VqSize::from(VIRTIO_MAX_QUEUE_SIZE),
400				VqIndex::from(2u16),
401				self.dev_cfg.features.into(),
402			)
403			.unwrap(),
404		));
405		// Interrupt for event packets is wanted
406		self.event_vq.enable_notifs();
407
408		// At this point the device is "live"
409		self.com_cfg.drv_ok();
410
411		Ok(())
412	}
