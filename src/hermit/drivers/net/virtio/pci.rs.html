17// Backend-dependent interface for Virtio network driver
18impl VirtioNetDriver<Uninit> {
19	fn map_cfg(cap: &PciCap) -> Option<NetDevCfg> {
20		let dev_cfg = pci::map_dev_cfg::<virtio::net::Config>(cap)?;
21
22		let dev_cfg = VolatileRef::from_ref(dev_cfg);
23
24		Some(NetDevCfg {
25			raw: dev_cfg,
26			dev_id: cap.dev_id(),
27			features: virtio::net::F::empty(),
28		})
29	}
30
31	/// Instantiates a new (VirtioNetDriver)[VirtioNetDriver] struct, by checking the available
32	/// configuration structures and moving them into the struct.
33	pub(crate) fn new(
34		caps_coll: UniCapsColl,
35		device: &PciDevice<PciConfigRegion>,
36	) -> Result<Self, error::VirtioNetError> {
37		let device_id = device.device_id();
38		let UniCapsColl {
39			com_cfg,
40			notif_cfg,
41			isr_cfg,
42			dev_cfg_list,
43			..
44		} = caps_coll;
45
46		let Some(dev_cfg) = dev_cfg_list.iter().find_map(VirtioNetDriver::map_cfg) else {
47			error!("No dev config. Aborting!");
48			return Err(error::VirtioNetError::NoDevCfg(device_id));
49		};
50
51		Ok(VirtioNetDriver {
52			dev_cfg,
53			com_cfg,
54			isr_stat: isr_cfg,
55			notif_cfg,
56			inner: Uninit,
57			num_vqs: 0,
58			irq: device.get_irq().unwrap(),
59			checksums: ChecksumCapabilities::default(),
60		})
61	}
62
63	/// Initializes virtio network device by mapping configuration layout to
64	/// respective structs (configuration structs are:
65	/// [ComCfg](structs.comcfg.html), [NotifCfg](structs.notifcfg.html)
66	/// [IsrStatus](structs.isrstatus.html), [PciCfg](structs.pcicfg.html)
67	/// [ShMemCfg](structs.ShMemCfg)).
68	///
69	/// Returns a driver instance of
70	/// [VirtioNetDriver](structs.virtionetdriver.html) or an [VirtioError](enums.virtioerror.html).
71	pub(crate) fn init(
72		device: &PciDevice<PciConfigRegion>,
73	) -> Result<VirtioNetDriver<Init>, VirtioError> {
74		// enable bus master mode
75		device.set_command(CommandRegister::BUS_MASTER_ENABLE);
76
77		let drv = match pci::map_caps(device) {
78			Ok(caps) => match VirtioNetDriver::new(caps, device) {
79				Ok(driver) => driver,
80				Err(vnet_err) => {
81					error!("Initializing new network driver failed. Aborting!");
82					return Err(VirtioError::NetDriver(vnet_err));
83				}
84			},
85			Err(err) => {
86				error!("Mapping capabilities failed. Aborting!");
87				return Err(err);
88			}
89		};
90
91		let initialized_drv = match drv.init_dev() {
92			Ok(initialized_drv) => {
93				info!(
94					"Network device with id {:x}, has been initialized by driver!",
95					initialized_drv.get_dev_id()
96				);
97				initialized_drv
98			}
99			Err(vnet_err) => {
100				return Err(VirtioError::NetDriver(vnet_err));
101			}
102		};
103
104		if initialized_drv.is_link_up() {
105			info!("Virtio-net link is up after initialization.");
106		} else {
107			info!("Virtio-net link is down after initialization!");
108		}
109
110		Ok(initialized_drv)
111	}
112}
