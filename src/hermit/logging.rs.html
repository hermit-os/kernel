7pub static KERNEL_LOGGER: KernelLogger = KernelLogger::new();

9/// Data structure to filter kernel messages
10pub struct KernelLogger {
11	time: AtomicBool,
12}
13
14impl KernelLogger {
15	pub const fn new() -> Self {
16		Self {
17			time: AtomicBool::new(false),
18		}
19	}
20
21	pub fn time(&self) -> bool {
22		self.time.load(Ordering::Relaxed)
23	}
24
25	pub fn set_time(&self, time: bool) {
26		self.time.store(time, Ordering::Relaxed);
27	}
28}
29
30impl log::Log for KernelLogger {
31	fn enabled(&self, _: &Metadata<'_>) -> bool {
32		true
33	}
34
35	fn flush(&self) {
36		// nothing to do
37	}
38
39	fn log(&self, record: &Record<'_>) {
40		if !self.enabled(record.metadata()) {
41			return;
42		}
43
44		// FIXME: Use `super let` once stable
45		let time;
46		let format_time = if self.time() {
47			time = Microseconds(crate::processor::get_timer_ticks());
48			format_args!("[{time}]")
49		} else {
50			format_args!("[            ]")
51		};
52		let core_id = crate::arch::core_local::core_id();
53		let level = ColorLevel(record.level());
54		// FIXME: Use `super let` once stable
55		let target = record.target();
56		let (crate_, modules) = target.split_once("::").unwrap_or((target, ""));
57		let (_modules, module) = modules.rsplit_once("::").unwrap_or(("", modules));
58		let target = if !module.is_empty() && crate_ == "hermit" {
59			module
60		} else {
61			crate_
62		};
63		let format_target = if cfg!(feature = "log-target") {
64			format_args!(" {target:<10}")
65		} else {
66			format_args!("")
67		};
68		let args = record.args();
69		println!("{format_time}[{core_id}][{level}{format_target}] {args}");
70	}
71}

73struct Microseconds(u64);
74
75impl fmt::Display for Microseconds {
76	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
77		let seconds = self.0 / 1_000_000;
78		let microseconds = self.0 % 1_000_000;
79		write!(f, "{seconds:5}.{microseconds:06}")
80	}
81}
82
83struct ColorLevel(Level);
84
85impl fmt::Display for ColorLevel {
86	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
87		let level = self.0;
88
89		if no_color() {
90			write!(f, "{level:<5}")
91		} else {
92			let color = match level {
93				Level::Trace => AnsiColor::Magenta,
94				Level::Debug => AnsiColor::Blue,
95				Level::Info => AnsiColor::Green,
96				Level::Warn => AnsiColor::Yellow,
97				Level::Error => AnsiColor::Red,
98			};
99
100			let style = anstyle::Style::new().fg_color(Some(color.into()));
101			write!(f, "{style}{level:<5}{style:#}")
102		}
103	}
104}
105
106fn no_color() -> bool {
107	option_env!("NO_COLOR").is_some_and(|val| !val.is_empty())
108}

110pub unsafe fn init() {
111	log::set_logger(&KERNEL_LOGGER).expect("Can't initialize logger");
112	// Determines LevelFilter at compile time
113	let log_level: Option<&'static str> = option_env!("HERMIT_LOG_LEVEL_FILTER");
114	let mut max_level = LevelFilter::Info;
115
116	if let Some(log_level) = log_level {
117		max_level = if log_level.eq_ignore_ascii_case("off") {
118			LevelFilter::Off
119		} else if log_level.eq_ignore_ascii_case("error") {
120			LevelFilter::Error
121		} else if log_level.eq_ignore_ascii_case("warn") {
122			LevelFilter::Warn
123		} else if log_level.eq_ignore_ascii_case("info") {
124			LevelFilter::Info
125		} else if log_level.eq_ignore_ascii_case("debug") {
126			LevelFilter::Debug
127		} else if log_level.eq_ignore_ascii_case("trace") {
128			LevelFilter::Trace
129		} else {
130			error!("Could not parse HERMIT_LOG_LEVEL_FILTER, falling back to `info`.");
131			LevelFilter::Info
132		};
133	}
134
135	log::set_max_level(max_level);
136}
