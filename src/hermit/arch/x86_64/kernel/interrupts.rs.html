36pub(crate) fn load_idt() {
37	// FIXME: This is not sound! For this to be sound, the table must never be
38	// modified or destroyed while in use. This is _not_ the case here. Instead, we
39	// disable interrupts on the current core when modifying the table and hope for
40	// the best in regards to interrupts on other cores.
41	unsafe {
42		(*IDT.data_ptr()).load_unsafe();
43	}
44}

47pub(crate) fn enable_and_wait() {
48	#[cfg(feature = "idle-poll")]
49	unsafe {
50		asm!("pause", options(nomem, nostack, preserves_flags));
51	}
52
53	#[cfg(not(feature = "idle-poll"))]
54	if crate::processor::supports_mwait() {
55		use core::ptr;
56
57		let addr = ptr::from_ref(core_scheduler().get_priority_bitmap()).cast::<u8>();
58
59		unsafe {
60			if crate::processor::supports_clflush() {
61				core::arch::x86_64::_mm_clflush(addr);
62			}
63
64			asm!(
65				"monitor",
66				in("rax") addr,
67				in("rcx") 0,
68				in("rdx") 0,
69				options(readonly, nostack, preserves_flags)
70			);
71
72			// The maximum waiting time is an implicit 64-bit timestamp-counter value
73			// stored in the EDX:EBX register pair.
74			// Test timeout by changing "b" => (0xffffffff) or "d"((wakeup >> 32) + 1)
75			// ECX bit 31 indicate whether timeout feature is used
76			// EAX [0:3] indicate sub C-state; [4:7] indicate C-states e.g., 0=>C1, 1=>C2 ...
77			asm!(
78				"sti; mwait",
79				in("rax") 0x2,
80				in("rcx") 0 /* break on interrupt flag */,
81				options(readonly, nostack, preserves_flags)
82			);
83		}
84	} else {
85		#[cfg(feature = "smp")]
86		crate::CoreLocal::get().hlt.store(true, Ordering::Relaxed);
87		enable_and_hlt();
88	}
89}

91pub(crate) fn install() {
92	let mut idt = IDT.lock();
93
94	set_general_handler!(&mut *idt, abort, 0..32);
95	set_general_handler!(&mut *idt, handle_interrupt, 32..);
96
97	unsafe {
98		for i in 32..=255 {
99			let addr = idt[i].handler_addr();
100			idt[i].set_handler_addr(addr).set_stack_index(0);
101		}

103		idt.divide_error
104			.set_handler_fn(divide_error_exception)
105			.set_stack_index(0);
106		idt.debug.set_handler_fn(debug_exception).set_stack_index(0);
107		idt.breakpoint
108			.set_handler_fn(breakpoint_exception)
109			.set_stack_index(0);
110		idt.overflow
111			.set_handler_fn(overflow_exception)
112			.set_stack_index(0);
113		idt.bound_range_exceeded
114			.set_handler_fn(bound_range_exceeded_exception)
115			.set_stack_index(0);
116		idt.invalid_opcode
117			.set_handler_fn(invalid_opcode_exception)
118			.set_stack_index(0);
119		idt.device_not_available
120			.set_handler_fn(device_not_available_exception)
121			.set_stack_index(0);
122		idt.invalid_tss
123			.set_handler_fn(invalid_tss_exception)
124			.set_stack_index(0);
125		idt.segment_not_present
126			.set_handler_fn(segment_not_present_exception)
127			.set_stack_index(0);
128		idt.stack_segment_fault
129			.set_handler_fn(stack_segment_fault_exception)
130			.set_stack_index(0);
131		idt.general_protection_fault
132			.set_handler_fn(general_protection_exception)
133			.set_stack_index(0);
134		idt.page_fault

163	IRQ_NAMES.lock().insert(7, "FPU");
164}

170fn handle_interrupt(stack_frame: ExceptionStackFrame, index: u8, _error_code: Option<u64>) {
171	debug!("received interrupt {index}");
172
173	crate::arch::x86_64::swapgs(&stack_frame);
174	use crate::arch::kernel::core_local::core_scheduler;
175	use crate::scheduler::PerCoreSchedulerExt;
176
177	if let Some(handlers) = IRQ_HANDLERS.get()
178		&& let Some(map) = handlers.get(&(index - 32))
179	{
180		for handler in map.iter() {
181			handler();
182		}
183	}
184
185	apic::eoi();
186	increment_irq_counter(index);
187
188	crate::executor::run();
189
190	core_scheduler().reschedule();
191	crate::arch::x86_64::swapgs(&stack_frame);
192}

243extern "x86-interrupt" fn device_not_available_exception(stack_frame: ExceptionStackFrame) {
244	swapgs(&stack_frame);
245	// We set the CR0_TASK_SWITCHED flag every time we switch to a task.
246	// This causes the "Device Not Available" Exception (int #7) to be thrown as soon as we use the FPU for the first time.
247
248	increment_irq_counter(7);
249
250	// Clear CR0_TASK_SWITCHED so this doesn't happen again before the next switch.
251	unsafe {
252		asm!("clts", options(nomem, nostack));
253	}
254
255	// Let the scheduler set up the FPU for the current task.
256	core_scheduler().fpu_switch();
257	swapgs(&stack_frame);
258}

337pub(crate) fn add_irq_name(irq_number: u8, name: &'static str) {
338	debug!("Register name \"{name}\" for interrupt {irq_number}");

347	InterruptSpinMutex::new(BTreeMap::new());

349pub(crate) struct IrqStatistics {
350	pub counters: [AtomicU64; 256],
351}
352
353impl IrqStatistics {
354	pub const fn new() -> Self {
355		#[allow(clippy::declare_interior_mutable_const)]
356		const NEW_COUNTER: AtomicU64 = AtomicU64::new(0);
357		IrqStatistics {
358			counters: [NEW_COUNTER; 256],
359		}
360	}
361
362	pub fn inc(&self, pos: u8) {
363		self.counters[usize::from(pos)].fetch_add(1, Ordering::Relaxed);
364	}
365}

367pub(crate) fn print_statistics() {
368	panic_println!("Number of interrupts");
369	for (core_id, irg_statistics) in IRQ_COUNTERS.lock().iter() {
370		for (i, counter) in irg_statistics.counters.iter().enumerate() {
371			let counter = counter.load(Ordering::Relaxed);
372			if counter > 0 {
373				match get_irq_name(i.try_into().unwrap()) {
374					Some(name) => {
375						panic_println!("[{core_id}][{name}]: {counter}");
376					}
377					_ => {
378						panic_println!("[{core_id}][{i}]: {counter}");
379					}
380				}
381			}
382		}
383	}
