82	/// Configure the bar at register `slot`
83	pub fn set_bar(&self, slot: u8, bar: Bar) {
84		let value = match bar {
85			Bar::Io { port } => (port | 1) as usize,
86			Bar::Memory32 {
87				address,
88				size: _,
89				prefetchable,
90			} => {
91				if prefetchable {
92					(address | (1 << 3)) as usize
93				} else {
94					address as usize
95				}
96			}
97			Bar::Memory64 {
98				address,
99				size: _,
100				prefetchable,
101			} => {
102				if prefetchable {
103					(address | (2 << 1) | (1 << 3)) as usize
104				} else {
105					(address | (2 << 1)) as usize
106				}
107			}
108		};
109		let mut header = EndpointHeader::from_header(self.header(), &self.access).unwrap();
110		unsafe {
111			header.write_bar(slot, &self.access, value).unwrap();
112		}
113	}

115	/// Memory maps pci bar with specified index to identical location in virtual memory.
116	/// no_cache determines if we set the `Cache Disable` flag in the page-table-entry.
117	/// Returns (virtual-pointer, size) if successful, else None (if bar non-existent or IOSpace)
118	pub fn memory_map_bar(&self, index: u8, no_cache: bool) -> Option<(VirtAddr, usize)> {
119		let (address, size, prefetchable, _width) = match self.get_bar(index) {
120			Some(Bar::Io { .. }) => {
121				warn!("Cannot map IOBar!");
122				return None;
123			}
124			Some(Bar::Memory32 {
125				address,
126				size,
127				prefetchable,
128			}) => (
129				u64::from(address),
130				usize::try_from(size).unwrap(),
131				prefetchable,
132				32,
133			),
134			Some(Bar::Memory64 {
135				address,
136				size,
137				prefetchable,
138			}) => (address, usize::try_from(size).unwrap(), prefetchable, 64),
139			_ => {
140				return None;
141			}
142		};
143
144		if address == 0 {
145			return None;
146		}
147
148		debug!("Mapping bar {index} at {address:#x} with length {size:#x}");
149
150		if !prefetchable {
151			warn!("Currently only mapping of prefetchable bars is supported!");
152		}
153
154		// Since the bios/bootloader manages the physical address space, the address got from the bar is unique and not overlapping.
155		// We therefore do not need to reserve any additional memory in our kernel.
156		// Map bar into RW^X virtual memory
157		let physical_address = address;
158		let virtual_address =
159			crate::mm::map(PhysAddr::new(physical_address), size, true, true, no_cache);
160
161		Some((virtual_address, size))
162	}

164	pub fn get_irq(&self) -> Option<InterruptLine> {
165		let header = self.header();
166		let endpoint = EndpointHeader::from_header(header, &self.access)?;
167		let (pin, line) = endpoint.interrupt(&self.access);
168		// PCIe specification v5 section 7.5.1.1.13 (Interrupt Pin Register)
169		match pin {
170			0 => {
171				warn!("The function uses no legacy interrupt message(s).");
172				None
173			}
174			1..=4 => {
175				// PCI specification v3 footnote 43
176				#[cfg(target_arch = "x86_64")]
177				if matches!(line, 16..254) {
178					error!("Reserved IRQ number");
179					return None;
180				} else if line == 255 {
181					error!("Unknown IRQ line or no connection to the interrupt controller");
182					return None;
183				}
184
185				Some(line)
186			}
187			5.. => {
188				error!("Reserved interrupt pin value returned.");
189				None
190			}
191		}
192	}

195		let mut header = EndpointHeader::from_header(self.header(), &self.access).unwrap();

218impl<T: ConfigRegionAccess> fmt::Display for PciDevice<T> {
219	fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
220		let header = self.header();
221		let header_type = header.header_type(&self.access);
222		let (vendor_id, device_id) = header.id(&self.access);
223		let (_dev_rev, class_id, subclass_id, _interface) = header.revision_and_class(&self.access);
224
225		if let Some(endpoint) = EndpointHeader::from_header(header, &self.access) {
226			#[cfg(feature = "pci-ids")]
227			let (class_name, vendor_name, device_name) = {
228				use pci_ids::{Class, Device, FromId, Subclass};
229
230				let class_name = Class::from_id(class_id).map_or("Unknown Class", |class| {
231					class
232						.subclasses()
233						.find(|s| s.id() == subclass_id)
234						.map(Subclass::name)
235						.unwrap_or_else(|| class.name())
236				});
237
238				let (vendor_name, device_name) = Device::from_vid_pid(vendor_id, device_id)
239					.map(|device| (device.vendor().name(), device.name()))
240					.unwrap_or(("Unknown Vendor", "Unknown Device"));
241
242				(class_name, vendor_name, device_name)
243			};
244
245			#[cfg(not(feature = "pci-ids"))]
246			let (class_name, vendor_name, device_name) =
247				("Unknown Class", "Unknown Vendor", "Unknown Device");
248
249			// Output detailed readable information about this device.
250			write!(
251				f,
252				"{:02X}:{:02X} {} [{:02X}{:02X}]: {} {} [{:04X}:{:04X}]",
253				self.address.bus(),
254				self.address.device(),
255				class_name,
256				class_id,
257				subclass_id,
258				vendor_name,
259				device_name,
260				vendor_id,
261				device_id
262			)?;
263
264			// If the devices uses an IRQ, output this one as well.
265			let (_, irq) = endpoint.interrupt(&self.access);
266			if irq != 0 && irq != u8::MAX {
267				write!(f, ", IRQ {irq}")?;
268			}
269
270			let mut slot: u8 = 0;
271			while usize::from(slot) < MAX_BARS {
272				if let Some(pci_bar) = endpoint.bar(slot, &self.access) {
273					match pci_bar {
274						Bar::Memory64 {
275							address,
276							size,
277							prefetchable,
278						} => {
279							write!(
280								f,
281								", BAR{slot} Memory64 {{ address: {address:#X}, size: {size:#X}, prefetchable: {prefetchable} }}"
282							)?;
283							slot += 1;
284						}
285						Bar::Memory32 {
286							address,
287							size,
288							prefetchable,
289						} => {
290							write!(
291								f,
292								", BAR{slot} Memory32 {{ address: {address:#X}, size: {size:#X}, prefetchable: {prefetchable} }}"
293							)?;
294						}
295						Bar::Io { port } => {
296							write!(f, ", BAR{slot} IO {{ port: {port:#X} }}")?;
297						}
298					}
299				}
300				slot += 1;
301			}
302		} else {
303			// Output detailed readable information about this device.
304			write!(
305				f,
306				"{:02X}:{:02X} {:?} [{:04X}:{:04X}]",
307				self.address.bus(),
308				self.address.device(),
309				header_type,
310				vendor_id,
311				device_id
312			)?;
313		}
314
315		Ok(())
316	}

341impl PciDriver {

343	fn get_console_driver(&self) -> Option<&InterruptTicketMutex<VirtioConsoleDriver>> {
344		#[allow(unreachable_patterns)]

346			Self::VirtioConsole(drv) => Some(drv),
347			_ => None,

349	}

353		#[allow(unreachable_patterns)]
354		match self {
355			Self::VirtioVsock(drv) => Some(drv),
356			_ => None,
357		}
358	}
359
360	#[cfg(feature = "virtio-fs")]
361	fn get_filesystem_driver(&self) -> Option<&InterruptTicketMutex<VirtioFsDriver>> {
362		match self {
363			Self::VirtioFs(drv) => Some(drv),
364			#[allow(unreachable_patterns)]
365			_ => None,
366		}
367	}
368
369	fn get_interrupt_handler(&self) -> (InterruptLine, fn()) {
370		#[allow(unreachable_patterns)]
371		match self {
372			#[cfg(feature = "virtio-vsock")]
373			Self::VirtioVsock(drv) => {
374				fn vsock_handler() {
375					let Some(driver) = get_vsock_driver() else {
376						return;
377					};
378
379					driver.lock().handle_interrupt();
380				}
381
382				let irq_number = drv.lock().get_interrupt_number();
383
384				(irq_number, vsock_handler)
385			}
386			#[cfg(feature = "virtio-fs")]
387			Self::VirtioFs(drv) => {
388				fn fuse_handler() {
389					let Some(driver) = get_filesystem_driver() else {
390						return;
391					};
392
393					driver.lock().handle_interrupt();
394				}
395
396				let irq_number = drv.lock().get_interrupt_number();
397
398				(irq_number, fuse_handler)
399			}
400			#[cfg(feature = "virtio-console")]
401			Self::VirtioConsole(drv) => {
402				fn console_handler() {
403					let Some(driver) = get_console_driver() else {
404						return;
405					};
406
407					driver.lock().handle_interrupt();
408				}
409
410				let irq_number = drv.lock().get_interrupt_number();
411				(irq_number, console_handler)
412			}
413			_ => todo!(),
414		}
415	}
416}

422pub(crate) fn get_interrupt_handlers() -> HashMap<InterruptLine, InterruptHandlerQueue, RandomState>
423{
424	let mut handlers: HashMap<InterruptLine, InterruptHandlerQueue, RandomState> =
425		HashMap::with_hasher(RandomState::with_seeds(0, 0, 0, 0));
426
427	for drv in PCI_DRIVERS.finalize().iter() {
428		let (irq_number, handler) = drv.get_interrupt_handler();
429
430		handlers.entry(irq_number).or_default().push_back(handler);
431	}
432
433	#[cfg(target_arch = "x86_64")]
434	{
435		use crate::kernel::serial::get_serial_handler;
436		let (irq_number, handler) = get_serial_handler();
437
438		handlers.entry(irq_number).or_default().push_back(handler);
439	}
440
441	#[cfg(any(feature = "rtl8139", feature = "virtio-net"))]
442	if let Some(device) = NETWORK_DEVICE.lock().as_ref() {
443		handlers
444			.entry(device.get_interrupt_number())
445			.or_default()
446			.push_back(crate::executor::network::network_handler);
447	}
448
449	handlers
450}

482pub(crate) fn init() {
483	// virtio: 4.1.2 PCI Device Discovery
484	without_interrupts(|| {
485		for adapter in PCI_DEVICES.finalize().iter().filter(|x| {
486			let (vendor_id, device_id) = x.id();
487			vendor_id == 0x1af4 && (0x1000..=0x107f).contains(&device_id)
488		}) {
489			info!(
490				"Found virtio device with device id {:#x}",
491				adapter.device_id()
492			);
493
494			#[cfg(feature = "virtio")]
495			match pci_virtio::init_device(adapter) {
496				#[cfg(feature = "virtio-console")]
497				Ok(VirtioDriver::Console(drv)) => {
498					register_driver(PciDriver::VirtioConsole(InterruptTicketMutex::new(*drv)));
499					info!("Switch to virtio console");
500					crate::console::CONSOLE
501						.lock()
502						.replace_device(IoDevice::Virtio(VirtioUART::new()));
503				}
504				#[cfg(feature = "virtio-fs")]
505				Ok(VirtioDriver::Fs(drv)) => {
506					register_driver(PciDriver::VirtioFs(InterruptTicketMutex::new(*drv)));
507				}
508				#[cfg(all(not(feature = "rtl8139"), feature = "virtio-net"))]
509				Ok(VirtioDriver::Net(drv)) => *crate::executor::device::NETWORK_DEVICE.lock() = Some(*drv),
510				#[cfg(feature = "virtio-vsock")]
511				Ok(VirtioDriver::Vsock(drv)) => {
512					register_driver(PciDriver::VirtioVsock(InterruptTicketMutex::new(*drv)));
513				}
514				Err(err) => error!("Could not initialize virtio-pci device: {err}"),
515			}
516		}
517
518		// Searching for Realtek RTL8139, which is supported by Qemu
519		#[cfg(feature = "rtl8139")]
520		for adapter in PCI_DEVICES.finalize().iter().filter(|x| {
521			let (vendor_id, device_id) = x.id();
522			vendor_id == 0x10ec && (0x8138..=0x8139).contains(&device_id)
523		}) {
524			info!(
525				"Found Realtek network device with device id {:#x}",
526				adapter.device_id()
527			);
528
529			match rtl8139::init_device(adapter) {
530				Ok(drv) => *crate::executor::device::NETWORK_DEVICE.lock() = Some(drv),
531				Err(err) => error!("Could not initialize rtl8139 device: {err}"),
532			}
533		}
534	});
535}
