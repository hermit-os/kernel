10struct addrinfo {

21#[derive(TryFromPrimitive, IntoPrimitive, PartialEq, Eq, Clone, Copy, Debug)]
22#[repr(i32)]
23enum Eai {
24	Again = 2,
25	Badflags = 3,
26	Fail = 4,
27	Family = 5,
28	Memory = 6,
29	Nodata = 7,
30	Noname = 8,
31	Service = 9,
32	Socktype = 10,

52#[hermit_macro::system]
53#[unsafe(no_mangle)]
54pub extern "C" fn sys_gai_strerror(ecode: i32) -> *const c_char {
55	let Ok(ecode) = Eai::try_from(ecode) else {
56		return c"Unknown error".as_ptr();
57	};
58
59	let s = match ecode {
60		Eai::Again => c"Try again",
61		Eai::Badflags => c"Invalid flags",
62		Eai::Fail => c"Non-recoverable error",
63		Eai::Family => c"Unrecognized address family or invalid length",
64		Eai::Memory => c"Out of memory",
65		Eai::Nodata => c"Name has no usable address",
66		Eai::Noname => c"Name does not resolve",
67		Eai::Service => c"Unrecognized service",
68		Eai::Socktype => c"Unrecognized socket type",
69		Eai::System => c"System error",
70		Eai::Overflow => c"Overflow",
71	};
72
73	s.as_ptr()
74}
