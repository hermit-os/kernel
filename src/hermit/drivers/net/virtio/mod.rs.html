54	pub features: virtio::net::F,
55}

57fn determine_mtu(dev_cfg: &NetDevCfg) -> u16 {
58	// If VIRTIO_NET_F_MTU is negotiated, "the driver uses mtu as the maximum MTU value"
59	// (VirtIO specification, 5.1.3, "Feature bits")
60	if dev_cfg.features.contains(virtio::net::F::MTU) {
61		dev_cfg.raw.as_ptr().mtu().read().to_ne()
62	} else {
63		// Otherwise, we can just use the MTU we want to use
64		mtu()
65	}
66}

68fn determine_rx_buf_size(dev_cfg: &NetDevCfg) -> u32 {
69	// See Virtio specification v1.1 - 5.1.6.3.1 and 5.1.4.2
70
71	// Our desired minimum buffer size - we want it to be at least the MTU generally
72	let mut min_buf_size = determine_mtu(dev_cfg).into();
73
74	// If VIRTIO_NET_F_MRG_RXBUF is negotiated, each buffer MUST be at least the size of the struct virtio_net_hdr.
75	// We just use MTU in that case, but otherwise...
76	if dev_cfg.features.contains(virtio::net::F::MRG_RXBUF)
77		&& let Some(my_mrg_rxbuf_size) = hermit_var!("HERMIT_MRG_RXBUF_SIZE")
78	{
79		let my_mrg_rxbuf_size = u32::from_str(&my_mrg_rxbuf_size).unwrap();
80		assert!(
81			my_mrg_rxbuf_size > 0,
82			"VIRTIO does not allow buffer elements of size 0."
83		);
84		min_buf_size = my_mrg_rxbuf_size;
85	} else {
86		// If [...] are negotiated, the driver SHOULD populate the receive queue(s) with buffers of at least 65562 bytes.
87		if dev_cfg.features.contains(virtio::net::F::GUEST_TSO4)
88			|| dev_cfg.features.contains(virtio::net::F::GUEST_TSO6)
89			|| dev_cfg.features.contains(virtio::net::F::GUEST_UFO)
90		{
91			min_buf_size = u32::max(min_buf_size, 65562 - size_of::<Hdr>() as u32);
92		} else {
93			// Otherwise, the driver SHOULD populate the receive queue(s) with buffers of at least 1526 bytes.
94			min_buf_size = u32::max(min_buf_size, 1526 - size_of::<Hdr>() as u32);
95		}
96	}
97
98	min_buf_size
99}

101pub struct RxQueues {
102	vqs: Vec<VirtQueue>,
103	buf_size: u32,
104}
105
106impl RxQueues {
107	pub fn new(vqs: Vec<VirtQueue>, dev_cfg: &NetDevCfg) -> Self {
108		Self {
109			vqs,
110			buf_size: determine_rx_buf_size(dev_cfg),
111		}
112	}
113
114	/// Adds a given queue to the underlying vector and populates the queue with RecvBuffers.
115	///
116	/// Queues are all populated according to Virtio specification v1.1. - 5.1.6.3.1
117	fn add(&mut self, mut vq: VirtQueue) {
118		let num_bufs = vq.size() / constants::BUFF_PER_PACKET;
119		fill_queue(&mut vq, num_bufs, self.buf_size);
120		self.vqs.push(vq);
121	}
122
123	fn get_next(&mut self) -> Option<UsedBufferToken> {
124		self.vqs[0].try_recv().ok()
125	}
126
127	fn enable_notifs(&mut self) {
128		for vq in &mut self.vqs {
129			vq.enable_notifs();
130		}
131	}
132
133	fn disable_notifs(&mut self) {
134		for vq in &mut self.vqs {
135			vq.disable_notifs();
136		}
137	}
138
139	fn has_packet(&self) -> bool {
140		self.vqs.iter().any(|vq| vq.has_used_buffers())
141	}
142}
143
144fn buffer_token_from_hdr(
145	hdr: Box<MaybeUninit<Hdr>, DeviceAlloc>,
146	buf_size: u32,
147) -> AvailBufferToken {
148	AvailBufferToken::new(SmallVec::new(), {
149		SmallVec::from_buf([
150			BufferElem::Sized(hdr),
151			BufferElem::Vector(Vec::with_capacity_in(
152				buf_size.try_into().unwrap(),
153				DeviceAlloc,
154			)),
155		])
156	})
157	.unwrap()
158}
159
160fn fill_queue(vq: &mut VirtQueue, num_bufs: u16, buf_size: u32) {
161	for _ in 0..num_bufs {
162		let buff_tkn = buffer_token_from_hdr(Box::<Hdr, _>::new_uninit_in(DeviceAlloc), buf_size);
163
164		// BufferTokens are directly provided to the queue
165		// TransferTokens are directly dispatched
166		// Transfers will be awaited at the queue
167		if let Err(err) = vq.dispatch(buff_tkn, false, BufferType::Direct) {
168			error!("{err:#?}");
169			break;
170		}
171	}
172}

174/// Structure which handles transmission of packets and delegation
175/// to the respective queue structures.
176pub struct TxQueues {
177	vqs: Vec<VirtQueue>,
178	buf_size: u32,
179}
180
181impl TxQueues {
182	pub fn new(vqs: Vec<VirtQueue>, dev_cfg: &NetDevCfg) -> Self {
183		Self {
184			vqs,
185			buf_size: determine_mtu(dev_cfg).into(),
186		}
187	}
188
189	#[allow(dead_code)]
190	fn enable_notifs(&mut self) {
191		for vq in &mut self.vqs {
192			vq.enable_notifs();
193		}
194	}
195
196	#[allow(dead_code)]
197	fn disable_notifs(&mut self) {
198		for vq in &mut self.vqs {
199			vq.disable_notifs();
200		}
201	}
202
203	/// Polls all queues for buffers whose transmission has been completed and returns the number of such buffers.
204	fn poll(&mut self) -> u32 {
205		let mut released_buffers = 0u32;
206		for vq in &mut self.vqs {
207			// We don't do anything with the buffers but we need to receive them for the
208			// ring slots to be emptied and the memory from the previous transfers to be freed.
209			while vq.try_recv().is_ok() {
210				released_buffers += 1;
211			}
212		}
213		released_buffers
214	}
215
216	fn add(&mut self, vq: VirtQueue) {
217		// Currently we are doing nothing with the additional queues. They are inactive and might be used in the
218		// future
219		self.vqs.push(vq);
220	}
221}

256impl Drop for TxToken<'_> {
257	fn drop(&mut self) {
258		*self.send_capacity += u32::from(BUFF_PER_PACKET);
259	}

261
262impl smoltcp::phy::TxToken for TxToken<'_> {
263	fn consume<R, F>(self, len: usize, f: F) -> R
264	where
265		F: FnOnce(&mut [u8]) -> R,
266	{
267		// When the token is consumed, the capacity cannot be returned until its buffer is marked by the device as used.
268		// Thus, we bypass the Drop implementation that would do that prematurely and let the call to poll in the next
269		// call to this function return the capacity.
270		let mut token = ManuallyDrop::new(self);
271		assert!(len <= usize::try_from(token.send_vqs.buf_size).unwrap());
272		let mut packet = Vec::with_capacity_in(len, DeviceAlloc);
273		let result = unsafe {
274			let result = f(packet.spare_capacity_mut().assume_init_mut());
275			packet.set_len(len);
276			result
277		};
278
279		let mut header = Box::new_in(<Hdr as Default>::default(), DeviceAlloc);
280
281		// If a checksum calculation by the host is necessary, we have to inform the host within the header
282		// see Virtio specification 5.1.6.2
283		if let Some((ip_header_len, csum_offset)) =
284			VirtioNetDriver::should_request_checksum(&token.checksums, &mut packet)
285		{
286			header.flags = HdrF::NEEDS_CSUM;
287			header.csum_start =
288				(u16::try_from(ETHERNET_HEADER_LEN).unwrap() + ip_header_len).into();
289			header.csum_offset = csum_offset.into();
290		}
291
292		let buff_tkn = AvailBufferToken::new(
293			SmallVec::from_buf([BufferElem::Sized(header), BufferElem::Vector(packet)]),
294			SmallVec::new(),
295		)
296		.unwrap();
297
298		token.send_vqs.vqs[0]
299			.dispatch(buff_tkn, false, BufferType::Direct)
300			.unwrap();
301
302		result
303	}
304}

311impl smoltcp::phy::RxToken for RxToken<'_> {
312	fn consume<R, F>(self, f: F) -> R
313	where
314		F: FnOnce(&[u8]) -> R,
315	{
316		let Some(mut buffer_tkn) = self.recv_vqs.get_next() else {
317			// We overpromised a frame. The best we can do is to provide an empty frame to smoltcp and let it handle it as a faulty reception.
318			return f(&[]);
319		};
320		// Safety: any buffers that do not start with a `Hdr` must have been consumed by the previous call
321		// to this function.
322		let first_header = unsafe {
323			buffer_tkn
324				.used_recv_buff
325				.pop_front_downcast::<Hdr>()
326				.unwrap()
327		};
328		let first_packet = buffer_tkn.used_recv_buff.pop_front_vec().unwrap();
329
330		// According to VIRTIO spec v1.2 sec. 5.1.6.3.2, "num_buffers will always be 1 if VIRTIO_NET_F_MRG_RXBUF is not negotiated."
331		// Unfortunately, NVIDIA MLX5 does not comply with this requirement and we have to manually set the value to the correct one.
332		let num_buffers = if self.is_mrg_rxbuf_enabled {
333			first_header.num_buffers.to_ne()
334		} else {
335			1
336		};
337
338		let mut combined_packets = first_packet;
339
340		let first_tkn = buffer_token_from_hdr(
341			// SAFETY: Box<T> -> Box<MaybeUninit<T>> is sound
342			unsafe {
343				mem::transmute::<Box<Hdr, DeviceAlloc>, Box<MaybeUninit<Hdr>, DeviceAlloc>>(
344					first_header,
345				)
346			},
347			self.recv_vqs.buf_size,
348		);
349		self.recv_vqs.vqs[0]
350			.dispatch(first_tkn, false, BufferType::Direct)
351			.unwrap();
352
353		for _ in 1..num_buffers {
354			let mut buffer_tkn = self.recv_vqs.get_next().unwrap();
355			// The descriptor that was meant for the header of another frame was used for a portion of the current frame's contents.
356			// Thus, we cannot cast it to a Hdr.
357			let (header_descriptor, used_len) = buffer_tkn.used_recv_buff.pop_front_raw().unwrap();
358			combined_packets.extend_from_slice(unsafe {
359				slice::from_raw_parts((&raw const *header_descriptor).cast::<u8>(), used_len)
360			});
361
362			let packet = buffer_tkn.used_recv_buff.pop_front_vec().unwrap();
363			combined_packets.extend_from_slice(&packet);
364
365			let header = header_descriptor.downcast::<MaybeUninit<Hdr>>().unwrap();
366
367			let tkn = buffer_token_from_hdr(
368				// SAFETY: Box<T> -> Box<MaybeUninit<T>> is sound
369				header,
370				self.recv_vqs.buf_size,
371			);
372			self.recv_vqs.vqs[0]
373				.dispatch(tkn, false, BufferType::Direct)
374				.unwrap();
375		}
376
377		f(&combined_packets)
378	}
379}

424impl smoltcp::phy::Device for VirtioNetDriver {
425	type TxToken<'a> = TxToken<'a>;
426	type RxToken<'a> = RxToken<'a>;
427
428	fn capabilities(&self) -> DeviceCapabilities {
429		let mut device_capabilities = DeviceCapabilities::default();
430		device_capabilities.medium = smoltcp::phy::Medium::Ethernet;
431		device_capabilities.max_transmission_unit = self.inner.mtu.into();
432		device_capabilities.max_burst_size =
433			Some(usize::try_from(self.inner.send_capacity).unwrap() / usize::from(BUFF_PER_PACKET));
434		device_capabilities.checksum = self.checksums.clone();
435		device_capabilities
436	}
437
438	fn receive(
439		&mut self,
440		_timestamp: smoltcp::time::Instant,
441	) -> Option<(Self::RxToken<'_>, Self::TxToken<'_>)> {
442		if !self.inner.recv_vqs.has_packet() {
443			return None;
444		}
445
446		self.free_up_send_capacity();
447
448		self.inner.send_capacity = self
449			.inner
450			.send_capacity
451			.checked_sub(u32::from(BUFF_PER_PACKET))?;
452
453		Some((
454			RxToken {
455				recv_vqs: &mut self.inner.recv_vqs,
456				is_mrg_rxbuf_enabled: self.dev_cfg.features.contains(virtio::net::F::MRG_RXBUF),
457			},
458			TxToken {
459				send_vqs: &mut self.inner.send_vqs,
460				checksums: self.checksums.clone(),
461				send_capacity: &mut self.inner.send_capacity,
462			},
463		))
464	}
465
466	fn transmit(&mut self, _timestamp: smoltcp::time::Instant) -> Option<Self::TxToken<'_>> {
467		self.free_up_send_capacity();
468
469		self.inner.send_capacity = self
470			.inner
471			.send_capacity
472			.checked_sub(u32::from(BUFF_PER_PACKET))?;
473
474		Some(TxToken {
475			send_vqs: &mut self.inner.send_vqs,
476			checksums: self.checksums.clone(),
477			send_capacity: &mut self.inner.send_capacity,
478		})
479	}
480}

571	/// If necessary, sets the TCP or UDP checksum field to the checksum of the
572	/// pseudo-header and returns the IP header length and the checksum offset.
573	/// Otherwise, returns None.
574	fn should_request_checksum<T: AsRef<[u8]> + AsMut<[u8]>>(
575		checksums: &ChecksumCapabilities,
576		frame: T,
577	) -> Option<(u16, u16)> {
578		if checksums.tcp.tx() && checksums.udp.tx() {
579			return None;
580		}
581
582		let ip_header_len: u16;
583		let ip_packet_len: usize;
584		let protocol;
585		let pseudo_header_checksum;
586		let mut ethernet_frame = EthernetFrame::new_unchecked(frame);
587		match ethernet_frame.ethertype() {
588			smoltcp::wire::EthernetProtocol::Ipv4 => {
589				let ip_packet = Ipv4Packet::new_unchecked(&*ethernet_frame.payload_mut());
590				ip_header_len = ip_packet.header_len().into();
591				ip_packet_len = ip_packet.total_len().into();
592				protocol = ip_packet.next_header();
593				pseudo_header_checksum =
594					partial_checksum::ipv4_pseudo_header_partial_checksum(&ip_packet);
595			}
596			smoltcp::wire::EthernetProtocol::Ipv6 => {
597				let ip_packet = Ipv6Packet::new_unchecked(&*ethernet_frame.payload_mut());
598				ip_header_len = ip_packet.header_len().try_into().expect(
599					"VIRTIO does not support IP headers that are longer than u16::MAX bytes.",
600				);
601				ip_packet_len = ip_packet.total_len();
602				protocol = ip_packet.next_header();
603				pseudo_header_checksum =
604					partial_checksum::ipv6_pseudo_header_partial_checksum(&ip_packet);
605			}
606			// If the Ethernet protocol is not one of these two above, for which we know there may be a checksum field,
607			// we default to not asking for checksum, as otherwise the frame will be corrupted by the device trying
608			// to write the checksum.
609			_ => return None,
610		};
611
612		let csum_offset;
613		let ip_payload = &mut ethernet_frame.payload_mut()[ip_header_len.into()..ip_packet_len];
614		// Like the Ethernet protocol check, we check for IP protocols for which we know the location of the checksum field.
615		if protocol == smoltcp::wire::IpProtocol::Tcp && !checksums.tcp.tx() {
616			let mut tcp_packet = smoltcp::wire::TcpPacket::new_unchecked(ip_payload);
617			tcp_packet.set_checksum(pseudo_header_checksum);
618			csum_offset = 16;
619		} else if protocol == smoltcp::wire::IpProtocol::Udp && !checksums.udp.tx() {
620			let mut udp_packet = smoltcp::wire::UdpPacket::new_unchecked(ip_payload);
621			udp_packet.set_checksum(pseudo_header_checksum);
622			csum_offset = 6;
623		} else {
624			return None;
625		};
626
627		Some((ip_header_len, csum_offset))
628	}
629
630	fn free_up_send_capacity(&mut self) {
631		// We need to poll to get the queue to remove elements from the table and open up capacity if possible.
632		self.inner.send_capacity += self.inner.send_vqs.poll() * u32::from(BUFF_PER_PACKET);
633	}
634}

636impl VirtioNetDriver<Uninit> {
637	/// Initializes the device in adherence to specification. Returns Some(VirtioNetError)
638	/// upon failure and None in case everything worked as expected.
639	///
640	/// See Virtio specification v1.1. - 3.1.1.
641	///                      and v1.1. - 5.1.5
642	pub fn init_dev(mut self) -> Result<VirtioNetDriver<Init>, VirtioNetError> {
643		// Reset
644		self.com_cfg.reset_dev();
645
646		// Indicate device, that OS noticed it
647		self.com_cfg.ack_dev();
648
649		// Indicate device, that driver is able to handle it
650		self.com_cfg.set_drv();
651
652		let minimal_features = virtio::net::F::VERSION_1 | virtio::net::F::MAC;
653
654		// If wanted, push new features into feats here:
655		let features = minimal_features
656			// Indirect descriptors can be used
657			| virtio::net::F::INDIRECT_DESC
658			// Packed Vq can be used
659			| virtio::net::F::RING_PACKED
660			| virtio::net::F::NOTIFICATION_DATA
661			// MTU setting can be used
662			| virtio::net::F::MTU
663			// Driver can merge receive buffers
664			| virtio::net::F::MRG_RXBUF
665			// the link status can be announced
666			| virtio::net::F::STATUS
667			// control queue support
668			| virtio::net::F::CTRL_VQ
669			// Multiqueue support
670			| virtio::net::F::MQ
671			// Checksum calculation can partially be offloaded to the device
672			| virtio::net::F::CSUM;
673
674		// Currently the driver does NOT support the features below.
675		// In order to provide functionality for these, the driver
676		// needs to take care of calculating checksum.
677		// | virtio::net::F::GUEST_TSO4
678		// | virtio::net::F::GUEST_TSO6
679
680		let negotiated_features = self
681			.com_cfg
682			.control_registers()
683			.negotiate_features(features);
684
685		if !negotiated_features.contains(minimal_features) {
686			error!("Device features set, does not satisfy minimal features needed. Aborting!");
687			return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
688		}
689
690		// Indicates the device, that the current feature set is final for the driver
691		// and will not be changed.
692		self.com_cfg.features_ok();
693
694		// Checks if the device has accepted final set. This finishes feature negotiation.
695		if self.com_cfg.check_features() {
696			info!(
697				"Features have been negotiated between virtio network device {:x} and driver.",
698				self.dev_cfg.dev_id
699			);
700			// Set feature set in device config fur future use.
701			self.dev_cfg.features = negotiated_features;
702		} else {
703			error!("The device does not support our subset of features.");
704			return Err(VirtioNetError::FailFeatureNeg(self.dev_cfg.dev_id));
705		}
706
707		let mut inner = Init {
708			mtu: determine_mtu(&self.dev_cfg),
709			ctrl_vq: None,
710			recv_vqs: RxQueues::new(Vec::new(), &self.dev_cfg),
711			send_vqs: TxQueues::new(Vec::new(), &self.dev_cfg),
712			send_capacity: 0,
713		};
714
715		debug!("Using RX buffer size of {}", inner.recv_vqs.buf_size);
716
717		self.dev_spec_init(&mut inner)?;
718		info!(
719			"Device specific initialization for Virtio network device {:x} finished",
720			self.dev_cfg.dev_id
721		);
722
723		// At this point the device is "live"
724		self.com_cfg.drv_ok();
725
726		if self.dev_cfg.features.contains(virtio::net::F::CSUM)
727			&& self.dev_cfg.features.contains(virtio::net::F::GUEST_CSUM)
728		{
729			self.checksums.udp = Checksum::None;
730			self.checksums.tcp = Checksum::None;
731		} else if self.dev_cfg.features.contains(virtio::net::F::CSUM) {
732			self.checksums.udp = Checksum::Rx;
733			self.checksums.tcp = Checksum::Rx;
734		} else if self.dev_cfg.features.contains(virtio::net::F::GUEST_CSUM) {
735			self.checksums.udp = Checksum::Tx;
736			self.checksums.tcp = Checksum::Tx;
737		}
738		debug!("{:?}", self.checksums);
739
740		Ok(VirtioNetDriver {
741			dev_cfg: self.dev_cfg,
742			com_cfg: self.com_cfg,
743			isr_stat: self.isr_stat,
744			notif_cfg: self.notif_cfg,
745			inner,
746			num_vqs: self.num_vqs,
747			irq: self.irq,
748			checksums: self.checksums,
749		})
750	}
751
752	/// Device Specific initialization according to Virtio specifictation v1.1. - 5.1.5
753	fn dev_spec_init(&mut self, inner: &mut Init) -> Result<(), VirtioNetError> {
754		self.virtqueue_init(inner)?;
755		info!("Network driver successfully initialized virtqueues.");
756
757		// Add a control if feature is negotiated
758		if self.dev_cfg.features.contains(virtio::net::F::CTRL_VQ) {
759			let mut ctrl_vq = if self.dev_cfg.features.contains(virtio::net::F::RING_PACKED) {
760				VirtQueue::Packed(
761					PackedVq::new(
762						&mut self.com_cfg,
763						&self.notif_cfg,
764						VIRTIO_MAX_QUEUE_SIZE,
765						self.num_vqs,
766						self.dev_cfg.features.into(),
767					)
768					.unwrap(),
769				)
770			} else {
771				VirtQueue::Split(
772					SplitVq::new(
773						&mut self.com_cfg,
774						&self.notif_cfg,
775						VIRTIO_MAX_QUEUE_SIZE,
776						self.num_vqs,
777						self.dev_cfg.features.into(),
778					)
779					.unwrap(),
780				)
781			};
782
783			ctrl_vq.enable_notifs();
784			inner.ctrl_vq = Some(ctrl_vq);
785		}
786
787		Ok(())
788	}
789
790	/// Initialize virtqueues via the queue interface and populates receiving queues
791	fn virtqueue_init(&mut self, inner: &mut Init) -> Result<(), VirtioNetError> {
792		// We are assuming here, that the device single source of truth is the
793		// device specific configuration. Hence we do NOT check if
794		//
795		// max_virtqueue_pairs + 1 < num_queues
796		//
797		// - the plus 1 is due to the possibility of an existing control queue
798		// - the num_queues is found in the ComCfg struct of the device and defines the maximal number
799		// of supported queues.
800		if self.dev_cfg.features.contains(virtio::net::F::MQ) {
801			if self
802				.dev_cfg
803				.raw
804				.as_ptr()
805				.max_virtqueue_pairs()
806				.read()
807				.to_ne() * 2 >= MAX_NUM_VQ
808			{
809				self.num_vqs = MAX_NUM_VQ;
810			} else {
811				self.num_vqs = self
812					.dev_cfg
813					.raw
814					.as_ptr()
815					.max_virtqueue_pairs()
816					.read()
817					.to_ne() * 2;
818			}
819		} else {
820			// Minimal number of virtqueues defined in the standard v1.1. - 5.1.5 Step 1
821			self.num_vqs = 2;
822		}
823
824		// The loop is running from 0 to num_vqs and the indexes are provided in this way
825		// in order to allow the indexes of the queues to be in a form of:
826		//
827		// index i for receive queue
828		// index i+1 for send queue
829		//
830		// as it is wanted by the network network device.
831		// see Virtio specification v1.1. - 5.1.2
832		// Assure that we have always an even number of queues (i.e. pairs of queues).
833		assert_eq!(self.num_vqs % 2, 0);
834
835		for i in 0..(self.num_vqs / 2) {
836			if self.dev_cfg.features.contains(virtio::net::F::RING_PACKED) {
837				let mut vq = PackedVq::new(
838					&mut self.com_cfg,
839					&self.notif_cfg,
840					VIRTIO_MAX_QUEUE_SIZE,
841					2 * i,
842					self.dev_cfg.features.into(),
843				)
844				.unwrap();
845				// Interrupt for receiving packets is wanted
846				vq.enable_notifs();
847
848				inner.recv_vqs.add(VirtQueue::Packed(vq));
849
850				let mut vq = PackedVq::new(
851					&mut self.com_cfg,
852					&self.notif_cfg,
853					VIRTIO_MAX_QUEUE_SIZE,
854					2 * i + 1,
855					self.dev_cfg.features.into(),
856				)
857				.unwrap();
858				// Interrupt for communicating that a sent packet left, is not needed
859				vq.disable_notifs();
860
861				inner.send_capacity += u32::from(vq.size());
862				inner.send_vqs.add(VirtQueue::Packed(vq));
863			} else {
864				let mut vq = SplitVq::new(
865					&mut self.com_cfg,
866					&self.notif_cfg,
867					VIRTIO_MAX_QUEUE_SIZE,
868					2 * i,
869					self.dev_cfg.features.into(),
870				)
871				.unwrap();
872				// Interrupt for receiving packets is wanted
873				vq.enable_notifs();
874
875				inner.recv_vqs.add(VirtQueue::Split(vq));
876
877				let mut vq = SplitVq::new(
878					&mut self.com_cfg,
879					&self.notif_cfg,
880					VIRTIO_MAX_QUEUE_SIZE,
881					2 * i + 1,
882					self.dev_cfg.features.into(),
883				)
884				.unwrap();
885				// Interrupt for communicating that a sent packet left, is not needed
886				vq.disable_notifs();
887				inner.send_capacity += u32::from(vq.size());
888				inner.send_vqs.add(VirtQueue::Split(vq));
889			}
890		}
891
892		Ok(())
893	}
894}

923/// The checksum functions in this module only calculate the one's complement sum for the pseudo-header
924/// and their results are meant to be combined with the TCP payload to calculate the real checksum.
925/// They are only useful for the VIRTIO driver with the checksum offloading feature.
926///
927/// The calculations here can theoretically be made faster by exploiting the properties described in
928/// [RFC 1071 section 2](https://www.rfc-editor.org/rfc/rfc1071).
929mod partial_checksum {
930	use smoltcp::wire::{Ipv4Packet, Ipv6Packet};
931
932	fn addr_sum<const N: usize>(addr: &[u8; N]) -> u16 {
933		let mut sum = 0;
934		const CHUNK_SIZE: usize = size_of::<u16>();
935		for i in 0..(N / CHUNK_SIZE) {
936			sum = ones_complement_add(
937				sum,
938				(u16::from(addr[CHUNK_SIZE * i]) << 8) | u16::from(addr[CHUNK_SIZE * i + 1]),
939			);
940		}
941		sum
942	}
943
944	/// Calculates the checksum for the IPv4 pseudo-header as described in
945	/// [RFC 9293 subsection 3.1](https://www.rfc-editor.org/rfc/rfc9293.html#section-3.1-6.18.1) WITHOUT the final inversion.
946	pub(super) fn ipv4_pseudo_header_partial_checksum<T: AsRef<[u8]>>(
947		packet: &Ipv4Packet<T>,
948	) -> u16 {
949		let padded_protocol = u16::from(u8::from(packet.next_header()));
950		let payload_len = packet.total_len() - u16::from(packet.header_len());
951
952		let mut sum = addr_sum(&packet.src_addr().octets());
953		sum = ones_complement_add(sum, addr_sum(&packet.dst_addr().octets()));
954		sum = ones_complement_add(sum, padded_protocol);
955		ones_complement_add(sum, payload_len)
956	}
957
958	/// Calculates the checksum for the IPv6 pseudo-header as described in
959	/// [RFC 8200 subsection 8.1](https://www.rfc-editor.org/rfc/rfc8200.html#section-8.1) WITHOUT the final inversion.
960	pub(super) fn ipv6_pseudo_header_partial_checksum<T: AsRef<[u8]>>(
961		packet: &Ipv6Packet<T>,
962	) -> u16 {
963		warn!("The IPv6 partial checksum implementation is untested!");
964		let padded_protocol = u16::from(u8::from(packet.next_header()));
965
966		let mut sum = addr_sum(&packet.src_addr().octets());
967		sum = ones_complement_add(sum, addr_sum(&packet.dst_addr().octets()));
968		sum = ones_complement_add(sum, packet.payload_len());
969		ones_complement_add(sum, padded_protocol)
970	}
971
972	/// Implements one's complement checksum as described in [RFC 1071 section 1](https://www.rfc-editor.org/rfc/rfc1071#section-1).
973	fn ones_complement_add(lhs: u16, rhs: u16) -> u16 {
974		let (sum, overflow) = u16::overflowing_add(lhs, rhs);
975		sum + u16::from(overflow)
976	}
