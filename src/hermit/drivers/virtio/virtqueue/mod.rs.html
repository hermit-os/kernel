42/// The Virtq trait unifies access to the two different Virtqueue types
43/// [packed::PackedVq] and [split::SplitVq].
44///
45/// The trait provides a common interface for both types. Which in some case
46/// might not provide the complete feature set of each queue. Drivers who
47/// do need these features should refrain from providing support for both
48/// Virtqueue types and use the structs directly instead.
49#[enum_dispatch]
50pub trait Virtq: Send {
51	/// The `notif` parameter indicates if the driver wants to have a notification for this specific
52	/// transfer. This is only for performance optimization. As it is NOT ensured, that the device sees the
53	/// updated notification flags before finishing transfers!
54	fn dispatch(
55		&mut self,
56		tkn: AvailBufferToken,
57		notif: bool,
58		buffer_type: BufferType,
59	) -> Result<(), VirtqError>;
60
61	/// Dispatches the provided TransferToken to the respective queue and does
62	/// return when, the queue finished the transfer.
63	///
64	/// The returned [UsedBufferToken] can be copied from
65	/// or return the underlying buffers.
66	///
67	/// **INFO:**
68	/// Currently this function is constantly polling the queue while keeping the notifications disabled.
69	/// Upon finish notifications are enabled again.
70	#[allow(dead_code)]
71	fn dispatch_blocking(
72		&mut self,
73		tkn: AvailBufferToken,
74		buffer_type: BufferType,
75	) -> Result<UsedBufferToken, VirtqError> {
76		self.dispatch(tkn, false, buffer_type)?;
77
78		self.disable_notifs();
79
80		let result: UsedBufferToken;
81		// Keep Spinning until the receive queue is filled
82		loop {
83			// TODO: normally, we should check if the used buffer in question is the one
84			// we just made available. However, this shouldn't be a problem as the queue this
85			// function is called on makes use of this blocking dispatch function exclusively
86			// and thus dispatches cannot be interleaved.
87			if let Ok(buffer_tkn) = self.try_recv() {
88				result = buffer_tkn;
89				break;
90			}
91		}
92
93		self.enable_notifs();
94
95		Ok(result)
96	}
97
98	/// Enables interrupts for this virtqueue upon receiving a transfer
99	fn enable_notifs(&mut self);
100
101	/// Disables interrupts for this virtqueue upon receiving a transfer
102	fn disable_notifs(&mut self);
103
104	/// Checks if new used descriptors have been written by the device.
105	/// This activates the queue and polls the descriptor ring of the queue.
106	fn try_recv(&mut self) -> Result<UsedBufferToken, VirtqError>;
107
108	/// Dispatches a batch of [AvailBufferToken]s. The buffers are provided to the queue in
109	/// sequence. After the last buffer has been written, the queue marks the first buffer as available and triggers
110	/// a device notification if wanted by the device.
111	///
112	/// The `notif` parameter indicates if the driver wants to have a notification for this specific
113	/// transfer. This is only for performance optimization. As it is NOT ensured, that the device sees the
114	/// updated notification flags before finishing transfers!
115	#[allow(dead_code)]
116	fn dispatch_batch(
117		&mut self,
118		tkns: Vec<(AvailBufferToken, BufferType)>,
119		notif: bool,
120	) -> Result<(), VirtqError>;
121
122	/// Dispatches a batch of [AvailBufferToken]s. The tokens will be placed in to the `await_queue`
123	/// upon finish.
124	///
125	/// The `notif` parameter indicates if the driver wants to have a notification for this specific
126	/// transfer. This is only for performance optimization. As it is NOT ensured, that the device sees the
127	/// updated notification flags before finishing transfers!
128	///
129	/// The buffers are provided to the queue in
130	/// sequence. After the last buffer has been written, the queue marks the first buffer as available and triggers
131	/// a device notification if wanted by the device.
132	///
133	/// Tokens to get a reference to the provided await_queue, where they will be placed upon finish.
134	#[allow(dead_code)]
135	fn dispatch_batch_await(
136		&mut self,
137		tkns: Vec<(AvailBufferToken, BufferType)>,
138		notif: bool,
139	) -> Result<(), VirtqError>;
140
141	/// Returns the size of a Virtqueue. This represents the overall size and not the capacity the
142	/// queue currently has for new descriptors.
143	#[allow(dead_code)]
144	fn size(&self) -> u16;
145
146	// Returns the index (ID) of a Virtqueue.
147	#[allow(dead_code)]
148	fn index(&self) -> u16;
149
150	#[allow(dead_code)]
151	fn has_used_buffers(&self) -> bool;
152}

154/// These methods are an implementation detail and are meant only for consumption by the default method
155/// implementations in [Virtq].
156trait VirtqPrivate {
157	type Descriptor: VirtqDescriptor;
158
159	fn create_indirect_ctrl(
160		buffer_tkn: &AvailBufferToken,
161	) -> Result<Box<[Self::Descriptor]>, VirtqError>;
162
163	fn indirect_desc(table: &[Self::Descriptor]) -> Self::Descriptor {
164		Self::Descriptor::incomplete_desc(
165			DeviceAlloc
166				.phys_addr_from(table.as_ptr().cast_mut())
167				.as_u64()
168				.into(),
169			(mem::size_of_val(table) as u32).into(),
170			virtq::DescF::INDIRECT,
171		)
172	}
173
174	/// Consumes the [AvailBufferToken] and returns a [TransferToken], that can be used to actually start the transfer.
175	///
176	/// After this call, the buffers are no longer writable.
177	fn transfer_token_from_buffer_token(
178		buff_tkn: AvailBufferToken,
179		buffer_type: BufferType,
180	) -> TransferToken<Self::Descriptor> {
181		let ctrl_desc = match buffer_type {
182			BufferType::Direct => None,
183			BufferType::Indirect => Some(Self::create_indirect_ctrl(&buff_tkn).unwrap()),
184		};
185
186		TransferToken {
187			buff_tkn,
188			ctrl_desc,
189		}
190	}
191
192	// The descriptors returned by the iterator will be incomplete, as they do not
193	// have all the information necessary.
194	fn descriptor_iter(
195		buffer_tkn: &AvailBufferToken,
196	) -> Result<impl DoubleEndedIterator<Item = Self::Descriptor>, VirtqError> {
197		let send_desc_iter = buffer_tkn
198			.send_buff
199			.iter()
200			.map(|elem| (elem, elem.len(), virtq::DescF::empty()));
201		let recv_desc_iter = buffer_tkn
202			.recv_buff
203			.iter()
204			.map(|elem| (elem, elem.capacity(), virtq::DescF::WRITE));
205		let mut all_desc_iter =
206			send_desc_iter
207				.chain(recv_desc_iter)
208				.map(|(mem_descr, len, incomplete_flags)| {
209					Self::Descriptor::incomplete_desc(
210						DeviceAlloc
211							.phys_addr_from(mem_descr.as_ptr().cast_mut())
212							.as_u64()
213							.into(),
214						len.into(),
215						incomplete_flags | virtq::DescF::NEXT,
216					)
217				});
218
219		let mut last_desc = all_desc_iter
220			.next_back()
221			.ok_or(VirtqError::BufferNotSpecified)?;
222		*last_desc.flags_mut() -= virtq::DescF::NEXT;
223
224		Ok(all_desc_iter.chain([last_desc]))
225	}
226}

274	/// upon reuse.

280/// Public Interface for TransferToken
281impl<Descriptor> TransferToken<Descriptor> {
282	/// Returns the number of descritprors that will be placed in the queue.
283	/// This number can differ from the `BufferToken.num_descr()` function value
284	/// as indirect buffers only consume one descriptor in the queue, but can have
285	/// more descriptors that are accessible via the descriptor in the queue.
286	fn num_consuming_descr(&self) -> u16 {
287		if self.ctrl_desc.is_some() {
288			1
289		} else {
290			self.buff_tkn.num_descr()
291		}
292	}

295#[derive(Debug)]
296pub enum BufferElem {
297	#[allow(dead_code)]
298	Sized(Box<dyn Any + Send, DeviceAlloc>),
299	Vector(Vec<u8, DeviceAlloc>),
300}
301
302impl BufferElem {
303	// Returns the initialized length of the element. Assumes [Self::Sized] to
304	// be initialized, since the type of the object is erased and we cannot
305	// detect if the content is actually a [MaybeUninit]. However, this function
306	// should be only relevant for read buffer elements, which should not be uninit.
307	// If the element belongs to a write buffer, it is likely that [Self::capacity]
308	// is more appropriate.
309	pub fn len(&self) -> u32 {
310		match self {
311			BufferElem::Sized(sized) => mem::size_of_val(sized.as_ref()),
312			BufferElem::Vector(vec) => vec.len(),
313		}
314		.try_into()
315		.unwrap()
316	}
317
318	pub fn capacity(&self) -> u32 {
319		match self {
320			BufferElem::Sized(sized) => mem::size_of_val(sized.as_ref()),
321			BufferElem::Vector(vec) => vec.capacity(),
322		}
323		.try_into()
324		.unwrap()
325	}
326
327	pub fn as_ptr(&self) -> *const u8 {
328		match self {
329			BufferElem::Sized(sized) => ptr::from_ref(sized.as_ref()).cast::<u8>(),
330			BufferElem::Vector(vec) => vec.as_ptr(),
331		}
332	}
333}

335/// The struct represents buffers which are ready to be written or to be send.
336///
337/// BufferTokens can be written in two ways:
338/// * in one step via `BufferToken.write()
339///   * consumes BufferToken and returns a TransferToken
340/// * sequentially via `BufferToken.write_seq()
341///
342/// # Structure of the Token
343/// The token can potentially hold both a *send* and a *recv* buffer, but MUST hold
344/// one.
345/// The *send* buffer is the data the device will read during a transfer, the *recv* buffer
346/// is the data the device will write to during a transfer.
347///
348/// # What are Buffers
349/// A buffer represents multiple chunks of memory. Where each chunk can be of different size.
350/// The chunks are named descriptors in the following.
351///
352/// **For Example:**
353/// A buffer could consist of 3 descriptors:
354/// 1. First descriptor of 30 bytes
355/// 2. Second descriptor of 10 bytes
356/// 3. Third descriptor of 100 bytes
357///
358/// Each of these descriptors consumes one "element" of the
359/// respective virtqueue.
360/// The maximum number of descriptors per buffer is bounded by the size of the virtqueue.
361pub struct AvailBufferToken {
362	pub(crate) send_buff: SmallVec<[BufferElem; 2]>,
363	pub(crate) recv_buff: SmallVec<[BufferElem; 2]>,
364}
365
366pub(crate) struct UsedDeviceWritableBuffer {
367	elems: SmallVec<[BufferElem; 2]>,
368	remaining_written_len: u32,
369}
370
371impl UsedDeviceWritableBuffer {
372	/// # Safety
373	/// The type `T` of the [`MaybeUninit<T>`] that was provided for the [BufferElem] must have
374	/// been what the device will write at that portion of the buffer. This may not be the case if
375	/// the descriptor is used by the device for another type (e.g. in the case of merged buffers
376	/// in the network driver), as the objects are assumed to be initialized by this function and
377	/// it's undefined behavior to call [MaybeUninit::assume_init] when an object of the correct type
378	/// is not initialized.
379	#[allow(dead_code)]
380	pub unsafe fn pop_front_downcast<T>(&mut self) -> Option<Box<T, DeviceAlloc>>
381	where
382		T: Any,
383	{
384		if self.remaining_written_len < u32::try_from(size_of::<T>()).unwrap() {
385			return None;
386		}
387
388		// May panic, but we have written data remaining so there should always be an item
389		let elem = if self.elems.len() <= 2 {
390			self.elems.swap_remove(0)
391		} else {
392			self.elems.remove(0)
393		};
394
395		let BufferElem::Sized(sized) = elem else {
396			panic!("Attempted to pop elements in order different from insertion");
397		};
398
399		let cast = sized
400			.downcast::<MaybeUninit<T>>()
401			.expect("Attempted to downcast element to wrong type");
402
403		self.remaining_written_len -= u32::try_from(size_of::<T>()).unwrap();
404		Some(unsafe { cast.assume_init() })
405	}
406
407	pub fn pop_front_vec(&mut self) -> Option<Vec<u8, DeviceAlloc>> {
408		if self.elems.is_empty() {
409			return None;
410		}
411
412		let elem = if self.elems.len() <= 2 {
413			self.elems.swap_remove(0)
414		} else {
415			self.elems.remove(0)
416		};
417
418		let BufferElem::Vector(mut vector) = elem else {
419			panic!("Attempted to pop elements in order different from insertion");
420		};
421
422		let new_len = u32::min(
423			vector.capacity().try_into().unwrap(),
424			self.remaining_written_len,
425		);
426		self.remaining_written_len -= new_len;
427		unsafe { vector.set_len(new_len.try_into().unwrap()) };
428		Some(vector)
429	}
430
431	/// It is possible for devices to use descriptors for a type other than what they were meant.
432	/// (e.g. for a portion of the received frame in the network driver when [virtio::net::F::MRG_RXBUF] is negotiated).
433	/// In that case, we hand over the popped box directly with the used length.
434	///
435	/// We may not return a vector as its layout would be different and deallocation would not be correct
436	/// (see the information for [Box::into_non_null_with_allocator]).
437	#[allow(dead_code)]
438	pub fn pop_front_raw(&mut self) -> Option<(Box<dyn Any + Send, DeviceAlloc>, usize)> {
439		let elem = if self.elems.len() <= 2 {
440			self.elems.swap_remove(0)
441		} else {
442			self.elems.remove(0)
443		};
444
445		let BufferElem::Sized(sized) = elem else {
446			panic!("This function is meant for the Sized variant of the BufferElem enum.");
447		};
448
449		let capacity = u32::try_from(size_of_val(sized.as_ref())).unwrap();
450		let len = u32::min(capacity, self.remaining_written_len);
451		self.remaining_written_len -= len;
452		Some((sized, len.try_into().unwrap()))
453	}
454}

482// Public interface of BufferToken
483impl AvailBufferToken {
484	/// **Parameters**
485	/// * send: The slices that will make up the elements of the driver-writable buffer.
486	/// * recv: The slices that will make up the elements of the device-writable buffer.
487	///
488	/// **Reasons for Failure:**
489	/// * Both `send` and `recv` are empty, which is not allowed by Virtio.
490	///
491	/// * If one wants to have a structure in the style of:
492	/// ```
493	/// struct send_recv_struct {
494	///     // send_part: ...
495	///     // recv_part: ...
496	/// }
497	/// ```
498	/// they must split the structure after the send part and provide the respective part via the send argument and the respective other
499	/// part via the recv argument.
500	pub fn new(
501		send_buff: SmallVec<[BufferElem; 2]>,
502		recv_buff: SmallVec<[BufferElem; 2]>,
503	) -> Result<Self, VirtqError> {
504		if send_buff.is_empty() && recv_buff.is_empty() {
505			return Err(VirtqError::BufferNotSpecified);
506		}
507
508		Ok(Self {
509			send_buff,
510			recv_buff,
511		})
512	}
513}

528	/// As a result indirect descriptors result in a single descriptor consumption in the actual queue.
529	#[expect(dead_code)]
530	Indirect,
531}

533mod index_alloc {
534	use alloc::boxed::Box;

536	/// This type allows allocating indices.
537	///
538	/// The indices can be used as descriptor IDs.
539	pub struct IndexAlloc {
540		/// Zero bits are available.
541		bits: Box<[usize]>,
542	}
543
544	const USIZE_BITS: usize = usize::BITS as usize;
545	const LEADING_ONE: usize = 1 << (usize::BITS - 1);
546
547	impl IndexAlloc {
548		pub fn new(len: usize) -> Self {
549			let usizes = len.div_ceil(USIZE_BITS);
550			let extra_bits = len % USIZE_BITS;
551
552			let mut bits = vec![0; usizes].into_boxed_slice();
553
554			if extra_bits != 0 {
555				*bits.last_mut().unwrap() = usize::MAX >> extra_bits;
556			}
557
558			Self { bits }
559		}
560
561		#[inline]
562		pub fn allocate(&mut self) -> Option<usize> {
563			for (word_index, word) in self.bits.iter_mut().enumerate() {
564				let bit = word.leading_ones();
565
566				if bit >= usize::BITS {
567					continue;
568				}
569
570				let mask = LEADING_ONE >> bit;
571				*word |= mask;
572				let index = word_index * USIZE_BITS + usize::try_from(bit).unwrap();
573				return Some(index);
574			}
575
576			None
577		}
578
579		#[inline]
580		pub unsafe fn deallocate(&mut self, index: usize) {
581			let word_index = index / USIZE_BITS;
582			let bit = index % USIZE_BITS;
583			let mask = LEADING_ONE >> bit;
584
585			debug_assert!(self.bits[word_index] & mask == mask);
586			unsafe {
587				*self.bits.get_unchecked_mut(word_index) &= !mask;
588			}
589		}
590	}

658	// External Error Handling for users of the virtqueue.
659	#[derive(Error, Debug)]
660	pub enum VirtqError {
661		#[error("Virtq failure due to unknown reasons!")]
662		General,
663
664		/// Call to create a BufferToken or TransferToken without
665		/// any buffers to be inserted
666		#[error("Virtq detected creation of Token, without a BuffSpec")]
667		BufferNotSpecified,
668

670		/// is not known to the device and hence can not be used
