38/// Maps a given device specific pci configuration structure and
39/// returns a static reference to it.
40pub fn map_dev_cfg<T>(cap: &PciCap) -> Option<&'static mut T> {
41	if cap.cap.cfg_type != CapCfgType::Device {
42		error!("Capability of device config has wrong id. Mapping not possible...");
43		return None;
44	};
45
46	if cap.bar_len() < cap.len() + cap.offset() {
47		error!(
48			"Device config of device {:x}, does not fit into memory specified by bar!",
49			cap.dev_id(),
50		);
51		return None;
52	}
53
54	// Drivers MAY do this check. See Virtio specification v1.1. - 4.1.4.1
55	if cap.len() < u64::try_from(mem::size_of::<T>()).unwrap() {
56		error!(
57			"Device specific config from device {:x}, does not represent actual structure specified by the standard!",
58			cap.dev_id()
59		);
60		return None;
61	}
62
63	let virt_addr_raw = cap.bar_addr() + cap.offset();
64
65	// Create mutable reference to the PCI structure in PCI memory
66	let dev_cfg: &'static mut T =
67		unsafe { &mut *(ptr::with_exposed_provenance_mut(virt_addr_raw.try_into().unwrap())) };
68
69	Some(dev_cfg)
70}

72/// Virtio's PCI capabilities structure.
73/// See Virtio specification v.1.1 - 4.1.4
74///
75/// Indicating: Where the capability field is mapped in memory and
76/// Which id (sometimes also indicates priority for multiple
77/// capabilities of same type) it holds.
78///
79/// This structure does NOT represent the structure in the standard,
80/// as it is not directly mapped into address space from PCI device
81/// configuration space.
82/// Therefore the struct only contains necessary information to map
83/// corresponding config type into address space.
84#[derive(Clone)]
85pub struct PciCap {
86	bar: PciBar,
87	dev_id: u16,
88	cap: CapData,
89}

92	pub fn offset(&self) -> u64 {
93		self.cap.offset.to_ne()
94	}

96	pub fn len(&self) -> u64 {
97		self.cap.length.to_ne()
98	}
99
100	pub fn bar_len(&self) -> u64 {
101		self.bar.length
102	}
103
104	pub fn bar_addr(&self) -> u64 {
105		self.bar.mem_addr
106	}
107
108	pub fn dev_id(&self) -> u16 {
109		self.dev_id
110	}

112	/// Returns a reference to the actual structure inside the PCI devices memory space.
113	fn map_common_cfg(&self) -> Option<VolatileRef<'static, CommonCfg>> {
114		if self.bar.length < self.len() + self.offset() {
115			error!(
116				"Common config of the capability with id {} of device {:x} does not fit into memory specified by bar {:x}!",
117				self.cap.id, self.dev_id, self.bar.index
118			);
119			return None;
120		}
121
122		// Drivers MAY do this check. See Virtio specification v1.1. - 4.1.4.1
123		if self.len() < u64::try_from(mem::size_of::<CommonCfg>()).unwrap() {
124			error!(
125				"Common config of with id {}, does not represent actual structure specified by the standard!",
126				self.cap.id
127			);
128			return None;
129		}
130
131		let virt_addr_raw = self.bar.mem_addr + self.offset();
132		let ptr = NonNull::new(ptr::with_exposed_provenance_mut::<CommonCfg>(
133			virt_addr_raw.try_into().unwrap(),
134		))
135		.unwrap();
136
137		// Create mutable reference to the PCI structure in PCI memory
138		let com_cfg_raw = unsafe { VolatileRef::new(ptr) };
139
140		Some(com_cfg_raw)
141	}
142
143	fn map_isr_status(&self) -> Option<VolatileRef<'static, IsrStatusRaw>> {
144		if self.bar.length < self.len() + self.offset() {
145			error!(
146				"ISR status config with id {} of device {:x}, does not fit into memory specified by bar {:x}!",
147				self.cap.id, self.dev_id, self.bar.index
148			);
149			return None;
150		}
151
152		let virt_addr_raw = self.bar.mem_addr + self.offset();
153		let ptr = NonNull::new(ptr::with_exposed_provenance_mut::<IsrStatusRaw>(
154			virt_addr_raw.try_into().unwrap(),
155		))
156		.unwrap();
157
158		// Create mutable reference to the PCI structure in the devices memory area
159		let isr_stat_raw = unsafe { VolatileRef::new(ptr) };
160
161		Some(isr_stat_raw)
162	}

342			.update(|s| s | DeviceStatus::FEATURES_OK);

345	/// In order to correctly check feature negotiaten, this function
346	/// MUST be called after [self.features_ok()](ComCfg::features_ok()) in order to check
347	/// if features have been accepted by the device after negotiation.
348	///
349	/// Re-reads device status to ensure the FEATURES_OK bit is still set:
350	/// otherwise, the device does not support our subset of features and the device is unusable.
351	pub fn check_features(&self) -> bool {
352		memory_barrier();
353		self.com_cfg
354			.as_ptr()
355			.device_status()
356			.read()
357			.contains(DeviceStatus::FEATURES_OK)
358	}

361	///
362	/// After this call, the device is "live"!
363	pub fn drv_ok(&mut self) {
364		memory_barrier();
365		self.com_cfg
366			.as_mut_ptr()
367			.device_status()
368			.update(|s| s | DeviceStatus::DRIVER_OK);
369	}
370
371	/// Returns the features offered by the device.
372	pub fn dev_features(&mut self) -> virtio::F {
373		let com_cfg = self.com_cfg.as_mut_ptr();
374		let device_feature_select = com_cfg.device_feature_select();
375		let device_feature = com_cfg.device_feature();
376
377		// Indicate device to show high 32 bits in device_feature field.
378		// See Virtio specification v1.1. - 4.1.4.3
379		memory_barrier();
380		device_feature_select.write(1.into());
381		memory_barrier();
382
383		// read high 32 bits of device features
384		let mut device_features = u64::from(device_feature.read().to_ne()) << 32;
385
386		// Indicate device to show low 32 bits in device_feature field.
387		// See Virtio specification v1.1. - 4.1.4.3
388		device_feature_select.write(0.into());
389		memory_barrier();
390
391		// read low 32 bits of device features
392		device_features |= u64::from(device_feature.read().to_ne());
393
394		virtio::F::from_bits_retain(u128::from(device_features).into())
395	}
396
397	/// Write selected features into driver_select field.
398	pub fn set_drv_features(&mut self, features: virtio::F) {
399		let features = features.bits().to_ne() as u64;
400		let com_cfg = self.com_cfg.as_mut_ptr();
401		let driver_feature_select = com_cfg.driver_feature_select();
402		let driver_feature = com_cfg.driver_feature();
403
404		let high: u32 = (features >> 32) as u32;
405		let low: u32 = features as u32;
406
407		// Indicate to device that driver_features field shows low 32 bits.
408		// See Virtio specification v1.1. - 4.1.4.3
409		memory_barrier();
410		driver_feature_select.write(0.into());
411		memory_barrier();
412
413		// write low 32 bits of device features
414		driver_feature.write(low.into());
415
416		// Indicate to device that driver_features field shows high 32 bits.
417		// See Virtio specification v1.1. - 4.1.4.3
418		driver_feature_select.write(1.into());
419		memory_barrier();
420
421		// write high 32 bits of device features
422		driver_feature.write(high.into());
423	}

426/// Notification Structure to handle virtqueue notification settings.
427/// See Virtio specification v1.1 - 4.1.4.4
428pub struct NotifCfg {
429	/// Start addr, from where the notification addresses for the virtqueues are computed
430	base_addr: u64,
431	notify_off_multiplier: u32,
432	/// Preferences of the device for this config. From 1 (highest) to 2^7-1 (lowest)
433	rank: u8,
434	/// defines the maximum size of the notification space, starting from base_addr.
435	length: u64,
436}
437
438impl NotifCfg {
439	fn new(cap: &PciCap) -> Option<Self> {
440		if cap.bar.length < cap.len() + cap.offset() {
441			error!(
442				"Notification config with id {} of device {:x}, does not fit into memory specified by bar {:x}!",
443				cap.cap.id, cap.dev_id, cap.bar.index
444			);
445			return None;
446		}
447
448		let notify_off_multiplier = cap.cap.notify_off_multiplier?.to_ne();
449
450		// define base memory address from which the actual Queue Notify address can be derived via
451		// base_addr + queue_notify_off * notify_off_multiplier.
452		//
453		// Where queue_notify_off is taken from the respective common configuration struct.
454		// See Virtio specification v1.1. - 4.1.4.4
455		//
456		// Base address here already includes offset!
457		let base_addr = cap.bar.mem_addr + cap.offset();
458
459		Some(NotifCfg {
460			base_addr,
461			notify_off_multiplier,
462			rank: cap.cap.id,
463			length: cap.len(),
464		})
465	}
466
467	pub fn notification_location(&self, vq_cfg_handler: &mut VqCfgHandler<'_>) -> *mut le32 {
468		let addend = u32::from(vq_cfg_handler.notif_off()) * self.notify_off_multiplier;
469		let addr = self.base_addr + u64::from(addend);
470		ptr::with_exposed_provenance_mut(addr.try_into().unwrap())
471	}
472}

474/// Control structure, allowing to notify a device via PCI bus.
475/// Typically hold by a virtqueue.
476pub struct NotifCtrl {
477	/// Indicates if VIRTIO_F_NOTIFICATION_DATA has been negotiated
478	f_notif_data: bool,
479	/// Where to write notification
480	notif_addr: *mut le32,
481}
482
483// FIXME: make `notif_addr` implement `Send` instead
484unsafe impl Send for NotifCtrl {}
485
486impl NotifCtrl {
487	/// Returns a new controller. By default MSI-X capabilities and VIRTIO_F_NOTIFICATION_DATA
488	/// are disabled.
489	pub fn new(notif_addr: *mut le32) -> Self {
490		NotifCtrl {
491			f_notif_data: false,
492			notif_addr,
493		}
494	}
495
496	/// Enables VIRTIO_F_NOTIFICATION_DATA. This changes which data is provided to the device. ONLY a good idea if Feature has been negotiated.
497	pub fn enable_notif_data(&mut self) {
498		self.f_notif_data = true;
499	}
500
501	pub fn notify_dev(&self, data: NotificationData) {
502		// See Virtio specification v.1.1. - 4.1.5.2
503		// Depending in the feature negotiation, we write either only the
504		// virtqueue index or the index and the next position inside the queue.
505
506		if self.f_notif_data {
507			unsafe {
508				self.notif_addr.write_volatile(data.into_bits());
509			}
510		} else {
511			unsafe {
512				self.notif_addr
513					.cast::<le16>()
514					.write_volatile(data.vqn().into());
515			}
516		}
517	}
518}

551/// Shared memory configuration structure of Virtio PCI devices.
552/// See Virtio specification v1.1. - 4.1.4.7
553///
554/// Each shared memory region is defined via a single shared
555/// memory structure. Each region is identified by an id indicated
556/// via the capability.id field of PciCapRaw.
557///
558/// The shared memory region is defined via a PciCap64 structure.
559/// See Virtio specification v.1.1 - 4.1.4 for structure.
560///
561// Only used for capabilities that require offsets or lengths
562// larger than 4GB.
563// #[repr(C)]
564// struct PciCap64 {
565//    pci_cap: PciCap,
566//    offset_hi: u32,
567//    length_hi: u32
568pub struct ShMemCfg {
569	mem_addr: u64,
570	length: u64,
571	sh_mem: ShMem,
572	/// Shared memory regions are identified via an ID
573	/// See Virtio specification v1.1. - 4.1.4.7
574	id: u8,
575}
576
577impl ShMemCfg {
578	fn new(cap: &PciCap) -> Option<Self> {
579		if cap.bar.length < cap.len() + cap.offset() {
580			error!(
581				"Shared memory config of with id {} of device {:x}, does not fit into memory specified by bar {:x}!",
582				cap.cap.id, cap.dev_id, cap.bar.index
583			);
584			return None;
585		}
586
587		let offset = cap.cap.offset.to_ne();
588		let length = cap.cap.length.to_ne();
589
590		let virt_addr_raw = cap.bar.mem_addr + offset;
591		let raw_ptr = ptr::with_exposed_provenance_mut::<u8>(virt_addr_raw.try_into().unwrap());
592
593		// Zero initialize shared memory area
594		unsafe {
595			for i in 0..usize::try_from(length).unwrap() {
596				*(raw_ptr.add(i)) = 0;
597			}
598		};
599
600		// Currently in place in order to ensure a safe cast below
601		// "len: cap.bar.length as usize"
602		// In order to remove this assert a safe conversion from
603		// kernel PciBar struct into usize must be made
604		assert!(mem::size_of::<usize>() == 8);
605
606		Some(ShMemCfg {
607			mem_addr: virt_addr_raw,
608			length: cap.len(),
609			sh_mem: ShMem {
610				ptr: raw_ptr,
611				len: cap.bar.length as usize,
612			},
613			id: cap.cap.id,
614		})
615	}
616}

618/// Defines a shared memory locate at location ptr with a length of len.
619/// The shared memories Drop implementation does not dealloc the memory
620/// behind the pointer but sets it to zero, to prevent leakage of data.
621struct ShMem {
622	ptr: *mut u8,
623	len: usize,
624}
625
626impl core::ops::Deref for ShMem {
627	type Target = [u8];
628
629	fn deref(&self) -> &[u8] {
630		unsafe { core::slice::from_raw_parts(self.ptr, self.len) }
631	}
632}
633
634impl core::ops::DerefMut for ShMem {
635	fn deref_mut(&mut self) -> &mut [u8] {
636		unsafe { core::slice::from_raw_parts_mut(self.ptr, self.len) }
637	}
638}
639
640// Upon drop the shared memory region is "deleted" with zeros.
641impl Drop for ShMem {
642	fn drop(&mut self) {
643		for i in 0..self.len {
644			unsafe {
645				*(self.ptr.add(i)) = 0;
646			}
647		}
648	}
649}

672/// Reads all PCI capabilities, starting at the capabilities list pointer from the
673/// PCI device.
674///
675/// Returns ONLY Virtio specific capabilities, which allow to locate the actual capability
676/// structures inside the memory areas, indicated by the BaseAddressRegisters (BAR's).
677fn read_caps(device: &PciDevice<PciConfigRegion>) -> Result<Vec<PciCap>, PciError> {
678	let device_id = device.device_id();
679
680	let capabilities = device
681		.capabilities()
682		.unwrap()
683		.filter_map(|capability| match capability {
684			PciCapability::Vendor(capability) => Some(capability),
685			_ => None,
686		})
687		.map(|addr| CapData::read(addr, device.access()).unwrap())
688		.filter(|cap| cap.cfg_type != CapCfgType::Pci)
689		.map(|cap| {
690			let slot = cap.bar;
691			let (addr, size) = device.memory_map_bar(slot, true).unwrap();
692			PciCap {
693				bar: VirtioPciBar::new(slot, addr.as_u64(), size.try_into().unwrap()),
694				dev_id: device_id,
695				cap,
696			}
697		})
698		.collect::<Vec<_>>();
699
700	if capabilities.is_empty() {
701		error!("No virtio capability found for device {device_id:x}");
702		Err(PciError::NoVirtioCaps(device_id))
703	} else {
704		Ok(capabilities)
705	}
706}

708pub(crate) fn map_caps(device: &PciDevice<PciConfigRegion>) -> Result<UniCapsColl, VirtioError> {
709	let device_id = device.device_id();
710
711	// In case caplist pointer is not used, abort as it is essential
712	if !device.status().has_capability_list() {
713		error!("Found virtio device without capability list. Aborting!");
714		return Err(VirtioError::FromPci(PciError::NoCapPtr(device_id)));
715	}
716
717	// Get list of PciCaps pointing to capabilities
718	let cap_list = match read_caps(device) {
719		Ok(list) => list,
720		Err(pci_error) => return Err(VirtioError::FromPci(pci_error)),
721	};
722
723	let mut com_cfg = None;
724	let mut notif_cfg = None;
725	let mut isr_cfg = None;
726	let mut sh_mem_cfg_list = Vec::new();
727	let mut dev_cfg_list = Vec::new();
728	// Map Caps in virtual memory
729	for pci_cap in cap_list {
730		match pci_cap.cap.cfg_type {
731			CapCfgType::Common => {
732				if com_cfg.is_none() {
733					match pci_cap.map_common_cfg() {
734						Some(cap) => com_cfg = Some(ComCfg::new(cap, pci_cap.cap.id)),
735						None => error!(
736							"Common config capability with id {}, of device {:x}, could not be mapped!",
737							pci_cap.cap.id, device_id
738						),
739					}
740				}
741			}
742			CapCfgType::Notify => {
743				if notif_cfg.is_none() {
744					match NotifCfg::new(&pci_cap) {
745						Some(notif) => notif_cfg = Some(notif),
746						None => error!(
747							"Notification config capability with id {}, of device {device_id:x} could not be used!",
748							pci_cap.cap.id
749						),
750					}
751				}
752			}
753			CapCfgType::Isr => {
754				if isr_cfg.is_none() {
755					match pci_cap.map_isr_status() {
756						Some(isr_stat) => isr_cfg = Some(IsrStatus::new(isr_stat, pci_cap.cap.id)),
757						None => error!(
758							"ISR status config capability with id {}, of device {device_id:x} could not be used!",
759							pci_cap.cap.id
760						),
761					}
762				}
763			}
764			CapCfgType::SharedMemory => match ShMemCfg::new(&pci_cap) {
765				Some(sh_mem) => sh_mem_cfg_list.push(sh_mem),
766				None => error!(
767					"Shared Memory config capability with id {}, of device {device_id:x} could not be used!",
768					pci_cap.cap.id,
769				),
770			},
771			CapCfgType::Device => dev_cfg_list.push(pci_cap),
772
773			// PCI's configuration space is allowed to hold other structures, which are not virtio specific and are therefore ignored
774			// in the following
775			_ => continue,
776		}
777	}
778
779	Ok(UniCapsColl {
780		com_cfg: com_cfg.ok_or(VirtioError::NoComCfg(device_id))?,
781		notif_cfg: notif_cfg.ok_or(VirtioError::NoNotifCfg(device_id))?,
782		isr_cfg: isr_cfg.ok_or(VirtioError::NoIsrCfg(device_id))?,
783		sh_mem_cfg_list,
784		dev_cfg_list,
785	})
786}

788/// Checks existing drivers for support of given device. Upon match, provides
789/// driver with a [`PciDevice<PciConfigRegion>`] reference, allowing access to the capabilities
790/// list of the given device through [map_caps].
791pub(crate) fn init_device(
792	device: &PciDevice<PciConfigRegion>,
793) -> Result<VirtioDriver, DriverError> {
794	let device_id = device.device_id();
795
796	if device_id < 0x1040 {
797		warn!(
798			"Legacy/transitional Virtio device, with id: {device_id:#x} is NOT supported, skipping!"
799		);
800
801		// Return Driver error inidacting device is not supported
802		return Err(DriverError::InitVirtioDevFail(
803			VirtioError::DevNotSupported(device_id),
804		));
805	}
806
807	let id = virtio::Id::from(u8::try_from(device_id - 0x1040).unwrap());
808
809	match id {
810		#[cfg(all(
811			not(all(target_arch = "x86_64", feature = "rtl8139")),
812			any(feature = "tcp", feature = "udp")
813		))]
814		virtio::Id::Net => match VirtioNetDriver::init(device) {
815			Ok(virt_net_drv) => {
816				info!("Virtio network driver initialized.");
817
818				let irq = device.get_irq().unwrap();
819				crate::arch::interrupts::add_irq_name(irq, "virtio");
820				info!("Virtio interrupt handler at line {irq}");
821
822				Ok(VirtioDriver::Network(virt_net_drv))
823			}
824			Err(virtio_error) => {
825				error!(
826					"Virtio networkd driver could not be initialized with device: {device_id:x}"
827				);
828				Err(DriverError::InitVirtioDevFail(virtio_error))
829			}
830		},
831		#[cfg(feature = "vsock")]
832		virtio::Id::Vsock => match VirtioVsockDriver::init(device) {
833			Ok(virt_sock_drv) => {
834				info!("Virtio sock driver initialized.");
835
836				let irq = device.get_irq().unwrap();
837				crate::arch::interrupts::add_irq_name(irq, "virtio");
838				info!("Virtio interrupt handler at line {irq}");
839
840				Ok(VirtioDriver::Vsock(Box::new(virt_sock_drv)))
841			}
842			Err(virtio_error) => {
843				error!("Virtio sock driver could not be initialized with device: {device_id:x}");
844				Err(DriverError::InitVirtioDevFail(virtio_error))
845			}
846		},
847		#[cfg(feature = "fuse")]
848		virtio::Id::Fs => {
849			// TODO: check subclass
850			// TODO: proper error handling on driver creation fail
851			match VirtioFsDriver::init(device) {
852				Ok(virt_fs_drv) => {
853					info!("Virtio filesystem driver initialized.");
854					Ok(VirtioDriver::FileSystem(virt_fs_drv))
855				}
856				Err(virtio_error) => {
857					error!(
858						"Virtio filesystem driver could not be initialized with device: {device_id:x}"
859					);
860					Err(DriverError::InitVirtioDevFail(virtio_error))
861				}
862			}
863		}
864		id => {
865			warn!("Virtio device {id:?} is not supported, skipping!");
866
867			// Return Driver error inidacting device is not supported
868			Err(DriverError::InitVirtioDevFail(
869				VirtioError::DevNotSupported(device_id),
870			))
871		}
872	}
873}
